11: */            
12: double read_data(ifstream& in)
13: {  double highest;
15:    double next;
16:    if (in >> next) highest = next;
18:    else return 0;
20: 
21:    while (in >> next)
22:      if (next > highest) highest = next;
27:    return highest;
28: }
