16:    bool more = true;
17:    while (more)
18:    {
19:       char op = cin.peek();
20:       if (op == '+' || op == '-')
21:       {
22:          cin.get();
23:          int value = term_value();

64:    {
65:       cin.get();
66:       result = expression_value();
67:       cin.get(); // read ")"
68:    }

70:    {
71:       while (isdigit(c))
72:       {
73:          result = 10 * result + c - '0';
74:          cin.get();
75:          c = cin.peek();
76:       }
