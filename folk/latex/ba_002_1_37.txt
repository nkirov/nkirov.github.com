%Гено по сглява отива
%Хороводна (пее се на засевки)
%Джиджев:хороводни

%%(пее се на засевки)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Гено по сглява отива,

\uu{Гено льо, гюлчи цъфнало,}

по сглява на друго село,

Яна го през плет гледаше,

Яна на Гена думаше:
\kray

-- Бре, къде, Гено, ша идеш:

ако по раб'та отиваш,

да идеш, пак да се върнеш;

ако по сгледа отиваш,

да идеш, да се не върнеш?!

~~
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
