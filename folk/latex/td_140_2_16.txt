%Вайвода, вайвода
%На хоро
%Джиджев:хороводни

\begin{multicols}{2}
Вайвода, вайвода,

посъбрал вайвода (2)

от три села моми, (2)

от четери момци (2)

жетва да му женат. (2)

Женали, женали

ден ми до пладнина,
\kray

седнали да пладнят,
%%добре е да се сложи ударение върху "я" на "пладнЯт"

вайвода хни дума:

-- Моми, малки моми,

вий като женахте,

женахте й седнахте,

квеж да ми приляга

царюм за царица,

за бяла кадъна,

за тенка гъркиня?

Сички долу ничат,

ничат, сълзи ронят,

гюзел бяло Раде,
%%в ръкописа го разчитам като "бяло"

той долу не ничи,

той сълзи не рони,

той вайвода дума:

-- Вайвода, вайвода,

яж да ти приличам

царюм за царица,

за бяла кадъна,

ако ми направиш

от божур папуце,

от невене мести,

от бял зъмбак риза!
%%в ръкописа "от бял зъмбак риза" - не го пипам, че става неизпеваемо
\end{multicols}
