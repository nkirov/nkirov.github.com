%Иван Нидельо думаше
%Лазарска (хороводна) 
%Джиджев:лазарски

\begin{multicols}{2}
Иван Нидельо думаше: 

-- Нидельо, моме хубава,

пустата й, вашта махала,
%%"вашта махала" вм. "ваша махла" - по записа; за сричките

що има моми хубави!

Ниделя Иван думаше:

-- Байно льо, бачо Иване,

чи коя й мома хубава?

Иван Нидельо думаше:
\kray

-- Нидельо, моме хубава,

наточи вино в бял бакър,

чи са над бакър наведи --

която видиш в бакъра,

тя йе мома най-хубава,

аз нея мома ще взема!

~~
\end{multicols}

\editor{Колебание: Недельо -- Нидельо.}
