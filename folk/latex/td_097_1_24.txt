%Илено, Иленке, снахо най-малка
%На жътва (сутрин и по друго време)
%Джиджев:жътварски

\begin{multicols}{2}
Илено, Иленке, снахо най-малка,

най-малка, най-малка и най-хубава,

я викни, я викни, снахо, па запей: 

из едно, из едно гърло два гласа,
%%добавям началното повторение до края
\kray

из едни, из едни уста две думи!

-- Да викна, да викна, тейно, ни мога,

че имам, че имам любе хайдутин,

че той ми, че той ми гласа познава,

че той при, че той при мене ще доди,

мене със, мене със него ще земи,

а тебе, а тебе глава ще земи! 
\end{multicols}

\bel{В началото на първото изпяване си бемол не е утвърден.}
