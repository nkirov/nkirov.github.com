%Каква беше таз година
%На арман
%Джиджев:жътварски

%% (трудова, но не жетварска??? - Г.Г.)

\begin{multicols}{2}
Каква беше, \uu{лельо,} таз година,

таз година, дет се мина,
%%този и долния стих в ръкописа са два стиха надолу
%%пренареждам ги според записа

дет се мина и замина, (2)

таз година трасовища,
%%чува се: трасовиша (трасовита, от тресе) НК

таз година, двайс' и осма
%%двайс вм. двайсе - по записа, за сричките

дет си мина и замина,
%%не се пее тук НК

та избута дюкените,
%%по селата, из градищата,
%%стихът не е изпят

дюкените с ергените! (2)

Не ми й жалко, не ми й мило,
%%в муз. запис: "Не ми жалко, не ми мило"

не ми й жалко за дюкени,

най ми й жалко за ергени:
%%в муз. запис - "жалко"
\kray

дюкените направиха,

но ергени ни станаха --

засипа ги зимнотръсът,

засипа ги, затрупа ги!

Мойто либе, й то там беше,

дет го любих три години,

три години й половина!

~~
\end{multicols}
