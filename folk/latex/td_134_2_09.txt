%Я изгрей, слънчо, я изгрей
%Сватбарска (``когато бръснат зетя'')
%Джиджев:сватбарски

\begin{multicols}{2}
Я изгрей, слънчо, я изгрей

във момкови равни дворове!

Дор две са слънца изгреле

във момини равни дворове,

дор две са мъгли паднале.
%%следващите два стиха в записа са други: Йенка вм. Иванчо и т.н., но не могат да се съшият в тази песен, оставам както намерих

Мама на Йенка думаше:
\kray

-- Кога са, чедо, находи

на твойта герджик примяна,

на твойта хубост и младост,

та де я, чедо, намери?

-- Найдох я, мамо, найдох я --

на бели Дунав переше,

бели пешкири белеше,

за сватба, мамо, готвеше!

~~ 
\end{multicols}

\sing{Пяла Гела Желева Димитрова}
\village{с. Мустрак}
\area{Свиленградско}
\year{1926}
\re
