%Скрийте ма, брайне, скрийте ма
%Сватбарска (``га наближат портите на булката'') 
%Джиджев:сватбарски
%Сватбарска (га наближават портите на булката) 

\begin{multicols}{2}
-- Скрийте ма, брайне, скрийте ма

във вашта равна градина!
%%"вашта" вм. "ваща" - като в нотите, същото е, по-четивно е

-- Не бива, сестро, не бива,

сватове фъф д'ор дойдоха!
\end{multicols}

%%допълнение от тефтера: 
\bel{Не си спомня да са пяли и когато извеждат булката, 
когато [булката] се прощава, когато калесват кума. 
Също не пеят и когато бръснат младоженека.}

\sing{Пяла Пенка Петрова Славчева}
\village{с. Поляново}
\area{Харманлийско}
\year{1901}
\re
