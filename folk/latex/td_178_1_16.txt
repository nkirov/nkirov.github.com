\begin{multicols}{2}
На бял камък седи,

бяло лице мие -- 

колкото го мие,

дваж по-бяло става;

черни очи стреля --

колкото ги стреля, (2)

дваж по-черни стават;

Руси коси реши --

колкото ги реши, (2)

дваж по-руси стават!

То да бе лебеди,

то биха фръкнали,

\rows
{то да бе снегове, }
{биха се стопили!}
\end{multicols}
