%(пее се на засевки вкъщи)

%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
\rows
{Иван на Станка думаше:}
{-- Станки мо, беним балдъзо,}

\rows
{хубаво мети дворове,}
{кака ти по тях ша ходи,}
%%вместо "че кака ти" - според записа и за да може да се пее
%%съкращавам eдно двустишие - няма го в записа, възникнало е като комбинация от предното и следващото

\rows
{да си чехлите не окаля,}
{да си полите не оваля!}

\rows
{-- Станки мо, беним балдъзо,}
{кака ти зема, не зема,}

\rows
{кака ти зема, не зема,}
{ала ща тебе да зема,}

\rows
{че ти си, Станке, по-бяла,}
{по-бяла и по-червена,}

\rows
{по-бяла и по-червена,}
{по-тънка и по-висока!}
\end{multicols}
