%Добра йе змийка любила
%Сватбарско хоро (на меденците)
%Джиджев:сватбарски

\begin{multicols}{2}
Добра йе змийка любила,

майка си ни е казала,

майка хи да я лекува, 

лекува, лек да \`{и} търси; 
%%възстановявам стиха от записа

най я майка \`{и} оглави

за Никола болярина.

Главили, каул правили

малко главене да ходят --

година й польвина.

Дойде година и половина
\kray

Никол' си сватба повдигна, 
%%стихът отсъства в машинописа

калеса много сватове --

педесе коля сватове, 

тръгнали булче да вземат.

Вървели, що са вървели,

до сухо дърво допрели,

сухо хми дърво продума:

-- Вие, сватове, кумове,

къде сте толко тръгнали,

булче ви Добра загина!

Всичките вяра фанало,

Никола вяра не фана,

той сбута конче храняно, 
%%предпочитам "сбута"

та у Добрини отиде.

Добрини деветь братчета

на Добра сандък ковяха,

Добрини девет снашици

на Добря пряпол нижеха,

песни си глася виеха. 
%%коментар на Джиджев в ръкописа "(т.е. плачеха)"
\end{multicols}

\sing{Пяла Здравка Димитрова Делчева (Попконстантинова)}
\village{с. Любеново}
\area{Хасковско}
\year{1901}
\re
