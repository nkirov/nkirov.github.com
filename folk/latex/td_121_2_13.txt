%Мамо мо, да опустеят
%Хороводна (``припявка, по сватби и годежи'')
%Джиджев:сватбарски

\begin{multicols}{2}
Мамо мо, да опустеят, (2)

\uu{льох, льох, лили льох,}

\uu{льох, лили льох,}

\uu{лили, лили, лили льох,}

\uu{льох, лили льох,}

тез пусти вакли овчари -- (2)

изсякоха ни горъта (2)

за хурки и за кавали. (2)

Хурката й за бяла Радка, (2)

кавала -- за гирджик Стояна! (2)
\end{multicols}

\ed{Гирджик -- диалектна форма от ``герджик'' тур. 'спретнат, напет, гиздав'.}
