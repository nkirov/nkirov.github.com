%Ни сме знаяли
%На сватба (``на каниски'')
%Джиджев:сватбарски
%На сватба (на каниски, когато отиват да канят кума)

\begin{multicols}{2}
Ни сме знаяли, 

драги ми куме,

ни сме знаяли,

че щиш кум да си 

\uu{драги куме ле,}

че щиш кум да си 
%%чува се рефрен от тук до края: драги куми ле НК 

кум да кумуваш, (2)

да ти извием 
%%"да" вм. "жа" - по записа, в името на смисъла

вити краваи, 

вити-превити, 
%%стихът не е изпят

и на краваи -- 

златни башчийки, 
\kray

и на башчийки -- 

златни ябълки! 

~~ 
\end{multicols}

\ed{Башчийки -- вер. от ``бахча'' тур. 'зеленчукова градина'.}

\sing{Пяла Милка Неделчева Вълканова}
\village{с. Борец}
\area{Пловдивско}
\year{1909}
\rec
