%Думаше, Стоян мама си думаше
%На жътва (сутрин в тъмно)
%Джиджев:жътварски

\begin{multicols}{2}
%%Г.Г. - пренареждам, слагам отпяването отзад
%%пренареждам според стиховата организация - интонационната е друга

Ей, думаше Стоян, мама си думаше, 

ей, думаше: -- Я, ми дай, мамо, ножчито, 

ей, ножчито, във сърце да са убода, 
%%във вм. в - по записа, в името на сричките

ей, убода зарад тва бяло момиче, 
%%тва вм. това - по записа, в името на сричките

ей, момиче: в лицето -- бяла [х]артия,
\kray
%%в записа наместо този стих се пее "беличко билоликичко, ей, белоликичко"
%%оставям варианта от ръкописа, той е в логиката на проължението

ей, [х]артия, вежди му гайтан копринен, 

ей, копринен, уста му чаша сребърна, 

ей, сребърна, коса му сиви смугови, 

ей, смугови! А[з] в сандък да я затвора, 

ей, затвора, на мама да я изпрата, 

ей, изпрата, на мама, ощи на тати, 

ей, на тати -- да видат мама и тати, 

ей, и тати, какво си либи аз либа, 

ей, аз либа: беличко белоликичко, 

ей, белоликичко, чърничко, чърноокичко!
%%добавям веднъж "ей чърнокичко" - за стиха
\end{multicols}

\year{1902}

\area{Пловдивско}
\rec
