%Нивясто, обзаложи се нивяста
%Жътварска (вечер, по икиндия)
%Джиджев:жътварски

%%при първа коректура сигнатурата е 96-1-32 (поправено на 32), 
%%но пак не съвпада  -- вярното е 96-2-31  НК
%%подреждам по стиховата логика - интонационната е друга

\begin{multicols}{2}
Нивясто, 
обзаложи се нивяста, 
%%обзаложи вм. заложи - за смисъл, по муз. запис

нивяста, 
със нейни девят девяра, 

девяра, 
на нива девят уврата, 

уврата, 
със слънци да я зажъне, 
%%стихът не е изпят

зажъне,
със слънци да я дожъне,
%%дописвам рефрените надолу
\kray

дожъне.
Та я със слънци зажъна, 

зажъна 
и я със слънци дожъна, 

дожъна; 
кога си у тях отиди, 

отиди,
Билуда крава ялова, 

ялова, 
по двори ходи и риве, 

и риве,
нивяста си я запита, 

запита:
-- Билудо, краво ялова, 
%%вм. "Билудо" - подчертано: това трябва да значи; инак невъзможен стих

ялова,
що си по двори ревнала, 

ревнала,
дали ти тиле нямаше, 

нямаше?
Билуда \'{и} отговаря, 

'тговаря:
-- Мени е тиле при мени, 

при мени,
а теби дите нямаши, 

нямаши, 
ти го на нива забрави, 

забрави -- 
на нива девят уврата, 

уврата, 
девят го вълка идяя, 

идяя,
на девят пая диляя, 

диляя
и теби пая оставят, 

оставят:
пильнити, ощи повоя! 
%%"пильнити" вм. "пилинити" - за сричките 
\end{multicols}

\sing{Пяла: Куна Вълева Радкова}
\village{с. Бабек}
\area{Пловдивско}
\year{1904}
\rec
