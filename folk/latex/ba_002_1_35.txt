\begin{multicols}{2}
\rows
{Иван Найда дума: }
{-- Найдо, тънка Найдо,}

\rows
{гелим да тропаме, }
{да са натропаме!}

\rows
{Найда Ивану дума: }
{-- Байни ле Иване,}

\rows
{тропай, ко ще тропаш, }
{мене не ми са тропа:}
\kray
%%не трябва ли и в такива случаи да дописваме надолу "/22"???

\rows
{главата ме боли}
{за тънка махрама,}

\rows
{краката ме болят}
{за шитички чехли,}

\rows
{кръхчето ме боли}
{за сърмалъ коланче!}

~~
\end{multicols}
