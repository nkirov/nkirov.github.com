%Добра д'оря мете
%На хоро
%Джиджев:хороводни

\begin{multicols}{2}
Добра д'оря мете,

хем мете, хем плачи:

-- Боже, мили Боже,
%%чувам по-скоро "Божено, боже"

защо са на Добра

митени д'орове,

редени столове,

га си няма Добра

в село първо любе,
\kray

в село и на друго?

Още тва издума,

Стоян стрялка стрялна

и при Добра падна.

Добра я отзела

и във къщи внела,

и на мама дума:

-- Шийте, кройте, мамо,

чи сватове идат!
\end{multicols}

\sing{Пяла Марушка Жекова Петева}
\village{с. Криво поле}
\area{Хасковско}
\year{1888}
