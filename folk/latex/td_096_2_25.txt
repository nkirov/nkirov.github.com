%%в нотите не е фиксиран първия стих "Заръча, Нено, леля ти"

\begin{multicols}{2}
-- Заръча, Нено, леля ти, (2)

Нено, на гости да идеш (2)

по Великден, по хор\'{а}та, (2)

по Гергьовден, по сборове. (2)
%%предпочитам "сборове"
\kray

Нено си стана, отиде, (2)

гости го леля, пои го, (2)

чи го на хоро изведе, (2)

чи съ хвана ги Нено (2)

помиж две тънки-високи, (2)

помиж две бели-чървени. (2)

Момите съ посгледаха, (2)

бабите съ посгълчаха: (2)
%%предпочитам "пусгълчаха"

-- Каква беше тая майчица, (2)

дето из[х]рани този син, (2)

този син, този гидия?! (2)

Нено съ от оро пусна (2)
%%"ут" вм. "уд" - просто няма как да се произнесе "уд"; пък и така е същото, но по-ясно

и на леля си продума: (2)

-- Олиле леле, лельо ле, (2)

лоша ма глава заболя!  (2)
\end{multicols}

\bel{Kогато се съберат у кумата, т.е. в стая, на дюшеме.}

\sing{Пяла Куна Вълева Радкова}
\village{с. Бабяк}
\area{Пловдивско}
\year{1904}
\rec
