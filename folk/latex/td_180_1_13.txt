%Тодор на Тона думаше
%Хороводна
%Джиджев:хороводни

\begin{multicols}{2}
Тодор на Тона думаше:

-- Ой, мари, Тоно, Тодоро, (2)

заръчай, Тоно, поръчай, (2)

какъв армаган да зема!
%%насетне песента се декламира

Тона на Тодор думаше:

-- Земи ми, любе Тодоре,
%%"Тодоре" вм. "Тодаре"

на бяло лице -- белило,

на тънки вежди -- врътлило,

на руса коса -- кънило,

на бяла шийка -- герданче,

на тънко кръсче -- коланче!

~~
\end{multicols}

\ed{Врътило -- според контекста 'пособие за разкрасяване на вежди'.\\
Кънило -- вер. диалектна форма от ``кън\`{а}''.}

\sing{Пяла Яна Иванова Ганева}
\village{с. Бояджик}
\area{Ямболско}
\year{1914}
