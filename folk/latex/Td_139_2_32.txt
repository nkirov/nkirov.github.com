%Тва, що са белейе
%Сватбарска (``когато наближават момковите двори'') 
%Джиджев:сватбарски

\begin{multicols}{2}
-- Тва, що са белейе,

белейе, лъщейе,

срещу тих бял Дунав,

дали са сватове

или са снягове?

-- 'Ку бяха снягове,

щяха да са стопят,

'ку бяха сватове,

щяха да притръгнат!

Не били снягове,

най били сватове.

~~ 
\end{multicols}

\sing{Пяла Доба Петкова Гегова}
\village{с. Брягово}
\area{Хасковско}
\year{1925}
