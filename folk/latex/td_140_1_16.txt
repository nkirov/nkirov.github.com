\begin{multicols}{2}
-- Янке ле, дюлбер Габровко,

от земя ли си никнала,

от небо ли си летяла,

защо си толкоз хубава,
%%следващите 4 стиха отсъстват в ръкописа - тук по муз. запис

\five защо си тънка-висока,

защо си бяла-червена?

-- Байне ле, бате Иване,

катo ма питаш, да кажа: 

нито съм 'т земя никнала,
%%тук и долу "'т" вм. "ут" - по муз. запис, за сричките
%%спорно!!! НК

\ten нито съм 'т небо паднала,
%%в муз. запис - "паднала"

добра съм майка имала,

на добър ден ма добила,

я на личен ма кръстила --
%%добавям "ден" - от муз. запис; наместват се и сричките
%%чува се: я на личен ма кръстила НК

затуй съм толкоз хубава,
%%стихът не е изпят

\fifteen затуй съм тънка-висока, (2)

затуй съм бяла-червена!
\end{multicols}

\ed{\el{Дюлбер} -- диалектна форма на ,,дилбер`` {\em тур.} 'хубав, красив, гиздав, очарователен'.}
