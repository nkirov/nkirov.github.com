%Радовай са, мале
%На сватба (``когато въвеждат булката у момкови'') 
%Джиджев:сватбарски

\begin{multicols}{2}
-- Радовай са, мале, (2)

чи ти снаха водим, (2)

снаха помощница -- (2)

малко помощница,  (2)

пов'че разносница! (2)

~~
\end{multicols}

\sing{Пяла Рада Митева Цънцарова}
\village{с. Пъдарско}
\area{Пловдивско}
\year{1907}
\rec
