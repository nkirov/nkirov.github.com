%Завило са сиво орле
%Коледарска (на стопанин чорбаджия, ``цар'')
%Джиджев:коледарски
%Коледарска (на голям човек -- чорбаджия, на ``цар'')

\begin{multicols}{2}
Завило са сиво орле, \uu{коладе ле,}

\uu{над царьови, над сараи, коладе ле,} 

ви са два дня, ви са три дня,

чи са спусна сиво орле,

та си кацна на царево,

на царево, на коляно,

%%та разпери дясно крилце,
%%стихът не се пее НК

та разтърси дясно крилце,

та изтърси зилен фирман,

зилен фирман, бяло книже
%%я в фирманя бяло книже,
%%чува се "зилен фирман, бяло книже"

бяло книже, черно писмо. 
\kray

Чел го два дня, чел го три дня

и ни можал да прочите

бяло книже, черно писмо.

Йотговаря цар Костадин:
%%предпочитам "ютгуваря"

-- Я викнети поп Никола,

поп Никола и сина му -- 

поп Никола много ходи,

много ходи, много знаи!

Извикали поп Никола,

поп Никола и сина му.

Йотвъд иди поп Никола,

стрялна очи в бяло книже,

стрялна очи, йотговаря:
%%предпочитам "йотгуваря"

-- Ой та тебе, цар Костадин,

с\'{я} ми ходи идна вяра,

идна вяра павликянска,

павликянска, клети турци,

дет да й, младо поробили,

дет да й, старо потъпкали -- 

напреж върви самси господ,

сабя върти, море дели,

от сух пясък турци излизат...

Наздраве ти, сиво орле,

от бога ти тва наздраве,

догодина пак със здраве!
\end{multicols}
