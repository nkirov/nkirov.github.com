\begin{multicols}{2}
Снощи си минах, мамо, заминах

край Солунските, мамо, ливади,

тамка си, мила мамо, заварих

моето любе, мамо, заспало.
\kray

Жал ми бе да го, мамо, събудя,

страх ми бе да го, мамо, оставя,

па сложих глава, мамо, до него

сладко си, мила мамо, позаспах.

Кога се от сън, мамо, събудих,
%%се вм. си - за яснота

любе го няма, мамо, до мене.

~~
\end{multicols}

\sing{Пяла Катя Генова Кацарска}
\year{1932}
\village{с. Розовец}
\area{Пловдивско}
