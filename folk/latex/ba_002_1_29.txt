\begin{multicols}{2}
\rows
{Миранза дума мама си: }
{-- Мар мамо, мар мила мамо, }

\rows
{като се с Гергя любехме, }
{голяма севда имахме, }

\rows
{ален си сукман заносех, }
{жангърлиена престилка.}
%% разлика с музикалния запис НК
\kray
%%подозирам, че и насетне има повторения на двустишия, не трябва ли да ги сложим????

Като ма Герги остави,

мене ме много догневя,

че му къшлата запалих

с дребни шарени агънца.

Като агънца горяха,

горяха, мамо, ревяха, (2)

синьото небе деляха,

като им майки ревяха,

черната земя деляха.
\end{multicols}

\ed{Жангърлиена -- според контекста вер. 'вид плат'.}
