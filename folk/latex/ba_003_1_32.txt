\begin{multicols}{2}
Запаляла бяла Мита, \uu{мари,}

\uu{мари Мито, бяла Мито,}

запаляла ракията,

ракията от лулата,

запали са й бяла Мита,
\kray

бяла Мита Топалова,

Топалова, Иванова.

Мита гори, Янко гаси.

~~
\end{multicols}
