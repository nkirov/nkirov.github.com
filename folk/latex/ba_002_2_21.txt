%Разиграло съ й Черното море
%Хороводна (пее се на засевки на двора)
%Джиджев:хороводни

%%(пее се на засевки на двора)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Разиграло съ й Черното море, (2)

Черното море и бял Дунава, (2)

та са играли, та са трептели (2)

трини ми дена и три вечера, (2)
\kray

та потрошили триста гимийки, (2)

триста гимийки с гимиджиите. (2)

Все гимиджии все по край ходят, (2)

все бога молят и курбан колят, (2)

Андон гимиджий по край не ходи, (2)

бога не моли, курбан не коли, (2)

Андон гимиджий фалба са фали: (2)

-- Мойта й гимийка нова правяна, (2)

нова правяна, скоро пратяна! (2)
\end{multicols}

\bel{Си -- нестабилен тон, $\approx$ си бемол.}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
