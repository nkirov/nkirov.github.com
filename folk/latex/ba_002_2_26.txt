%Пасъл е Митьо вакли шилята
%Хороводна (пее се на засевки)
%Джиджев:хороводни

%%(пее се на засевки)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Пасъл е Митьо вакли шилята,

\uu{кадън Елено, мари, кадън Еленке,}

че си загуби медни кавали,

%%следващият стих не е изпят. 
медни кавали със калъпите.

Намеряла ги й кадън Елена,

моляше \`{и} са Митьо Кирчето:
\kray

-- Мо, ой та й тебе, кадън Елено,

да ми ги дадеш, бъх да не хванеш!

Отговаря му кадън Елена:

-- Твойти кавали мама ги дигна 

в пъстър сандък, при тенки дари!
\end{multicols}

\ed{Бъх да не хванеш -- от ``хващам бъх, ловя бъх'' 'скривам нещо, не казвам както е'.}

\editor{Колебание: Елена -- Илена.}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
