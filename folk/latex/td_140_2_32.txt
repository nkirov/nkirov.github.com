%Протекъла вечерничка
%На хоро (вечерно)
%Джиджев:хороводни

%%като че ли има начално "и"

\begin{multicols}{2}
Протекъла вечерничка, (2)

мътна иде, кална тече, (2)

дърва носи и камене: (2)

дърва мята по брягове, (2)
\kray

я камене по дълбове.

Вакъл овчер покрай ходи,

покрай ходи, с кавал свири,

кавала му отговаря:

-- Спадни, спадни, вечерничко,

да идеме у Радини! --

У Радини огън гори,

огън гори на хамбара -- 

дали Рада болна лежи,

или Рада болен гледа?

Нето Рада болна лежи,

нето Рада болен гледа,

пълна къща главежници

ядът, пият, Рада главят.

И Рада хни права стои,

права стои, конци суче,

конци суче, момци кълне:

-- Кък са късат тея конце,

тъй д' са пукат селски момци!
\end{multicols}

\dic{Дълбове -- вергове.}

\ed{Дълбове -- диалектна форма от ``длъб'' 'жлеб, улей'. }
