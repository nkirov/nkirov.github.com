%Що ти пирцето трипери
%Хороводна (eсенно)
%Джиджев:хороводни

%%(eсенно) е моя добавка от ръкописа и машинописа

\begin{multicols}{2}
-- Що ти пирцето трипери, \uu{мари,}

\uu{Гани ле, годинички ле:}

дали го вятър люлеи,

дали го слънце лъщейе?
%%"лъщеи" вм. "къщем" - по записа и ръкописа, вж. и по-долу
%%този и следващите три стиха не са изпяти

-- Нито го вятър витреи,

нито го слънце лъщейе, 
%%"лъщейе" вм. "къщейе" - вж. и по-долу

тъй ми го й мама купила -- 

без вятър да се люлейе,

без слънце да се лъщейе, 
\kray

люби ми с него харесва!

~~
\end{multicols}
