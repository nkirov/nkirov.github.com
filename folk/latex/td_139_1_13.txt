%Желво мари, вакла желво
%Сватбарска (``весела песен'') 
%Джиджев:сватбарски

\begin{multicols}{2}
-- Желво мари, вакла желво,

\uu{ти-ри-ни-гани,}

желво мари, вакла желво,

\uu{трам-ба-ба-ром,}

тръгнала йе вакла желва

да ми иде на нов пазар,

да си купи нова рокля. (2)

Де я срещна Ежо-Пежо, (2)

прегърна я, целуна я

между двете черни вежди,

между двете черни очи. (2)

Разсърди се вакла желва, 
\kray

че отиде при съдия,

на съдия, на давия,

съдия хни отговаря:

-- Желво мари, вакла желво,

той е момче -- това търси,

ти си мома -- в дома седи! 

~~
\end{multicols}

\sing{Пяла Здравка Димитрова Делчева (Попконстантинова)}
\village{с. Любеново}
\area{Хасковско}
\year{1901}
\re
