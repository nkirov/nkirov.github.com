%Тоне ле, мари хубава
%На хоро
%Джиджев:хороводни

\begin{multicols}{2}
-- Тоне ле, мари хубава,

що караш г'веда и плачеш?

Тонка Ивана думаше:

-- Байне ле, бате Иване,

как ни ща, бате, да плача --

майка ми лежи умряла,
\kray

тейно ми горе с кирийка,
%%"кирийка" вм. "кирийкя" - според ръкописа, неразчетен правилно почерк

брайно ми вързан караха!
%%"караха" вм. "карахя" - според ръкописа, неразчетен правилно почерк

Иван на Тонка думаше:

-- Тоне ле, мари хубава,

дано майка ти умряше,

дано тейно ти не дойде,

дано брайно ти н' отвързат,

ний с тебе да се вземиме!

Тонка Ивана думаши:

-- Байне ле, бате Иване,

недей та дума хортува,

че от хората срамота,

срамота, байне, грехота:

нали сме върла роднина,

тейно и чичо -- два брата,

мама и чина -- етърви,

и ний със тебе -- братвчеди!

-- Тоне ле, мари хубава,

честа корийка сяк няма,
%%"сяк" вм. "сян" - вер. неправилно разчитане на почерка; разбирам го като "няма изсичане"

ситни пясъци брой нямат, 

дълбока 'ода брод няма, (2)

хубава мома род няма!
\end{multicols}


\sing{Пяла Пенка Петрова Славчева}
\village{с. Поляново}
\area{Хаманлийско}
\year{1901}
