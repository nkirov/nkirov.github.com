%Овчар Калинка залюби
%Седенкарска (на летни попрелки)
%Джиджев:седенкарски

\begin{multicols}{2}
Овчар Калинка залюби,

любил я овчар, лъгал я

година и половина,

та че я овчар напусна,

та друго либе залиби,

та друго, та по-хубаво.

На Калинка са наеди,

по сърце жално нажали,

тя му саята запали
\kray

с петстотин вакли ойнове,

със триста майки сирици,

с шестстотин вакли агънца.

Кога майчици блейнаха,

млечaна ряка протека,

кога агънца блейнаха,

кървава ряка протека,

кога ойнове блейнаха,

синьо са небе дереше.

~~ 
\end{multicols}

\ed{Ойнове -- диалектна форма от ``ойня, овня'' 'овен'.}

\sing{Пяла: Доба Петкова Гегова}
\village{с. Брягово}
\area{Хасковско}
\bornYear{1925}
\rec
