%Мама Елка буди
%На моабет
%Джиджев:трапезни

%%вариантите в тефтера и записа са толкова различни, че ги давам отделно. 

%%1. запис
\begin{multicols}{2}

\singvar
Мама Елка буди:

-- Стани, Елко, стани, (2)

на Дунав да идеш (2)

за вода студена!

Елка мама дума: 

-- Аз ни зная, мамо,

Дунав отде 'звира (2)

и къде минава!

Елка мама дума: 

-- Мамо, мила мамо,

откъде минава

и къде са влива?

Мама Елка дума: 

-- Дунав, Елко, 'звира (2)

от Маджар балкана,

тихичко минава

приз росни ливади
\kray

%%2. тефтер (отначало)

\textvar

Мама Елка дума: 

-- Стани, Елко, стани (2)

на Дунав да идеш,

на Дунав за вода!

\rows
{А Елка ни знае }
{отде тиче Дунав}

и къде минава. 

Мама Елка дума: 

-- Дунав тиче, Елко,

Дунава извира

от Маджар балкана; 

и минава, Елко,

през росни ливади;

и са влива, Елко,

във Златна ябълка!

Елка покрай ходи,
%%Елка вм. "Елке" - по смисъл

росни цветя бере, (2)

вити винци прави,

в Дунава ги хвърля

и на винци дума: 

-- Много здраве носете

на мойето либе --

то йe гимиджийче --

и вий му кажете,

че съм изтъкала

триста лакти платно,

та да си отдаря

триста гимиджии!
\end{multicols}

\sing{Пяла Смина Иванова Пиронкова}
\year{1919}
\village{с. Бабек}
\area{Пловдивско}
\rec
