%Слънцето трепна да зайде
%На хоро (Великденско, по мръкнало)
%Джиджев:хороводни

\begin{multicols}{2}
Слънцето трепна да зайде,
%%начално "и" 

хорото кояк играе, (2)
%%мисля, че става дума за "хорото коя как играе"

Маринка хоро остаи, (2)

и си у дома отиде,
%%че си във дома отиде,
%%добавям "във" - според ръкописа
%%в музикалния запис се чува: и си у дома отиде НК

и отзе бели бакъри,
%%"и отзе" вместо "че отзе", защото "че" в ръкописа е разчетено като "и"

през хорото йе минала
%%стихът отсъства в ръкописа - давам го според записа
\kray

че си за водъ отиде

и си на Ванча смахува:
%%"смехува" вм. "смахува" - според ръкописа; спорно е дали е написано "смехува" или "смяхува"
%%според ме е "смахува", от "маха", а не от "смях" НК

-- Любе Ванчо ле, Ванчо ле,

я хайде, любе, след мене

да вземем вода студена

от нея чешма шарена,

шарена, гиди, писана!

Иванчо тръгна след нехи,

отишли вода да зимат

от тая чешма шарена,

шарена, гиди, писана.

Маринка вода налива,

Ванчо хи вода разлива

по тея бели камъни,

по тея ситни пясъци.

-- Любе Ванчо льо, Ванчо льо,

нимой ми вода разлива

по тея бели камъни,

по тея ситни пясъци,

че й от хората срамота

и от Господа грехота!
\end{multicols}

\ed{Кояк -- нефиксирана диалектна дума; според контекста 'вид хоро'.\\
Смахувам -- диалектна форма от ``смахвам'' 'давам знак с глава, кимам'.
 }
