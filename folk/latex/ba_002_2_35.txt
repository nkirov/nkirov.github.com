\begin{multicols}{2}
-- Мойто люби парясано ходи! (2)

-- Нека, нека ходи, акъл да му доди, (2)

че то не е до мене й до него, (2)
%%че то не е до него й до мене, (2)
%%музикален запис: че то не е до мене й до него НК

че то най е до неговата майка! (2)

\five Тя където ходи се с мене се хвали: (2)
%%"с мене" вместо "свене" - според записа, в името на смисъла; 
%%за да се пее трябва дори да е "се с мен"

,,Туй момиче, туй нашто комшийче, (2)
%%"нашто" вместо "нащо" - същото, но по-четивно
\kray

туй момиче моя сина люби, (2)

туй момиче снаха ще ми стане! (2)

Китки бере от мойта градинка, (2)

\ten конци зима от мойте вретена, (2)

китки вие по мойте сайвани, (2)

и ги дава на мояго сина!`` (2)

~~
\end{multicols}
