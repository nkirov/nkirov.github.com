%%добавям "(пее се на ловец)" - по муз. запис

\begin{multicols}{2}
Разболя съ й болян Добри,
%%чува се: са болян, и по-долу е написано така! НК

\uu{хей, Добре, колади ле,}

\uu{мой Добре, колади ле,}

та е лежал, та е болял
%%чува се: "та е лежал, та е болял"

малко-много, девят годян.

Девят постеля изгноил
%%чува се "постелки издои" - вероятно лапсус
%%предполагам "издуи", т.е. издухал НК
\kray

и девят пъстри възглавки,

и девят китни губера.

Кат лежеше болян Добри,

и на бога се молеше:

-- Помогни ми, мили боже,

пак да стана от постеля!

~~
\end{multicols}

\sing{Пял: Стоян Иванов Вандов}
\village{c. Кортен}
\area{Сливенско}
\year{1908}
\rec
