%Хранила майка, гледала
%Сватбарска (на пряпора)
%Джиджев:сватбарски

\begin{multicols}{2}
Хранила майка, гледала

два сина, дор два близнака,

та чи ги мама изхрани
%%Храни ги майка, та ги изхрани
%%вм. та чи ги мама израни - по записа

голями момци станали, 

та чи им майка направи
%%чува се: майка НК

двамата една примяна,
\kray

та чи ги майка примени

двамата [с] една примяна,

двамата с бели навои,

двамата с черни сукари. 

Кого си мама главеше,

нему си гримни леише,

кого си майка женеше,

~~
\end{multicols}

\dic{Сукари -- върви.} 

\sing{Пяла Жела Господинова Филипова}
\village{с. Навъсен}
\area{Харманлийско}
\year{1907}
\re
