\begin{multicols}{2}
Съшила й Станка ръкави,

\uu{Станo, мари Дилджилеркена,}

съшила й Станка ръкави,

\uu{Станo, мари, разсипници мо,}

три пера за три месеца, (2)
\kray

всичките за три години. (2)

Разсипа мами къщата (2)

и на баща си хамбаря, (2)

и на буля си раклата, (2)

й на брат си Гергя чердака! (2)

~~
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
