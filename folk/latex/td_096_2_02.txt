%Завардили са Ивана
%Хороводна (право, винаги) 
%Джиджев:хороводни

%% (право, винаги) е моя добавка от ръкописа и машинописа

\begin{multicols}{2}
\uu{Иване,} завардили са Ивана,

на два го друма вардили,

на трети са го хванали,
%%чува се: на трети

чи го в тъмница запрели,

в Черногорската тъмница. 

Ивановата майчица,
\kray

тя край тъмници ходише 

и на Ивана думаше: 

-- Мъчно ли ти йе в тъмница?

\uu{Мамо льо,} диня ми й мъчно и ни йе,

а нощя много усилно -- 

кога момите запеят,

момите по сиденките!
\end{multicols}

\sing{Пяла Куна Вълева Радкова}
\year{1904}
\village{Бабек}
\bornArea{Пловдивско}
\record{декември 1964}
