%Слънцето трепти, захожда
%Жътварска (на мръкване)
%Джиджев:жътварски

\begin{multicols}{2}
Слънцето трепти, захожда,

Димитър чекъм захваща
%%вм. следващите три стиха в муз. запис директно "и на Тодорка думаше:"
%%чува се ясно: захваща НК

с негова сестра Тодорка,

с негово булче Петкана

и на сестра си думаше:

-- Бързай, побързай, Тодорке, (2)
%%Тодорке вм. Тудорке - уеднаквявам; информацията е същата
%%чува се ясно: побързай НК

на края да си излезем!
%%чува се ясно: излезем НК
\kray

Тодорка дума Димитър: 

-- Брайне Димитре, Димитре,

слънцето трепти, захожда,

пък ти си чекъм захващаш,

ти не знаеш ли, брайно ле,

че в гора има хайдути?

...
\end{multicols}
