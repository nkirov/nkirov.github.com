%Дойна д'ори мела
%Сватбарска (хоро на пряпора)
%Джиджев:сватбарски

\begin{multicols}{2}
Дойна д'ори мела,

\uu{ти, Дойне, ти, Добро,}

със бял стрък босиляк.

Га й видяла, Дойна,

сватове пристигнали,
%%вм. "дуйдоха" - по записа, насетне пак несвидетелско: "напущали", "намятали"

кони напущали,
\kray

юзди намятали

под бял трендафил.

Дойна са в къщи спусна

и майка си дума:

-- Мар, мамо, мар, мила мамо,

сватове са дошле ...

~~
\end{multicols}

\sing{Пяла Гела Желева Димитрова}
\village{с. Мустрак}
\area{Свиленградско}
\year{1926}
\re
