%Събрали са, набрали са
%На хоро (по Великден)
%Джиджев:хороводни

\begin{multicols}{2}
Събрали са, набрали са,

\uu{Дено мари, кара Дено,}

йот три села по тровица,

чи стануват девет души,

чи у Денини отишли,
\kray

у Денини на вратата.
%%"вратата" вм. "врататя" в машинописа, според мен неправилно разчетен ръкопис.

Порукали, почукали:

-- От[в]ори ни, кара Дено!

-- Ни отварям, ни отключвам!

-- Ние не сме чузди хора,

ние сме ти бащините -- 

бащините баш гов'дари,

баш гов'дари, баш овчаре!

Ний сме дошли соль да земем,

соль да земем, соль и брашно,

да накърмим имането!

И Дена хни от[в]орила,

от[в]орила, отключила.

С казань вари прясно мляко,

със мляко га поливаха:

-- Кажи, Дено, имането!

Я Дена хни отговаря:

-- Я ни хни знам имането,

я съм скоро доведена -- 

има-няма три недели,

сберала съм три хиляди -- 
%%предпочитам "сберала"

ей ги тука на полица,

на полица, под паница,

земете ги, идете си!
%%"идете си" вм. "йдете си" - според ръкописа

-- Лъжи тизе кога лъжеш,

харамия са ни лъже! --

Чии бяха тия деца

по Великден по хората,

по Гьоргьо(в)ден по църквите

със низани калпачета?!
\end{multicols}

\sing{Пяла Гина Ангелова Георгиева}
\village{ с. Лешниково}
\area{Хаманлийско}
\year{1936}
