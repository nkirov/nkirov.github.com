\begin{multicols}{2}
Върбанчо Радки думаши:

-- Радки мо, Върбанчовата, (2)

често край дома минавай,

на мама селям отдавай: (3)

добър ден и добър вечер! (2)
\kray
%% всеки втори стих се повтаря?? НК

Дано та мама хареса,

хареса, че и бийндиса,

да земе тебе за меня!

Радка Върбанчо думаше:

-- Върбанчо, любе Върбанчо,

минавах, любе, минавах,

на мама ти селям давах,

пак ма мама ти не рачи,

тя не ще меня за тебе!

~~
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
