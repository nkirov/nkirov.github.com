%Карагьоз Петро льо, нивясто
%На межия
%Джиджев:седенкарски

\begin{multicols}{2}
-- Карагьоз Петро льо, нивясто,

\rows
{ти иди кажи на леля си, }
{на леля си, по-малката,}

милнинкуш да ма погледни: (2)

двесте ще да хи наброя, 

двестето бели грошове,

а другож да ми продума,

триста ще да хи наброя,

тристато бели грошове!
\kray

Карагьоз Петра нивяста,

тя на Петъра думаше:

-- Деверьо Петре, деверьо,

я съм от леля по-хитра,

по-хитра и по-хубава,

я ще те еднъж погледна:

двестето да ми наброиш,

а другож да ти продумам,

тристато да ми наброиш!

-- Карагьоз Петро ле, нивясто,

мълчи, нивясто, не думай,

тва е от бога грехота, 

а от людето срамота!

~~
\end{multicols}
