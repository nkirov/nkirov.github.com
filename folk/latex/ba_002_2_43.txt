%%(пее се на засевки на двора)

\begin{multicols}{2}
-- Мари Радо, бяла Радо,

хайди да са спобратимим, (3)

да ни й лесно да се любим! (2)
\kray

Любели са Рада й Иван,

любели са три години,

три години дъжд не капна,

дъжд не капна, сняг не падна.

Събрали са поповете,

че отишле в черковата,

разгънали тефтерите,

намер'ли Рада и Ивана,

Рада и Иван побратими,

побратим и посестрима --

любели са Рада й Иван.

Че събрали селяните,

докарали кола дърва,

кола дърва й кола тръни,

наклали са буян огън,

хвърляли са Рада й Иван,

изгорели Рада й Иван.

~~~
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
