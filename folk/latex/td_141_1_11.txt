%Брайне ле, кара Димитре
%Жътварска (вечер, по икиндия)
%Джиджев:жътварски

\begin{multicols}{2}
-- Брайне ле, кара Димитре,

немой ми чекъм захваща,
%%в муз. запис "нимой ми"

че са я время мръкнато, (2)

че е година хайдушка,

хайдушка, янкеседжийска! (2)
%%стихът отсъства в ръкописа, тук по муз. запис

Още си това хортуват,
%%в зап. "хортуват"

и хайдутето дойдаха,
\kray

те са Димитра хванали,

назад му ръце вързали

и го в гората отвлекли.

~~ 
\end{multicols}
