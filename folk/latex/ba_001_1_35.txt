%Събрали са събор момци
%Коледарска (из пътя)
%Джиджев:коледарски

%%добавям "из пътя" - по муз. запис

\begin{multicols}{2}
Събрали са събор момци,

\uu{хей, колади ле, мой колади ле,}

тръгнали са да коледуват,
%%добавям "са" - по муз. запис

че нямало кой да ги води
%%в муз. запис. следва стих "Отговарят сабор момци" - не го вписвам, защото се къса смисъла
\kray

из тази града Никополска,

Никополска, триста къщи.

Отговаря най-стар коледар:

-- Ой ва вази, събор момци,

я пробийте душна бъчва,

че напълнете жълта бъклица,

че повикайте леком гърчето,

че то леко ходи, леко дохожда,

да иди да калеса Банов Петра --

тъй да му рече леко гърчето:

``Ой та тебе, Банов Петре,

много ти здраве от събор момци,

проводиха ма да те калесам,

ти да дойдеш, да ги водиш,

да ходят да коледуват

из тази града Никополска,

Никополска, триста къщи!'' 
\end{multicols}

\sing{Пял: Стоян Иванов Вандов}

\area{Сливенско}
\year{1908}
\rec
