%Мари Дено, кара Дено
%На Лазарица
%Джиджев:лазарски

\begin{multicols}{2}
-- Мари Дено, кара Дено,

спиш ли, Дено, събуди са,

будна ли си, обади са! 

Не сме, Дено, лоши хора,

най сме, Дено, ваши хора: (2)

бащините баш овчари,

братовите баш гов'дари.

И Дена им отворяла.
%%чува се: отворяла ! НК
\kray

Най са били зли хайдути: 

казан мляко на огъня,

с млякото я поливаха,

поливаха, разпитваха:

-- Казвай, казвай, мари Дено,

бащиното си имане!
%%предлагам: бащиното НК

Отговаря Денина майка:

-- Ей го, ей го, на полица,

вземете го, идете си,

само Дена не халайте,

че е Дена скоро дошла,

има-няма два дни, три дни,

два дни, три дни, до неделя!
\end{multicols}

\dic{Халайте -- закачайте.} 

\sing{Пяла Калинка Динева Динева}
\village{с. Овчи кладенец}
\area{Ямболско}
\year{1938}
\re
