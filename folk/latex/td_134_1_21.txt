%Никол Тодоро думаше
%Хороводна (вечерно)
%Джиджев:хороводни

%%добавям (вечерно) - от ръкописа и машинописа

\begin{multicols}{2}
Никол Тодоро думаше:

-- Любе Тодоро, Тодорке,

йоткато са залюбихме,

та са нийде не найдахме,

не найдахме, не срещнахме.

Дори снощя на кладенци

малко сме много поствяли,

поствяли, похортували, 

дорде слънцето да зайде,

зайдьом зайде, дойдюм дойде,
\kray

месечинка рог подаде, (2)

рог подаде да изгрява! (2)

Тодорка дума Никола:

-- Пусни ма, любе Никола,

мойта я майка мащеха,

я га си ида ю нази,

каква лъжа да га лъжа?

Никол Тодоро думаше:

-- Вярна га лъжа излъжи:

``Царя я дошъл, мале ле,

и на кладенци кондиса,

мене бакъри йотзеха,

царьом конето поиха!''
\end{multicols}

\sing{Пяла Димка Запрянова Матева}
\year{1933}
\village{Навъсен}
\area{Харманлийско}
