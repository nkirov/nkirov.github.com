%На жътва

%%в ръкописа и муз. запис стиховете са разместени - тук ги давам според ръкописа, по-логично е

\begin{multicols}{2}
Гарван грачи накрай нива, \uu{бяла Радки ле,}

ем си грачи, и говори: 
%%добавям "и" - по записа, за сричките

-- Жъни, Радке, жъни, либе,

ча ще ми, Радке, пристанеш!

\five Не е било черен гарван,

най е било вакъл овчар,
\kray

че изтърва сиво стадо,

че опасе царю нива,

че глобиха вакъл чобан,

\ten глобиха го тежка глоба: 

на овчица по жълтица,

на класчето по алтънче.

~~ 
\end{multicols}
