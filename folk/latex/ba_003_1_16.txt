\begin{multicols}{2}
\rows
{Бяла Недо Димирюва, \uu{мари,}}
{Димирюва, дюлгерюва,}

\rows
{недей люби чифчияна, \uu{мари,}}
{чифчия се мъчно гледа:}
\kray

\rows
{сърдит отива на нива,}
{сърдит се връща от нива,}

\rows
{ти го каниш да се мие,}
{той се кани да те бие}

\rows
{със теслата по главата,}
{с копралята по краката!}

\rows
{Най залюби бакалчето,}
{бакалин се лесно гледа:}

\rows
{ти го каниш да се мие,}
{той те кани със смокини!}
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
