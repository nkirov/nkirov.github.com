%Женала й Каля, Кальо мо
%На жетва (по пладне)
%Джиджев:жътварски

\begin{multicols}{2}
Женала й Каля, Кальо мо,

хем жени Каля, хем гледа

дали с' зададе някоя
%%c' вм. си - намествам сричките

обяда да \`{и} донисе. 

\five Не са й задало никого, 

най са й задало, задало (2)

султанче -- сиди на коня,
\kray

храняна коня ихреше,
%%трябва да е "хранена коня" - аляна вер. е дошло по невнимание
%%от долния стих

аляна феса кривеше

\ten и си на Каля думаше:

-- Кальо мо, мома хубава,

хайде със мене да дойдеш,

на висок чардак да стоиш,

желта коприна да предеш,

\fifteen желти алтъни да нижеш! 
\end{multicols}

\dic{Ихреше -- яреше.}
