%Изведе си дели Марко
%Коледарска (в двора на къщата)
%Джиджев:коледарски

\begin{multicols}{2}
\rows
{Изведе си дели Марко, \uu{коладе ле}, }
{негова си [х]ранян коня, \uu{коледо льо},}
%%чува се: негова се раня коня НК

\rows
{оседла го синьо седло, }
{обезда го жълти юзди, }
%%"обезда го" вм. "синьо седло" - по муз. запис
%%следващите два стиха не са дешифрирани - правя го аз

\rows
{качи му се, връзседна го, }
{че го й пусна тука долу}
\kray

да обиди Света гора -- 

що е гора погрозняла, 

погрозняла, почерняла. 

Отговаря дели Марко:
%%предпочитам "отговаря"

-- Ой та й тебе, Света гора, 

дали те е жар жарила

или слана изгорила? 

Отговаря Света гора:
%%предпочитам "отговаря"

-- Нито ма е жар жарила, 

нито слана изгорила,

снощ минаха три синджиря,

три синджиря млади роби:

първи синджир -- с\'{е} юнаци,

втори синджир -- с\'{е} невести,

трети синджир -- малки моми!

Отговаря дели Марко:
%%предпочитам "отговаря"

-- Ой та тебе, [х]ранян кончо,

я ми фръкни, та подфръкни

да застигнем три синджиря,

три синджиря млади роби: 

ако ли ги ти настигнеш,

гривата ти ще позлатя,

копита ти ще посребря,

ще те поя с ройно вино,

ще те храня с ситен ориз! 

И той фръкна, та подфръкна,

та настигна три синджиря,

три синджиря млади роби.

Извади си тънка сабя,

развъртя се, брей, наляво,

брей, наляво, брей, надясно,

унищожи друга вяра,
%%съкращавам "та" - за да се вместят сричките

друга вяра, турска вяра.

Млади булки са запяли,

млади момци заиграли,

малки моми са засмяли,

Света гора лист подаде,

месечина рог подаде.

~~ 
\end{multicols}
