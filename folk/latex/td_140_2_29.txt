%Остана Ванчо сираче
%На хоро (зимно хоро)
%Джиджев:хороводни

\begin{multicols}{2}
Остана Ванчо сираче, \uu{мари,}

\uu{Ванче ле, вакло ягънце,}

биз майка Ванчо, биз тейно, 

чи раста Ванчо, нараста, 

голям е момък нарастъл.
\kray

Купи си серес биволи,
%%"серез" вм. "сирез" - според ръкописа

отиде оран да оре

горе ми, горе над село,

над село, над градините.

Биволи ойгун въряха,
%%"ойгун" вм. "оигун" - според ръкописа

угар са лесно риеше,

на Ванча драго додрагна

и още жално дожаля,

той рукна, Ванчо, да пеи

една ми песен йевелска,

йевелска, йевелзаманска.

Нему са глася зачуват --

дали му мома отпява

или му овца отблява?

Не било овца в гората,

най било мома в нивята,
%%по смисъл трябва да е "в нивята", аз бих го заменил в името на смисъла; иначе обяснявайте какво е "увята"

тя си на Ванчо отпява:

-- Любе Ванчо льо, Ванчо льо,

разори църни угари,

та посей бяла пченица, 

да даде Господ да стане,

напълни, любе, хамбари,

на сюрмасци я харизвай,
%%"сюрмасци" вм. "сюрмаци" - според ръкописа

на зенгини я продавай!
%%"зенгини" вм. "зангин" - според ръкописа
\end{multicols}

%%добавям от ръкописа
