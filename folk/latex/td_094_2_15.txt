%Рано ми рани млад Стоян
%За арман (за чобаните) 
%Джиджев:жътварски

%% (трудова, но не жетварска??? - Г.Г.)

\begin{multicols}{2}
Рано ми рани млад Стоян,

рано ми рани да дои:

дорде дружина -- две видра,

Стоян издои четири
%%в записа се чува "надои" - не променям

и па си стадо изкара

на връй, на Стара планина,

и със кавали засвири. 

Кавал му дума-казува:
\kray

-- Де си, Минке ле, да дойдеш,

да дойдеш, Минке, да видиш

кви съм чардаци направил:

гридити му са, Минке ле,

от дърво, от дафиново;

мартаци му са, Минке ле,

от дърво, от маслиново;

педаври му са, Минке ле,

от дърво, от чимширово --

от хубос, Минке, да легнеш,

от миризмата да заспиш!
\end{multicols}
