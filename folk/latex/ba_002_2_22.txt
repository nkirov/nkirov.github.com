%Тънка Найда вода гази
%Хороводна (пее се на засевки вкъщи)
%Джиджев:хороводни

%%(пее се на засевки вкъщи)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Тънка Найда вода гази,

\uu{Найдо, Найдо, тънка Найдо,}

вода гази, жедна ходи,

в ръка държи огледало,

хем си гледа, хем си дума:
\kray

-- Каквато съм белолика,

малко да съм чернаока,

турците ша разтурчувам,

българите потурчувам,

кьор Хасана кярдосовам! (2)

~~
\end{multicols}

\ed{Кярдосвам -- нефиксирана диалектна дума; според контекста 'вземам', може би производна от ``кяр'' тур. 'печалба, придобивка, облага'.}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
