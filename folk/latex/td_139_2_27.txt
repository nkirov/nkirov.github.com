%Имал е Черньо, Черньо льо
%На хоро (по Гергьовден)
%Джиджев:хороводни

\begin{multicols}{2}
Имал е Черньо, Черньо льо,

до девет млади аргати, (2)

до девет чифта волове (2)
%%"волове" вм. "олове" - чува се ясно, няма нужда да правим енигматичен един първично ясен текст

и девет нови орала. 
%%"нови урала" вместо "рала железни" - според записа; "рала железни" можем да оставим като вариант - има го в ръкописа.

\five Черньо аргате думаше:

-- Аргате, млади аргате,
\kray

утре щем рано да станем,

рано на нива да идем,

рани сусаме да сеем!

\ten Аргате думат на Черня:

-- Черньо льо, Черньо чорбаджи,

утре е личен Гьоргьовден,

кък щим на нива да идем --

%%последователността на долните два стиха е обърната в машинописа; връщам варианта от ръкописа - 
%%той е в съзвучие с логическата последователност и с последователността горе
утре се оран не оре,

\fifteen утре се сусам ни сее?!

Черньо аргате думаше:

-- Аргате, млади аргате,

...
\end{multicols}
