\begin{multicols}{2}
Кальо, бяла българко,

фърли, метни ябълка, (3)

разбий града да вляза, (2)
\kray

самси ща ги разпозна, (2)

коя й бяла българка, 

коя й бяла кадъна, (3)

коя й черна циганка! (2)

Коя й бяла българка,

тя си носи бял дюлбен; (2)

която й бяла кадъна,

тя й с фередже покрита; (2)

коя й черна циганка,

тя си има шалвари. (2)
\end{multicols}

\ed{Дюлбен -- диалектна форма на ``тюлбен'' тур. 'бяла женска кърпа за забраждане; забрадка, чембер'.}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
