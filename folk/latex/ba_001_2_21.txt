%(пее се на засевки)

%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските 
\begin{multicols}{2}
Ябълка йе род родила, (2)

от рода се е потрошила. (2)
%%наместо "ут рода се путрошила" - според записа

Няма пъдар да я варди, (2)
%%"варди" вместо "пази" - според записа

приставяли мома й момък, (2)
%%"момък" вместо "момка" - според записа
\kray

мома й момък да я вардят. (2)
%%"вардят" вместо "пазят" - според горната промяна

Мома момко отговаря: (2)

-- Бери, любе, да береме, (2)

бягай, любе, да бягаме, (2)

че ша идем в Цариграда: (2)

аз ша стана царицата, (2)

ти ша станеш самси царя --  (2)

аз ша съдя на момите, (2)

ти ша съдиш на момците! (2)
\end{multicols}
