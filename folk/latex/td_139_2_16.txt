%Имал е Стоян, Стояне
%На хоро (на кавали, заг'велски)
%Джиджев:на кавали

%%добавям ", заг'велски" - от тефтера
%%"За песните, които се пеят една седмица преди заговелки,
%%срещу месните заг'велки и срешу сирните, казват, че се пеят "на кавали"

\begin{multicols}{2}
Имал е Стоян, Стояне,

\uu{Стояне, лелин Стояне}

до девет сестри женети,

женети, неповърнати,

снощи ги Стоян калеса

за сношна вечер да дойдат,
\kray

да дойдат, да са повърнат.

Сичките сестри дойдоха,

най-малката му сестрица,

тя кара овца с ягънче

и носи мъжко детенце.

Седнали, та вечеряли,

легнали, сладко заспали,

тъкмо поср\'{я}дом н\'{о}щове

мъжко детенце приплака,

той рече [в]ода, та [в]ода",
%%дали не е по-добре "то", за да се разбира, че е детенцето

нийде са [в]ода не лаха:
%%"лаха" вм. "леха" - по тефтера

вино му дават -- не пие,

грозде му дават -- не яде.

Че стана Стоян, отиде

на хайдушкото кладенче,

че там си Стоян завари

две жени, три магьосници:

едната била леля му,

другата била чина му --

с решето вода вадеха,

месечинката сваляха,

та я на крава правеха,

от нея мляко дояха

и на Стояна думаха:

``Ти ли са л\'{а}ха, Стояне,

бърже във къщи да идеш

майка ти да та лекува --

от жива змия сърцето,

от усойница главата!''
\end{multicols}
