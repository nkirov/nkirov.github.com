%Мари Димитро льо
%Хороводна (пее се на засевки вкъщи)
%Джиджев:хороводни

%%(пее се на засевки вкъщи)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
\uu{Мари} Димитро льо, русокосо момиче, (2)

иди кажи на майка си, Димитро,

иди кажи на майка си, хайде де,
%%това се чува в музикалния запис! НК
\kray

да не ражда друго чедо като теб,

да не гори други ерген като мен!

Заречи са, закълни са, Димитро,

че не любиш друго люби, освен мен!
%%"че" вместо "как" - по подобие на по-долу

\uu{Абре,} заричам са, заклевам са, Иване, 

че не любя друго люби, освен теб --

да изгори бащяна ми пилифар,

под пилифар дор два студни кладенца,

на кладенци дор два ситни синджира,

на синджири дор два сиви гълъба,

гълъбите деня й нощя гогуват!

~~
\end{multicols}

\ed{Гогуват -- диалектна форма от ``гукам, гугукам''.\\
Пилифар -- вер. диалектна форма от ``пеливор, пиливор'' гр. 'градина'.
}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
