\begin{multicols}{2}
-- Чули сме та и разбрали,
%%и вм. й - по записа, за сричките

\uu{станинине, господине,}

че си имаш малко момче,

малко момче, златокосо,
%%чувам друг стих след това: "златокосо, среброносо???"

пише му се цар да бъде,

цар да бъде, дорде бъде.

-- То йе малко, най-глупаво, 
\kray

то не може столен държа

то не може войска реди!

-- То си има дор три сестри:

първа ще му столен държи,

втора ще му кон седлае,

трета ще му кон седлае!

\end{multicols}
