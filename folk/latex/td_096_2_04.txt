%Питровка вода налива
%Хороводна (право, винаги)
%Джиджев:хороводни

%% (право, винаги) е моя добавка от ръкописа и машинописа

%%коментар върху нотите: "Певицата започва песента като "пайдушко" - 5/16 
%%и я завършва като "право хоро" в 2/4
%%затова и са приложени два нотописа - за първа и втора мелодия

\begin{multicols}{2}
Питровка вода налива,
%%започва с 5/16 

\uu{Питровко льо, хубава,}

на студен бистър кладиниц, 

Иван при нея отиди

и на Питровка думаше: 

-- Да са провали мама ти,

\uu{Питровко льо, любе ле,}
%%смяна на рефрена НК

дето та мени ни дади, 

ами та дади другиму, 
%%рефренът тук е изтърван

другиму, на наш комшия!

Питровка Иван думаше: 
%%стихът не е изпят

-- Иване ле, любе ле, 

нидей проваля мама ми, (2)
%%оттук до края рефрен не се пее НК

а най проваляй мама си!
%%"си" вм. "ти" - по записа и по граматиката

Мойта ма мама даваше,

а твойта ма ни щеше:

чи съм била мъничка --

ни мога къща да въртя, (2)

ни мога челяд да гледам!

Иване ле, люби ле,
%%от този стих е 2/4  НК

кукувица и голяма, (2)

ни можи гниздо да вии, (2)
%%"можи" вм. "мой" - тук и долу, по записа, наместват се и сричките

ни можи челяд да гледа; 
%%горните три стиха в машинописа под долните 3; 
%%в записа - наопаки; преобърнал съм ги, така е и по смисъл

птичинце йе мъничко, 

то си гнизце извива (2)

и си челяд отглежда.
%%\kray

%%Кукувицата й голяма
%%тези 3 стиха липсваха, има ги в машинописа - дописвам ги НК
%%тя ни мой гнездо да вии
%%и ни мой челяд да гледа.
\end{multicols}

\bel{Певицата започва песента като ``Пайдушко'' (5/16) и я завършва като
``Право хоро'' (2/4).}

\sing{Пяла Куна Вълева Радкова}
\year{1904}
\village{Бабек}
\bornArea{Пловдивско}
\record{декември 1964}
