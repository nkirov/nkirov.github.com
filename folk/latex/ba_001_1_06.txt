%Изтекли са три пръчици
%Коледарска (дето има три моми)
%Джиджев:коледарски

%%добавям (дето има три моми) - по муз. запис

\begin{multicols}{2}
Изтекли са три пръчици,

\uu{драги, благи явлане, мой коладе,}

три пръчици еднобойки,
%%стихът не се пее НК

то не било три пръчици,
%%чува се: то ни бяло НК

най си били три девойки,
%%чува се: най са бяли НК

три девойки еднобойки. 
\kray

Излезли са да са хвалят

че чий люби най-хубаво,

най-хубаво, най-прочуто

надалеком, нашироком. 

Първа девойка свойто хвали:

-- Мойто й люби най-хубаво, 

най-хубаво, най-прочуто

надалеком, нашироком --

то си има татък долу,

татък долу на Марица,

на Марица воденица,

воденица с три камъка,

с три камъка, с три хамбаря,

три хамбаря бяло жито!

Втора мома свойто хвали: 

-- Мойто й люби най-хубаво, 

най-хубаво, най-прочуто

надалеком, нашироком --

то си има татък долу,

татък долу в Цариграда

три дюкяна праматарски!

Трета мома свойто хвали:

-- Мойто й люби най-хубаво, 

най-хубаво, най-прочуто

надалеком, нашироком --

то си има татък горе,

татък горе в планината,

в планината, в равнината

три сюрии вакли руди!
\end{multicols}

\sing{Пял: Стоян Иванов Вандов}
\village{c. Кортен}
\area{Сливенско}
\year{1908}
\rec
