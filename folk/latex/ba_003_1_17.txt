%Отишла и Дена
%Хороводна (пее се на засевки на двора)
%Джиджев:хороводни

%%(пее се на засевки на двора)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
\rows
{Отишла и Дена}
{за вода студена,}

\rows
{насреща \'{и} иде}
{керван силна сватба,}
%%по-горният стих е различен в машинописа, давам го според записа
%%предлагам "сватба", а не "свадба"
\kray

\rows
{със сватбата върви}
{Маньо добър юнак.}

\rows
{Маньо Дени дума:}
{-- Дено, малка моме,}

\rows
{ела, Дено, с мене}
{в черкова да идем,}

\rows
{кума да ми станеш,}
{мене да венчееш,}

\rows
{мене да венчееш}
{с булка чернаока!}

\rows
{Дена Маньо дума:}
{-- Постой, почакай ме}

\rows
{във дома да ида,}
{мама да попитам --}

\rows
{ако ма мама пусни,}
{тогава ще дойда!}

\rows
{Повърна са Дена,}
{във дома отиде,}

\rows
{Дена мама дума:}
{-- Мамо, мила мамо,}

\rows
{меня Маньо вика}
{кума да му стана,}

\rows
{него да венчея}
{с булка чернаока!}

\rows
{Мама Дена дума:}
{-- Дено, казъм Дено,}

\rows
{Маньо кума има,}
{Маньо булка няма,}

\rows
{тебе Маньо вика}
{не кума да станеш,}

\rows
{най та Маньо вика}
{булка да му станеш!}
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
