%Слънчице мило мамино
%Жътварска (по икиндия)
%Джиджев:жътварски

\begin{multicols}{2}
-- Слънчице мило мамино,

дали ти мама заръча (2)

до пладня, синко, да грейеш, (2)

от пладня зайдом да зайдеш

да та жътвари ни кълнат: 

жътвари по чуждо ниви,
%%чува се: по чуждо ниви НК
\kray

косачи в росно ливаде?!

Слънчо мама си думаше:

-- Чи яз щях, мамо, да зайда,

като поседях, погледах

двама са братя диляха,

със шиник пари мериха,

пък с половина жълтици

и на майка си пай правят:
%%пай вм. най - как го четеш ти, Николай? И как го разбираш?
%%надолу губя смисъла - имате ли идеи?
%%да - всичко е наред !! НК

чуждата людска работа;

лете чифчийски сърпове,

есен юрушки дараци,
%%есен вм. есенно - сричка повече, вер. грешка

зиме търговски станове --

да работи, да ся прихрани!
%%сричка повече - не разбирам смисъла, не смея да поправя
\end{multicols}

\sing{Пяла: Дона Симова Керева}
\village{с. Свежен}
\area{Пловдивско}
\year{1914}
\rec
