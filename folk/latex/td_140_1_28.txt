\begin{multicols}{2}
Драгана рано ранила, (2)

ранинко рано в ниделя, (2)

та си главата омила (2)
\kray

и си къщата измела, (2)
%%може би "измела" ?? НК

и си сметове фърлила, (2)

сметове между плетове. (2)

Че са запали, запали

царюви най-баш сарафе, (2)

царица с мъжко детянце (2)

тя си са, холам, отрука: (2)

-- Фанете ми я Драганка, (2)
%%-- Фанете ми ги, Драганке,
%%по смисъл трябва да е "Фанете ми я Драганка" - препоръчвам замяната в името на смисъла

сама ще да га убия (2)
%%"га" вм. "та" - според ръкописа

сряд село, на хорището! (2)
\end{multicols}
