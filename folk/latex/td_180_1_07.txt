%Бре йовчарко, йоловарко
%Хороводна
%Джиджев:хороводни

\begin{multicols}{2}
-- Бре йовчарко, йоловарко,

изведи ма йот тъз гора, (2)

йот тъз гора ясенова,

ясенова, росенова,

заведи ма на наш баир,

на наш баир, нашо село! --

нашо село огън гори,

огън гори, митла мите! (2)

-- Нека мите, да измите (2)

мойта майка, моят баща, (2)

те храняли, те гледали (2)
%%"храняли" вм. "краняли" - според записа, възстановявам смисъла

девет чеда изгледали -- (2)

и дивиттях момичета; (2)

всяко село по йдно дали, (2)

за мен село не останало, (2)

че ме дали на йдин турчин, (2)

на йдин турчин камиларин (2)

да му паса камилите, (2)

камилите, пущините -- (2)

денем лежат, нощем пасат!
\end{multicols}

\sing{Пяла Цона Колева Славова}
\village{с. Бояджик}
\area{Ямболско}
\year{1930}
