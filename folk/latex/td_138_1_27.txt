%Мама Елка плете
%На попрелки (на моабет)
%Джиджев:седенкарски

%%последователността на изброяването е различна в муз. запис и машинописа; 
%%затова стихове 5-7 изглежда като да отсъстват

\begin{multicols}{2}
Мама Елка плете,

плете, люто кълне:

-- Елко льо, дъщерьо,

ти да са ожениш

кога чуиш, Елко,

камен да попеи 
%%този и още 2 стиха не са изпяти НК

от Стара планина;

кога чуиш, Елко,

риба да продума (2)

от тих бял Дунава;
\kray

кога видиш, Елко, 

сама кукувица

гняздо да извие,

пилци да изпили!

И Елка чакала

три-четри години, 

ни е чула Елка

камен да попеи (2)

от Стара планина; 

ни е чула Елка 

риба да продума (2)

от тих бял Дунава;

не видела Елка

сама кукувица 

гняздо да извие, 

пилци да изпили. 

Отзела е Елка

менци с кобилица

и отиде Елка

на Дунав за вода, 

седнала е Елка

край Дунав, на брега,

писмо написала:

``Мамо, мила мамо,

ти да чакаш Елка 

вода да донесе

кога чуиш, мамо,

камен да попеи (2)

от Стара планина;

риба да продума 

от тих бял Дунава; 

кога видиш, мамо,

сама кукувица

гняздо да извие,

пилци да изпили''! 

Че се метна Елка 

в тихи бял Дунава

и се удавила. 
\end{multicols}
