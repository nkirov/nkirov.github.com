%Стойене, абре Стойене
%На седянка
%Джиджев:седенкарски

\begin{multicols}{2}
Стойене, абре Стойене,

тебе ли хвалят хората, (2)

че рано рани млад Стоян, (2)

че рано рани, надои?!
\kray

Доде дружина да стане,

Стоян надои две ведра;

доде дружина -- две ведра,

Стоян надои четири;

доде дружина -- четири,

Стоян си стадо изкара,

чи го закара, закара

на връх, на Стара планина, (2)

на самодивско кладенче --
 
тамка си Стоян завари

три жени, три магесници. 
\end{multicols}
