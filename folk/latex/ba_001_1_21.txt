%Станинине, господине
%Коледарска (на стопанин с малко момче)
%Джиджев:коледарски

%%добавям "(на стопанин с малко момче)" - по муз. запис
%%разликите между муз. запис и машинопис са големи - правя два файла
%%тук е музикалния запис

\begin{multicols}{2}
\uu{-- Станинине, господине,}
%%муз. запис започва от втория "куплет" НК

станинине, отвори ни,

ний та чухми чут болярин, 

че си имаш много жито,

много жито немерено,

немерено, нехаяно!

Отговаря стар господин:

-- Имам си го, бъх не правя,

ала и то премерено!

Отговарят събор момци:
%%муз. запис започва оттук НК

-- Станинине, отвори ни,

ний та чухми чут болярин,

че си имал много пари,

много пари, нечетяни

нечетяни, нехаяни:

ний дойдохми да ги четем,

да ги четем, да ги хаем!

Отговаря стар господин:
\kray

-- Имам си ги, бъх не правя,

ала и те прочетени,
%%те са???

прочетени и хаени!

Отговарят събор момци:

-- Станинине, отвори ни,

ний та чухме чут болярин,

имаш сина златокоса: 

ний додохме да го вземем

с нас да доди, цар да бъде,

на стол да седи, войска да реди,

войска да реди, земя да дели!

Отговаря стар господин:

-- Имам си го, бъх не правя,

ала и той малък още: 

той не можи на стол седя, 

на стол седя, войска реди,

войска реди, земя дели!

Отговарят събор момци:

-- Дай го, дай го, стар господин,

ний имаме три робинки,

три робинки, три гъркинки: 

първа ша го на стол държи,

втора ша му войска реди,

трета ша му земя дели!
\end{multicols}

\ed{Нехаяни -- производно от ``нехая'' 'бездействам, не обръщам внимание'; в случая ``нехаяни'' 'незнайни'. }
