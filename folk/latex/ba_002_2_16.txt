%Мари бульо Гаджовице
%Хороводна (хумористична, пее се на засевки вкъщи)
%Джиджев:хороводни

%%(хумористична, пее се на засевки вкъщи)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
-- Мари бульо Гаджовице, (2)

била ли си батя Гаджа? (2)

-- Била съм го й пак ша го бия, (2)

да ни ходи по седенки,
%%вместо "зам да ни ходи вечер на седенки" - според записа, за да може да се пее

по седенки, по кладенци,
%%вместо "вечер на седенки, сутрин по кладенци" - според записа, за да може да се пее
\kray

да ни намига на малките моми, (2)

на малките моми, на младите булки! (2)

Била съм го й пак ша го бия!

~~
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
