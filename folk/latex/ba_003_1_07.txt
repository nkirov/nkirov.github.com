%Бати Кольо, мамо, бати Кольо
%Хороводна (на засевки на двора)
%Джиджев:хороводни

%%(пее се на засевки на двора)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Бати Кольо, мамо, бати Кольо,

банянчето, мамо, банянчето,

баня гради, мамо, баня гради,

цвете сади, мамо, цвете сади!
%%горните два стиха намясто един: "баня гради, мамо, цвете сади" - според записа
\kray

\five Какво беше, мамо, цветено му? --

Ран босилек, мамо, й пити златор,

пити златор, мамо, й жълта ружа.

Банята са, мамо, разливаше,

цветето му, мамо, поливаше.
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}



\placedate{с.\,Кортен, общ.\,Нова Загора, 17 февруари 1966\,г.}
