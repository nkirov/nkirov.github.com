\begin{multicols}{2}
Мама Калина будеше:

-- Я стани рано, Калино, (2)

главата да си омиеш, (2)

та ситно да те оплета (2)
%%вм. "да те оплета, Калино" - може да е вариант; според записа

на овчар хляб да занесеш, (2)
%%стихът е пропуснат

на овчар и на кучета! (2)
%%стихът е пропуснат

Калина плачи, не рачи: (2)

-- Я не ща, мамо, да ида: (2)
\kray
%%добавям /2 надолу

овчарето са далеко -- (2)

на връх на Стара планина -- (2)

и имат лоши кучета! (2)

Мама Калина пак буди: (2)

-- Стани, Калино, да идеш! (2)

Калина майци пак дума: (2)

-- Мар, кък ще, мамо, да ида, (2)

нашто малкото овчарче (2)
%%нашто вм. нащо - същото, но по-четивно

лоши ми думи хортува: (2)

``Хайди Калино със мене, (2)

със мене моя да станеш, (2)

със мене у нас да дойдеш, (2)

да дойдеш, наша да станеш!''
\end{multicols}
