%Пребила й Драгана
%На жътва (и преди, и след обед)
%Джиджев:жътварски

\begin{multicols}{2}
Пребила й Драгана

два друма до царя, (2)

трети друм до баня: 
%%баня вм. баля - формула е, грешка е, долу е "бано"
%%подкрепено и от Лозанка

-- Царю нецарувал,

бано несборувал,
%%предпочитам "несборувал"

га ми не отвориш 

тъвната тъвница, (2)

люби да си видя --
\kray

почерняло ли йе,

погрозняло ли йе?

Цар Драганка дума:

-- Драгано, Драганке,

то йе почерняло,
%%тук и долу - сричка по-малко; трябва да е "то е почерняло / и е погрозняло"

и йе погрозняло --

коса му й до рамо,

брада му й до пояс! 
\end{multicols}
