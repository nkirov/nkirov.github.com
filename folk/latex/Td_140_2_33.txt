%Алтън бяла Борянинко
%На хоро (вечерно)
%Джиджев:хороводни

%%като че ли има начално "и"

\begin{multicols}{2}
-- Алтън бяла Борянинко, (2)

що си глава привързала, (2)

здрава глава с тенка кърпа, (2)

тънка кърпа мехранова -- (2)
%%стихът отсъства в ръкописа, давам го според записа
\kray

ройно ли си вино пила

или блага медовина?

-- Дружчици ле, кардашчице,
%%"друшчици" вм. "друмници" - според ръкописа; предпочитам "дружчици"

я съм вино не видела,

нето блага медовина --

я съм глава превързала,

здрава глава с тенка кърпа,

тенка кърпа мехранова,
%%"мехранова" вм. "механова" - според ръкописа; срв. и по-горе "мехранова"

от Перванови ядове!

Перван люби друго любе,

мене люби, друга гледа --

снощи вечер на герана

той хи даде огледало,

тя му даде китка невен.

~~
\end{multicols}

\ed{Мехранова -- производна от ``махрама'' тур. 'голяма кърпа за глава'.}
