\begin{multicols}{2}
Замръкна Лала биз вода,

\uu{Лали ле, Лале Вълева,}

биз вода и без вичеря,

чи дръпна Лала две стомни,

чи си на вода отиди. 

Тамка си Лала завари
\kray

до трима души хайдути,

те си на Лала думаха: 

-- Я хайде, Лале, със нази!
\end{multicols}

\sing{Пяла Смина Иванова Пиронкова}
\year{1919}
\village{Бабек}
\bornArea{Пловдивско}
\record{декември 1964}
