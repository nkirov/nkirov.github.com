%%този и следващите записи са от окръжен събор - надпяване в гр. Елхово. 2 естрада: певчески групи и танцови състави!!!

\begin{multicols}{2}
\rows
{Дойна двори мела,}
{двори като поли,}

\rows
{къща чардаклия,}
{хамбар пармаклия.}

\rows
{Иван по път вървеше}
{и на Дойна дума:}

\rows
{-- Дойне, малка моме,}
{и ти първо либе,}

\rows
{като двори метеш,}
{тъй ли не намери}

\rows
{мойте чифт кавали,}
{кавали с калъфи?}

\rows
{Дойна Иван дума:}
{-- Байне ле Иване,}

\rows
{аз да ги намеря,}
{тебе ли ще кажа --}

\rows
{на мама ще ги дам,}
{мама да ги скрие}

\rows
{в моя писан сандък,}
{при моите ризи,}

\rows
{при моите ризи,}
{ризи и сукмани!}
\end{multicols}
