%Разболя са Яна
%Жътварска (сутрин)
%Джиджев:жътварски

\begin{multicols}{2}
Разболя са Яна,
%%чувам "Яна"

Яно, моме Яно,
%%\uu{Яно, моме Яно}, 
%%коментар от Джиджев - "рефренът не се повтаря винаги"; факт е

на сарапска нива,

Яно, моме Яно,

под ярабска круша,
%%аз чувам "арабска", ама нейсе

Яно, моме Яно,

Янка бога моли: 

-- Боже, прибери ме,

Боже, отърви ме

от ярабски ръце,

стига да си тегля!

Яно, моме Яно,

Яраб богу моли:
%%пак чувам "араб"

-- Боже, дигни Янка, (2)

двамка да се вземим, (2)

двамата със Янка!
\end{multicols}

\bel{Рефренът не се повтаря винаги.}

%%\editor{Текстът е даден точно както се пее.}

\sing{Пяла: Куна Вълева Радкова}

\area{Пловдивско}
\year{1904}
\rec
