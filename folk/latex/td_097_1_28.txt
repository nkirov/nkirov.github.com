%Тук ни казаха, мома хубава
%Лазарска (в двора на къщата, не се играе)
%Джиджев:лазарски

\begin{multicols}{2}
Тук ни казаха, \uu{Лазаре,} мома хубава, (2)

мома хубава, мома гиздава -- (2)

найвън да излези, да я питаме, (2)

да я питаме, сгодена ли йе! (2)
%%предпочитам "сгодена"
\kray

\five 'Ку нй е сгодена, да я сгодиме, (2) 
%%предпочитам "сгодена", "сгодиме"

да я сгодиме навръх Великден, (2)
%%предпочитам "сгодиме"

на връх Великден, със руйно вино; (2)

ак ни е винчана, да я винчейем, (2)

да я винчейем навръх Гергьовден, (2)

\ten на връх Гергьовден със шаро агне! (2)

~~
\end{multicols}
