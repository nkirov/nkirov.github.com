%Мама Ганка дума
%На хоро (по Великден)
%Джиджев:хороводни

\begin{multicols}{2}
Мама Ганка дума:

-- Нимой ходи, Гане,

във горната махла,

чи там турци има,

турци анадолци!
%%измислена нова дума - "арнадолци" - вер. ефект от колебание между две думи: "арнаути" и "анадолци"; не пипам

Ганка мама дума:

-- Мамо, мила мамо,

дето Ганка ходи,
\kray

чи там турци няма, 

турци анадолци,

чи там има, мамо,

едно малко момче --

булгарийка свире,

свире и говоре:

``Гане, севдим Гане,

пристани ми, Гане, (2)

приследи ма, Гане!''
\end{multicols}

\sing{Пяла Пенка Петрова Славчева}
\village{с. Поляново}
\area{Хаманлийско}
\year{1901}
