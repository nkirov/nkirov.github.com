%Даньо мами проговаря
%Коледарска (на ловец ерген)
%Джиджев:коледарски

%%добавям (на ловец ерген) от муз. запис
%%разликата между звук и машинопис е толкова голяма, че разделям файла на два варианта
%%тук са и двата варианта

\begin{multicols}{2}
\singvar

Даньо мами проговаря,

\uu{Даньо льо, Данчо льо, Дан вайвода:} 

-- Мари мамо, стара майко, 

изведи ми хранян коня,

чи щи иде Даньо юнак 

татък долу във полето, 

във полето, при морето.

\textvar

Седи Даньо на чардаци,

\uu{Даньо льо, Данчо льо, Дан вайводо,} 

кон му цвили на пайванти,

сокол пищи на пармаци.

Даньо мами отговаря:

-- Мари мамо, стара мамо,

назоби ми хранен коня,

назоби го, напои го,

оседлай го, обездай го,

че ша иде Даньо юнак

татък долу във полето, 

във полето, край морето
\kray

да си гони дребна лова

яребички, перперички!

Мама Даньо отговаря:

-- Седи, Даньо, недей ходи,

че майка ти й лош сън сънила,

че си отсяках малкото пръстче,

малкото пръстче на десна ръка!

Даньо майка си не слуша,

сам изведи хранен коня,

назоби го, напои го,

оседла го, обезда го. 

Възседна го Даньо юнак,

че си взема във десница,

във десница тънък маждарак,

във левица пъстър сокол, 

че отиде лов да гони,

татък долу във полето,

във полето, край морето. 

Ходи два дни, ходи три дни,

не изгони дребна лова,

яребички, перперички,

най изгони сиво птиче. 

Птиче хвръкна, сокол трепна,

сокол трепна, кон са сепна,

кон са сепна, подплаши са,

Даньо юнак от кон литна,

че налитя на тънък маждрак.

Клета му душа й проклета,

който мама си не слуша!
\end{multicols}

\sing{Пял: Стоян Иванов Вандов}
\village{c. Кортен}
\area{Сливенско}
\year{1908}
\rec
