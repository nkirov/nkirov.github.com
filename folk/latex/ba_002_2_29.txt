\begin{multicols}{2}
-- Ходи, бульо, думай добро за меня, (2)

дано доди пирпири Пенка при меня! (2)

-- Ходила съм, думала съм, драгинко, (2)

ала не е туй момиче за тебя -- (2)
\kray

Пенка носи за сто гроша фустанче, (2)

Пенка носи за хиляда коланче! (2)

~~
\end{multicols}

\ed{Пирпири -- диалектна форма от ``перпери'' тур. 'спрегнат'. }
