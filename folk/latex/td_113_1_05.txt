%Тука са ни казали
%Лазарска (на мома и ерген) 
%Джиджев:лазарски

\begin{multicols}{2}
Тука са ни казали,

има мома и ерген:

жин\'{e}те си момата,

год\'{e}те си ергеня

дор има вино чървено,

бистра-върла ракия, (2)

дор са млади ягънца!

~~
\end{multicols}

\sing{Пяла Ирина Минчова Славова}
\village{с. Кортен}
\area{Новозагорско}
\year{1905}
\re
