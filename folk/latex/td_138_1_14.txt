\begin{multicols}{2}
-- Леле, Яно, леле пиле, \uu{мари},

леле, Яно, леле, пиле,

ти ли любиш войводата, \uu{мари},
%%"ли" вм. "ми" - по записа

ти ли любиш войводата, 

войводата от конака? (2)

-- Лельо мила, лельо драга, (2)

те ме любят, аз ги лъжа, (2)
\kray

те се канят да ми режат (2)

руса коса до петите, (2)

да я плетат на камшици, (2)

да си карат бързи коне! (2)

-- Леле Яно, леле пиле, (2)

да не ходиш по сокаци, (2)

по сокаци, по мегдани, (2)
%%свалям "по мегдани" от горния стих, добавям "по сукаци" - в името на стиха

дето турци коне водят, (2) 

коне водят и разходят! (2)

~~
\end{multicols}
