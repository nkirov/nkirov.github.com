%Поведье Иван жътвари
%На жътва (на връщане от нивата) 
%Джиджев:жътварски

\begin{multicols}{2}
Поведье Иван жътвари,

\uu{Иване, драгоманяно}, 

поведье, та ги заведе

на Бекерлия, на жътва.
%%Бекерлия вм. Бекерлии - по записа, така е и по-долу
%%колебание и в ръкописа

И Тана иска да иде,

\uu{Тане, йедничка на мама,} 
\kray

мама си Тана не пуща.

Тана си мама не слуша,

и стана Тана, йотиде

във Бекерлия на жътва. 

Приз нощта вода придошла,

всичките се издавяле,

\uu{Иване, драгоманяно}, 

мама за Тана плачеше:

-- \uu{Тане, йедничка на мама,} 

нали та, Тане, не пущах
%%Нали вм. дали - за смисъл

във Бекерлия на жътва,

а ти ма, Тане, не слуша,

и стана, Тане, йотиде! 
\end{multicols}

\ed{Бекерлия -- с.\,Бекирлие, дн. Конево, общ.\,Върбица, обл.\,Варна.}
