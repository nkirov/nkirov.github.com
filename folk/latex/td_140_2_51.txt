%В село циганки дойдоха
%Седенкарска (на зимна седянка)
%Джиджев:седенкарски

%%това май не е гласа на Джиджев

\begin{multicols}{2}
В село циганки дойдоха,

циганки черни, омайни,

у Стоянови йотишли. (2)

Стоян'ва стара майчица,

тя си циганки думаше:
%%в муз. запис "думаше" НК

-- Циганки черни, омайни, (2)

имате ль биле омайно,
%%ль вм. ли - по муз. запис - за сричките

омайно биле, разделно?

-- Имаме, бабо, имаме,

яла го скъпо даваме:
\kray

за шиник бели грошове,

за шепа жълти алтъни!

~~ 
\end{multicols}
