%Прочу сай мома Йордана
%На хоро 
%Джиджев:хороводни

\begin{multicols}{2}
Прочу сай мома Йордана,

Йордано, моме хубава,

че била много работна,

работна, Дана, имотна.

Отишла й Дана на сбора,

на сбора на св'та Троица,

Гьорги ми Суруджалийски

и той на сбора отива,

и си на Дана думаше:

-- Хайде ми, Дано, пристани, (2)

доде си млада й хубава!
\kray

Дана на Гьорги думаше:

-- Гьорги бе, любе Гьорги бе,

мене ме Стоян излъга,

Стоян ми мансарлийчето,

Стоян ми [a]рабаджийчето,

че ще ми прави, направи

до три ми върви алтъни

и три желти махмудии!
\end{multicols}

\ed{Мансарлийчето -- с.\,Мансарлии, дн. Победа, общ.\,Тунджа, обл.\,Ямбол.\\
Суруджалийски -- с.\,Суруджалии, дн. Чарда, общ.\,Стралджа, обл.\,Ямбол. 
}
