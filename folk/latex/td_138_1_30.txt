%Ой вакла, вакла, джанъм, вакла Тодоро
%На моабет
%Джиджев:трапезни

%%пренаредена стихово: заради редуването на "джанъм", "мамо" не слагам подчертавания

%\begin{multicols}{2}
-- Ой, вакла, вакла, джанъм, вакла Тодоро,

що нъще не спиш, вакла, пък дене дремиш,
%%(бел.ред. -- нощем) 

овчар ли любиш, вакла, та нъще не спиш?

-- Ой, мамо, мамо, джанъм, мащеха да си,

ще ме повярваш, мамо, що нъще не спя,

що нъще не спя, мамо, пък дене дремя: 

мене ме люби, мамо, змийка усовна,

ти яла видей, мамо, в нъща, посряд нощ,
%%чува се: "погледни" вместо "видей" НК

в нъща, посряд нощ, мамо, по злото време,
%%стихът не е фиксиран в машинописа - тук по записа

по злото време, мамо, от пенджерката --
%%в записа "каквото видиш" вм. "пенджерката"; не го променям; 
%%пенджерката е допълнителна информация

каквото видиш, мамо, да си помъльчиш,
%%"да си помъльчиш" вм. "да не продумаш" - по записа; 
%%за продумването става дума долу, така че не се губи

ако продумаш, мамо, ще ме изгубиш!
\kray

Тодорина майка, джанъм, тя е отишла

в нъща, посряд нощ, джанъм, и погледнала
%%добавям " погледнала" заради стиховата организация

и погледнала, джанъм, в малката къща,

в малката къща, джанъм, от пенджерката. 

Га е видела, джанъм, Тодора седи,

Тодора седи, джанъм, и в ръка държи

сахан калайдисан, джанъм, пълен стафиди,

а в скут хи лежи, джанъм, змийка усовна -- 

еднъж змийка хапни, джанъм, от стафидите,

еднъж змийка хапни, джанъм, дваж Тодори даде. 

Га е видела, джанъм, Тодорина майка,

не си помълчала, джанъм, а повикала:

-- Варди се, варди, джанъм, вакла Тодоро,

да не те укльоса, джанъм, змийка усовна!

Змийка укльосала, джанъм, вакла Тодора,

укльосала я, джанъм, и се изгубила. 

Изпищяла е, джанъм, вакла Тодора:

-- Ой, мамо, мамо, джанъм, ой, мила мамо,

мащеха да беше, мамо, щеше да помълчиш!

Толкова продума, джанъм, вакла Тодора,

толкова продума, джанъм, и си загина. 
%\end{multicols}

\ed{Укльосвам 'ухапвам' (за змия).}
