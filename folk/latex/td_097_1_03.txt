\begin{multicols}{2}
Откък та, Гене, залюбих, \uu{мари,} (2)

\uu{Гени ле, гюлчи цъфнало,}

сладка вичеря ни ядах, (2)

сладка са съня ни наспах (2)
%%добавям надолу /2
\kray

от твойта хубос хубавa! (2)

Аз щъ са, Гене, oбеся (2)

на вашта мала вратничка -- (2)

тейко ти да ма намери, (2)

тейко ти глоба да плати! (2)

-- Нидей са, любе, обеся (2)

на нашта мала вратничка, (2)

ами са, любе, обеси (2)
%%"любе" вм. "Гене" - напук на ръкописа: трябва да е грешка; 
%%Гена се казва момата, тук става дума за момъка

сред село, насред мегданя -- (2)

селяни да та намерят, (2)

селяни глоба да платят, (2)

и тати глоба шъ плати -- (2)

идно счупено дукато! (2)
\end{multicols}
