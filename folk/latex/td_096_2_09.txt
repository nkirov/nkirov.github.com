%Качи са Рада на люшки

%На Гергьовден (припявка) 

%%не съм убеден, че насъбирането на Гергьовден и Еньовден в една група е коректно. 
%%Общо Гергьовските и Еньовденските песни спадат към групата на коледарните, 
%%към които се отнасят коледарските, лазарските и пр. 
%%Мисля, че за Джиджев е било работно. 

\begin{multicols}{2}
Качи са Рада на люшки,

\uu{Радо льо, убавици ле,}

Радо от люшки викаше: 
%%стихът е изпуснат в ръкописа и машинописа - тук по записа

-- Де да и Стойо, да доди,

да доди да я поеми,

чи ще от люшки да падне,

да падне, да са утрепи!

~~
\end{multicols}
