\begin{multicols}{2}
Тръгнала е вакла желва,

\uu{ти-ри-ни-га-ни,}

тръгнала е вакла желва,

\uu{трам-ба-ба-ром,}

на пазари, на сеире, (2)
%%"сеире" вм. "сийре" - по записа, за яснота

де я срещна Ежо-Межо, (2)

посрещна я, прегърна я, (2)
%%прегърна вм. "целуна" - по записа, за да стане отпяването

прегърна я, целуна я (2)
\kray

между двете черни очи, (2)

между двете тънки вежди. (2)

Тя отиде при съдия, (2)

на съдия се оплака, (2)

съдията отговаря:  (2)

-- Мари, желво, вакла желво, (2)

този ерген това търси! (2)
\end{multicols}
