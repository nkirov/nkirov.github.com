\begin{multicols}{2}
Русе, Русе, тенко Русе, \uu{джанъм,}

Русе, Русе, тенко Русе,

разшета се тенко Русе, \uu{джанъм,}

разшета се тенко Русе

от конаци до чердаци,(2)

загуби си ресен гердан, (2)
\kray
%%слагам до края /2

ресен гердан, сърмен колан. (2)

Тя набеди тенко гърче: (2)

-- Гърче, гърче, тенко гърче,  (2)

дай ми, гърче, ресен гердан, (2)

ресен гердан, сърмен колан! (2)

Гърче дума тенка Руса: (2)

-- 'Ку съм нашел твоя гердан, \uu{Русе,} (2)

да се свия като гердан (2)

окол твойта бяла шийка; (2)

aко съм ти колан нашел,  (2)

да се свия като колан  (2)

окол твойто тенко кръсле!  (2)

~~
\end{multicols}
