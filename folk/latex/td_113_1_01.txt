%Тука са ни казали
%Лазарска (на мома) 
%Джиджев:лазарски

\begin{multicols}{2}
-- Тука са ни казали,

има мома хубава, (2)

речети \'{и} да йзлези (2)
%%"Речите" вм. "Ричете" - по записа, в името на разбирането.
%%за мен "ричете" е по-ясно! НК 
%%чува се: "речети" НК

да ни види Лазара (2)

как хубаво играи!

Когото си хареса,

него шъ я дадеми. 
\kray

Чи излези, излези

назлън Иван йергеня,

чи изгледа Лазара,

открай-докрай момити,

чи хареса, хареса

назлъм Пенка Попова,

Попова, Попникол'ва.

Сички Пенка ни дават,

чи йе мома хубава,

чи им води Лазара

като овен овцети,

като квачка пиленца. 
\end{multicols}

\ed{Назлън -- диалектна форма от ``назлъм, назлъ'' тур. 'който се докарва, кокетен; превзет'. }
