%Тодоро льо, чернаока

\begin{multicols}{2}
Тодоро льо, чернаока,

чернаока, бялалика,

когото си погледнала,

погледнала, болки дала,

%%следващите два стиха не са дешифрирани - давам ги според записа
болки далай запоилки,

билки, момчо льо, запоилки!
%%ти така ли ги чуваш, Николай?

Погледнала й наш Ивана,

наш Ивана, болки дала,

%%следващите два стиха не са дешифрирани - давам ги според записа
болки дала запоилки,

билки, момчо льо, запоилки!
\kray

%%следващите два стиха за мен трябва да отпаднат - те са плод на разсеяно дешифриране
%%Наш Ивана болки дала,
%%болки дала заповилки!
%%"заповилки" вм. "запоилки" - аз навсякъде го чувам така

Кълни, момчо льо, кого кълнеш,

яла мен нидей кълна,

че ми майка болна лежи,

болна лежи на леглото!

~~
\end{multicols}

\dic{Болки дала запоилки -- привлякла го към себе си.}

\ed{Запоилки 'упойващи билки'.}
