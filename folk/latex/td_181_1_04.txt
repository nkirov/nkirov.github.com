%Ти нали знаеш, куме ле
%Сватбарска (``на кума, като се предава кумовото дърво'') 
%Джиджев:сватбарски

\begin{multicols}{2}
Ти нали знаеш, \uu{куме ле,}

че ще кумуваш? --

Да си приготвиш (2)

дълга трапеза, 

златни мисали!
\end{multicols}

\ed{Мисали -- диалектна форма от ``месал'' гр. 
1. 'кърпа; кърпа, с която се постила хляба да втаса'; 
2. 'покривка за маса'; 
3. 'четвъртито платнище, в което се увива нещо, бохча'.}

\sing{Пяла Стойка Иванова Вълчева}
\village{с. Межда}
\area{Ямболско}
\year{1932}
\re
