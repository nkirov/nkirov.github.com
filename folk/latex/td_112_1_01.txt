%В село златари додоха
%Лазарска (на ерген) 
%Джиджев:лазарски

\begin{multicols}{2}
В село златари додоха,
 
додоха и кондисаха,

кондисаха й повикнаха:

-- Михале, Михал чорбаджи, (2)

\five кой има сина за сгода,
%%предпочитам "сгода"

за сгода и за женини?
%%предпочитам "сгода"

-- Никой ми ни йе за сгода,

току ми Митьо за сгода,
%%предпочитам "сгода"

за сгода и за женини! 
%%предпочитам "сгода"

\ten Я нему Таня ни дали, 

Таня на мама думаши:
\kray

-- Ни давам Таня на Митьо,

на Митьо, на хайдутина:

той няма черни биволи

\fifteen да оре черни угари!
\end{multicols}
