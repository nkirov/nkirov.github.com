%Слънце, ясно слънце
%Жътварска (като напече)
%Джиджев:жътварски

\begin{multicols}{2}
-- Слънце, ясно слънце,

кому мама рече,

рече и поръча, 

ясно да ни греиш, 

ясно по пладнина,
%%стихът е пропуснат в ръкописа, тук - по муз. записа

чи ще да изгориш

по ниви жътваре,

по поле косаче?!

-- Мамо, мила мамо, 
\kray

кого да послушам? --
%%послушам вм. послуша - за смисъл

Две са моми молят:

``Греи, греи, слънце,
%%греи вм. грей - за сричките

та да си избелим

наште тънки даре!''

~~
\end{multicols}

\sing{Пяла: Здравка Димитрова Делчева (Попконстантинова)}
\village{с. Любеново}
\area{Хасковвско}
\bornYear{1901}
