%Стойке ле, снахо Нанева
%На хоро (по Великден)
%Джиджев:хороводни

\begin{multicols}{2}
-- Стойке ле, снахо Нанева, (2)

Нанева и Чорбаджива,
%%стихът не се изпява

във дома ходила ли си, (2)

вечеря сготвила ли си? (2)

-- Свекре ле, Нане чорбаджи, (2)

каква та вечера оди? -- (2)
\kray

Д'ора е пълен и равен,  (2)

пълен и равен с хайдуте, (2)

за тебе, свекре, питаха: (2)

``Къде е Нанко чорбаджи, (2)

къде е и къде ходи?'' (2)

-- Стойке ле, снахо Нанева, (2)

мож ли ги, снахо, отмахне? (2)

-- Отмахнувам ги, свекре ле, (2)

ако ми дадеш, свекре ле, (2)

на девет снахи съдбата (2)

и на западне нахтера! (2)
%%"зандане" вм. "запдане" - според ръкописа

-- Давам ги, давам, снахо льо!  (2)

Че си Тодорка отиде (2)

и на хайдутье думаше: (2)

-- Няма го, холам, дяда ми, (2)

дядо ми, холам, отиде (2)

със кола пари да доне, (2)

на овци беглик да плата (2)

и на нивите казастра! (2)
\end{multicols}

\dic{На девет снахи съдбата -- правото тя да ги командва.\\
Казастра -- вид данък.}

\ed{Нахтера -- диалектна форма от ``нахтар'' тур. 'ключ'.}

\sing{Пяла Марина Стоянова Русева}
\village{ с. Лешниково}
\area{Хаманлийско}
\year{1902}
