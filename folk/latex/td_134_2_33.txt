%Ябре момче, момче
%На Лазарица  (на ерген)
%Джиджев:лазарски

%%начално "и"

\begin{multicols}{2}
-- Ябре момче, момче,

грижно, тa й кахърно,

чуждо, та й незнайно,
%%чува се: чождо, тъй незнайно НК

дали ти е, момче,

пушка дотежала,
%%чува се: дотежала НК

салдатлък дойдяло?

-- Мари лельо, лельо,

майчина прилико,
\kray

нето ми е, лельо,

пушка дотежала,

нето ми е, лельо,

салдатлък дойдяло. 

Вчера минах, лельо,

приз нашито село -- 

на нашата вратня

две са хора вият,

два кавала свирят

и две моми пеят. 

Мойти девет братя

кавали правеха,

с кавали не свирят,

мене жив жалеят;

мойти двете сестри

на хоро пеяха, 

а сега пък, лельо,

разплетени ходят

мене жив жалеят; 

моя стари тейно 

на пътя сидеше,

вино продаваше --

на всекиму даде

по една ми чаша, 

а на мене, лельо,

две ми чаши даде

и на мене дума:

``Пий момче, войниче,

та спомини, момче,

моя син Стояна!''

Мойта стара майка,

тя по двори ходи,

мене жив жалеи. 
\end{multicols}
