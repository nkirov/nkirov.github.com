%Забляла йе стара вакла
%Коледарска
%Джиджев:коледарски

%%според клишето трябва да е "стара вакла", но навсъкъде е "сара"; така се и чува

\begin{multicols}{2}
Забляла йе стара вакла,
%%чува се: стара! НК

\uu{рудо льо, вакло льо, рудо овчице,}
%%дали пътвото Рудо не е име на овцата?? НК

диня-нощя, посреди нощи,

запита я Ян ки[х]ия: 
%%бележка, че става дума за производно от "кехая"
%%чува се: запита НК

-- Що ми блееш, стара вакло,
%%вместо този и следващите два стиха, в муз. запис: "Що ми пеиш, саро вакло"
%%чува се: Що ми блееш, стара вакло НК

що ми блееш диня-нощя,
%%не се пее НК

диня-нощя, посреди нощи --
%%не се пее НК

дали ни си трива гладна,
%%чува се: дали ни си трива гладна НК 

дали ни си сол безсолна?
%%в муз. запис "трева гладна, сола бесолна"
%%безсолна вм. бисолна - по муз. запис, за яснота
%%чува се: дали ни си сол бесолна НК
\kray

Стара вакла отговаря:

-- Нето съм си трива гладна,

нето съм си сол безсолна --

най си блея от овчари,

от овчари сидинкяри!

~~ 
\end{multicols}

\sing{Пяла: Куна Вълева Радкова}
\village{c. Бабек}
\area{Пловдивско}
\bornYear{1904}
\rec
