%Привела са й Калина 
%Лазарска (на мома за женене) 
%Джиджев:лазарски

\begin{multicols}{2}
Привела са й Калина, \uu{Лазаре,} 

приз два дола дълбоки, 

приз два дола дълбоки, \uu{Лазаре,} 

приз четири шароки. 
%%"шароки" вм. "шарони" - по записа, смисълът се изяснява

Сички моми миная, (2)
%%тук и долу добавям "/2" - не се пее, но е в логиката на останалата структура на песента

речи Калина да мини, (2)
%%"Калина" вм. "Пена" - надолу наместо Калина се появява "Пена"; 
%%надолу в машинописа Танас, в записа - "Петър"; 
%%заменям по записа "Пена" с "Калина" и "Танас" с "Петър"; 
%%другите две имена могат да останат като варианти???

привари я млад Петър, (2)
%%"привари" вм. "спривари" - по записа, чува се ясно

та \`{и} скъса мангъри. (2)
\kray

Седна Калина да плаче, (2)

Петър Калина говори: (2)

-- Мълчи, Калино, ни плачи, (2)

двама сми ги скъсали, (2)
%%предлагам да заменим "сни" със "сми" - за четивност; нищо не се печели от подобна диалектна коректност

двама ще ги нанижим!
\end{multicols}

\sing{Пяла Стояна Минчова Кюркчийска (Рангелова)}
\village{с. Пъдарско}
\area{Пловдивско}
\year{1912}
\rec
