%Мама Балаза думаши
%Хороводна (на засевки)
%Джиджев:хороводни

%%(пее се на засевки)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските
%%Име на песента според Джиджев: Де ти е Балазо герданя НК

\begin{multicols}{2}
Мама Балаза думаши:

-- Балазо, мила мамяна,

де ти, Балазо, гирданя,

гирданя, мирджанлията?

Балаза дума мама си: 
%%Балаза мами думаше:
%%Балаза дума мама си: - според музикалния запис НК
 
-- Дадох го, мамо, дадох го (2)
%%сменям "го" вместо "ги" - според записа и в името на смисъла 

на голи й боси цигани

работа да ми направят --
%%"работа" в случая трябва да има смисъл на "магия" - дали да не го поясним с бележка???

да зема, мамо, хаджия,

хаджия, баш чорбаджия,

хаджия, хаджи Василя!

~~
\end{multicols}

\ed{Мирджанлията -- диалектна форма от ``мерджанлия'' тур. 'мерджанов, коралов'.}

\sing{Пял Стоян Иванов Вандов}
