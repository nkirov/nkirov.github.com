%Сяла й мома ран босиляк
%Коледарска
%Джиджев:коледарски

\begin{multicols}{2}
Сяла й мома ран босиляк,

\uu{ей, момни ле, момке,}

\uu{вият две калинки,} 

ем го сяла, ем му пяла: 

-- Да припадне тъмна мъгла,

из мъглата ситна руса,

да нароси ран босиляк
%%чува се: да нароси НК
\kray

А овчар \'{и} отговаря:

-- Сей го, сей го, малка моме,

аз щи пусна шилетата

да опасат ран босиляк!

Малка мома отговаря:

-- Аз щи стана люта слана

да изтровя шилетата!

~~
\end{multicols}

\sing{Пяла: Куна Вълева Радкова}
\village{c. Бабек}
\area{Пловдивско}
\bornYear{1904}
\rec
