%Чичо на Неда тихом говори
%Сватбарска (``когато месят хляба'')
%Джиджев:сватбарски

\begin{multicols}{2}
Чичо на Неда думаше:
%%чува се: тихом говори, както е и под нотите НК 

-- Недо мо, мила Недо, чичова,

защо си рано, Недо, ранила
%%чува се: защо си рано, Недо, ранила НК
%%и така и по-долу НК

по двори ходиш, Недо, и плачеш:

двори ли нямаш, Недо, да ходиш,

къща ли нямаш, Недо, да живейш

или ти любе, Недо, нифеля?
%%стихът не е изпят

Неда на чичо тихом говори:
%%в записа: "Неда на чича тихом говори" - не променям

-- Кога ма питаш, чичо, да кажа,
%%в записа този и следващият стих не са изпяти

да кажа, да та, чичо, ни лъжа:

и двори имам, чичо, да ходя,
\kray

и къща имам, чичо, да живея,

ами ми либе, чичо, нифеля --
%%???навиля??? Трябва да се изясни значението на глагола и да се уеднакви тук и горе

то е [х]ем глухо, чичо, [х]ем нямо!

~~ 
\end{multicols}

\editor{Нифеля 'нефеля'.}
