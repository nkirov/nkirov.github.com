%Моми китки вият
%На Лазарица (из пътя -- от едната къща до другата)
%Джиджев:лазарски

\begin{multicols}{2}
Моми китки вият,

по пътя ги фъргат,

овчари ги берат,

на овци ги дават, (2)

люто ги кълняха:

-- Яжте, пущини, яжте,
%%пущ'ни вм. пущини - по записа, в името на изпеваемостта
%%не съм съгласен! НК

да са ни наядете, (2)
%%предлагам: наядете - късо ая звучи като й; найдете е друга дума НК

до идна да изпукате --
%%идна вм. ида - по записа, заради смисъла
\kray

на хоро да идим,

с любе да се видим!

~~
\end{multicols}

\sing{Пяла Минка Господинова Вълева}
\village{с. Скалица}
\area{Ямболско}
\year{1928}
\re
