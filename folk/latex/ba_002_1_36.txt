\begin{multicols}{2}
\rows
{-- Алтън бяла Борянчице, }
{алтън бяла, баш убава,}
%%май се чува "баш хубава", а не "дваж хубава", но не пипам

\rows
{алтън бяла, баш убава, }
{що ти й глава превързана:}
%% чува се: привързана НК

\rows
{\five руйно ли си вино пила }
{или сладка медовина?}
\kray

-- Не съм руйно вино пила,

нито сладка медовина,

най ми й глава превързана

\ten от ядове Първанови.

Мене ми са вяра не хващаше,

дорде снощи сама видях,

сама видях с очите си

Първан люби друго люби,

\fifteen Първан Стойни стомни държи,

пък Стойна му китка вие!

~~
\end{multicols}
