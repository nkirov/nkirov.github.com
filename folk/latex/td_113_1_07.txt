\begin{multicols}{2}
Която ни й кумата,

напред, напред да върви, (2)

да примите дворове, (2)

да нареди столове, (2)

да насядат попове, (2)

попове й попадие!  
\end{multicols}

\sing{Пяла Ирина Минчова Славова}
\village{с. Кортен}
\area{Новозагорско}
\year{1905}
