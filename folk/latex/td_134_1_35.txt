\begin{multicols}{2}
Раде вайвода думаше:

-- Любе вайводо, вайводо, (2)

стани се, любе, поди си, (2)

че ми йогиня догоря, (2)
\kray

на йедна главня остана: (2)

бар да йе главня мишова, (2)

ами е главня върбова! (2)

Вайвода Раде думаше: (2)

-- Любе Раде ле, Раде ле, (2)

мар, сиди, Раде, да сидим, (2)

да сидим, да хортуваме -- (2)

аз ще си кавал разсека (2)

да си огъня накладеш! (2)
\end{multicols}
