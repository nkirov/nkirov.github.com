\begin{multicols}{2}
Дуйни, дуйни, вятро,

вятро Тунджалия,

та разнеси, вятро,
%%в ръкописа е ясно: "разнеси" вместо "размести" НК 

мойте ясни глася

нагоре й надолу -- 
%%стихът е пропуснат в ръкописа - тук по муз. запис

да ма чуйе, вятро,

мойта стара майка,

да чуйе, да дойде,

да ма види, вятро,
\kray

на какво съм място:

дене жетва жена,

ноще оран ора!

~~
\end{multicols}
