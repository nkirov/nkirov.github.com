\begin{multicols}{2}
Покачила съ й Драганка (2)

на ершел-вершел високо
%%????

от согяр вода да зима.

Согяра се е поронял, (2)

падна Драганка в Дунава. (2)
\kray

Тя се изясно провикна: (2)

-- Нямам ли нийде някого

да дойде да ме извади?

Де я й зачула мама \'{и}, (2)

тичаше, виком викаше:

-- Дръж се, Драганке, не бой се, (2)

аз ида, ша та извадя! (2)

Вървя до брягом, възпря са, (2)

порони сълзи, върна са. (2)

Тичом тичаше тейно \'{и},

тичаше, виком викаше:

-- Дръж се, Драганке, не бой се!

Вървя до брягом, възпря са,

порони сълзи, върна са.

Тичом тичаше любе \'{и},

тичаше, виком викаше:

-- Дръж се, Драганке, не бой се,

аз ида ша та извадя!

Че във Дунава нагази,

Драганка 'т Дунав извади.
\end{multicols}

\bel{Тонът ла е много нестабилен.}
