%Пила Яна сношна йода
%На хоро (вечерно)
%Джиджев:хороводни

\begin{multicols}{2}
Пила Яна сношна йода,

\uu{мари Яно, сношна йода,}

сношна йода геранлия,

\uu{мари Яно, геранлия,}

та изпила люта змийка --

във гърло хи зимовала,
\kray

под йезикя летувала.

Янин тейно оброк даде:
%%"оброк" вм. "обрък" - по ръкописа; няма смисъл да измисляме думи

-- Който може да замине

от най страна бял Дунава,

да набере зелен брешнель,

да го счука на бял камен,

да напои Яна мома!

Ни са найе никой-някой,

найело са, найело са

царювото, най-малкото,

та набера зелен брашнель,

та го счука на бял камен,

та напои Яна мома.
\end{multicols}

\ed{Брашнель, брошнель -- вер. диалектна форма на ``бръшлян''.}
