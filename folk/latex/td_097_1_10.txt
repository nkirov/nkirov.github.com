%Снощи залюбих млади Марко
%На хоро (пролетно, га бирем гюл)
%Джиджев:хороводни

%%добавката "(пролетно, га берем гюл)" е моя от ръкописа и машинописа

\begin{multicols}{2}
Снощи залюбих млади Марко

и снощи му пристанах, (2)

той ма ю тях заведи (2)

и отдалеч повика: 

\five -- Я излез, мамо, найвънка,

че ти водя отмяна; (2)

на теби мамо отмяна, 
%%не се пее в записа

а на мени примяна!
\kray

Кога излези майка му,

\ten в ръце носи две чаши: 

идната чаша ракия,

а другата отрова --

която беше ракия,

тя я на Марко подаде;

\fifteen която беше отрова,

тя я на мен подаде.

Виднаж сръбнах, дваж викнах: 

-- Олеле, Марко, олеле,

майка ти ма отрови

\twenty с идна чаша отрова!
\end{multicols}
