%Коледарска
%Джиджев:коледарски

%%тази и други коледарски песни от 97 ги няма на първи коректури

\begin{multicols}{2}
Фръкнали са, \uu{момни ле,} два гълъба, (2)

горе фърчат, долу гледат, (2)

долу гледат де да кацнат. (2)

Кацнали са у момини,

у момини равни двори. 
\kray

Малка мома двори мите,

двори мите, с митла маха,

с митла маха да пропъди,

да пропъди два гълъба.

Не са били два гълъба, (2)

най са били два сгледаря,

два сгледаря, коледара.

~~ 
\end{multicols}

\sing{Пяла: Недялка Донева Йонкова}
\village{c. Розовец}
\area{Пловдивско}
\bornYear{1915}
\rec
