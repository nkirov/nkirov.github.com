%Стоян са юнак зажени
%Сватбарска (``когато отиват за булката, когато тръгва колата'')
%Джиджев:сватбарски

\begin{multicols}{2}
Стоян са юнак зажени,

той тежка сватба повдигна,

калеса много сватове --

педесе коля сватове.

Тръгнали булче да взимат 

далеко срещу бял Дунав, 
\kray

вървели, що са вървели,

га до бял Дунав допрели,

бял Дунав мътен дотече,

та ми отнесе, отнесе

на кума -- серес биволе,

на девер -- жълта бъклица,

на зълва -- перо сребърно. 
\end{multicols}

\sing{Пяла Здравка Димитрова Делчева (Попконстантинова)}
\village{с. Любеново}
\area{Хасковско}
\year{1901}
\re
