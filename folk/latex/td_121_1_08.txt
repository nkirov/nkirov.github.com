\begin{multicols}{2}
Забравил бях да се женя, (2)

\uu{Добриян, Добрийенино, иху,}

\uu{Ичеренино, иху,}

науми ме стара майка: (2)

-- Ожени се, клети сине, (2)

че момите поскъпнаха: (2)

една мома за хиляда, (2)

със косите за два града; (2)

един ерген за кош плява, (2)

и то да ми барим плява, (2)

пък то пуста ръжаница! (2)

~~
\end{multicols}

\ed{Ичера, Ичеренин -- с.\,Ичера, общ.\,Сливен.}
