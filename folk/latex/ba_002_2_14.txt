%Калафирьо руян, броян
%Хороводна (пее се на засевки на двора)
%Джиджев:хороводни

%%(пее се на засевки на двора)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
\rows
{-- Калафирьо руян, броян, }
{като растеш на планина,}

\rows
{като растеш на планина, }
{на планина, нависоко,}

\rows
{гори ли та люта слана, }
{пали ли та тъмна мъгла,}
\kray

\rows
{като мене Руса мома?}
{Руса мома росен брала,}

\rows
{из росена намеряла}
{се овчари, воловари.}

\rows
{Отговаря Руса мома:}
{-- Авърко льо, брат да ми си,}

\rows
{воловарко ми й побратим,}
{хайде да ма изведете}

\rows
{от таз гора яворова,}
{яворова, габърова!}

\rows
{Яворово огън гори,}
{нека гори пущината,}

\rows
{да изгори мойта майка,}
{мойта майка й моя баща,}

\rows
{че храняли много деца,}
{много деца момичета!}
%%"храняли" вместо "храняла", за да възстановя смисъла

\rows
{В сяко село мома дали,}
{за мен село не остана,}

\rows
{мене дали на конаря,}
{на конаря, камиларя,}

\rows
{да му паса камилити,}
{камилити, краставити!}

~~
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
