%Девер буля дума
%На хоро (по Великден)
%Джиджев:хороводни

\begin{multicols}{2}
Девер буля дума:

-- Бульо, мила бульо, (2)

мар, я щъ да ида (2)

ю гора зилена (2)
%%"ю" вместо "ъв" - според записа, колебания на Джиджев в ръкописа; "ъв" е трудно четивно

със върли хайдуте, (2)

хайдуте, сеймене!
\kray

Буля девер дума:

-- Деверьо Стояне,

седи, нимой ходи

из гора зилена

със върли хайдуте,

хайдуте, сеймене --

върло зъ та хванат,

мъки зъ та мъчат!

Девер буля не слуша,

той стана, отиде

из гора зелена

със върли хайдуте,

хайдуте, сеймене. 

Върло го фанали,

мъки го мъчили.

Девер буля дума:

-- Бульо, мила бульо,

продавай, залагай,

девер да откупиш

от върли хайдуте, 

хайдуте, сеймене!

Буля девер дума:

-- Деверьо Стояне,

продавах, залагах,

нищо ми не остана,

само ми остана 

булкинското було, (2)

и дятенце в ръки!
\end{multicols}


\sing{Пяла Марина Стоянова Русева}
\village{ с. Лешниково}
\area{Хаманлийско}
\year{1902}
