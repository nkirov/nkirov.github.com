\begin{multicols}{2}
-- Енке ле, мари лелина,

знаеш ли, Енке, помниш ли,

га беше, Енке при леля: 

бяла бе като кадъна,

цървена като ябълка,

сега се, Енке, ожени,

поцърня, Енке, потемня,

като църната къпина, (2)

като жълтата меранза?!
\kray

-- Знаеш ли, лельо, знаеш ли,

ние сме девет етърви,

кога на нива идеме,

нива на никак ни фата;

кога са вечер върнеме,

пътя на никак ни фата;

кога си в къщи идеме,

сякой си земе дитенце,

сякой си люшка люлчица,

я нямам дете да зема,

я зема, лельо, ведрите,

че ида крави да доя

с най-малкото си деверче!

Доде кравите издоя,

доде телците отлъча,

петлите по дваж пропяват.

...
\end{multicols}

\ed{Меранза -- диалектна форма от ``мирамза'' 'нерандза, вид портокал, Citrus aurantica'.}
