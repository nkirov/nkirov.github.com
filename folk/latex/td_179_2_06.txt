%Заспало ми й дитенце
%На Лазар (на малко дете)
%Джиджев:лазарски

\begin{multicols}{2}
Заспало ми й дитенце

под бял-чървен триндафил,
%%чува се: триндафил НК

под морова люляка.

Люляката цъфтеше,

цветовете й падаха,
%%и с ударение трябва да се пее, тука мисля, че е й НК

майка му го будеше:

-- Стани ми, стани, дитенце,

маминото момченце,
\kray

да си видиш Лазара, 

как е Лазар применен --

се със булска примяна!
%%"се" вм. "съ" - за малко повече разбиране
\end{multicols}

\ed{Моров -- диалектна форма от ``морав'' 'виолетов'.}

\sing{Пяла Минка Господинова Вълева}
\village{с. Скалица}
\area{Ямболско}
\year{1928}
\re
