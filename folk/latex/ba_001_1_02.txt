%Тъка мома тънко дари
%Коледарска (пее се на мома)
%Джиджев:коледарски

%%добавям "(пее се на мома)" - по муз. запис

\begin{multicols}{2}
Тъка мома тънко дари,

\uu{Владино, момни ле, ни вреда, вреда,}
%%"не вреда, вреда" да не е "не [е] вреда вреда"
%% чува се ясно "ни вреда" НК

\uu{ни в нова града, колади ле,}

тъкала го, изткала го, 

че отиде да го бели,
%%отиде вм. отишла - по муз. запис; разказва се в свидетелско наклонение

да го бели на бял Дунав. 
\kray

Дунав тече мътен-кален,

че повлечи малка мома,

блъскаше я от бряг във бряг. 

Покрай върви вакъл овчар,

мома му са жално моли:

-- Вляз, гидийо, извади ма, 

ша те даря добра дара,

добра дара, вита китка! 

Отговаря вакъл овчар:

-- 'Ко щях китка, извих ща си!

Пак са моли малка мома:

-- Вляз, гидийо, извади ма,

ша та даря добра дара,

добра дара, тънка риза! 

Отговаря вакъл овчар:

-- 'Ко щях риза, купи щях си! 

Пак са моли малка мома:

-- Вляз, гидийо, извади ма,

ша та даря добра дара,

добра дара, сама себе!

Запретна са вакъл овчар,

че нагази във бял Дунав,

та извади малка мома. 

Заведи я във дома си,

отдалече се провикна:

-- Н\'{a} ти, мамо, отмяната,

отмяна да та отмени

от тези тежки хизмети!

~~
\end{multicols}

\ed{Измет 'развалени, изхвърлени неща, смет, боклук'; 'слугуване'.}

\rec
