%Джиджев:седенкарски

%%начално и??

\begin{multicols}{2}
Раде вайвода думаше:

-- Любе вайводо, вайводо,

стани си, любе, иди си,

че ми огина догоря,

на йедна главня остана:

бар да я главня мишова,
\kray

ами я главня върбова!

Вайвода Ради думаше:

-- Мар, сиди, Ради, да сидим,

да сидим, да хортуваме!

~~ 
\end{multicols}

\ed{Мишова -- диалектна форма от ``мешов'' 'дъбов'.}
