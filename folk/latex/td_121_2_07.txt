%Чи стана Бижо Иванов
%Хороводна (мъдр\'{о}то) 
%Джиджев:хороводни

%%добавям (мъдр\'{о}то) - от ръкописа и машинописа

\begin{multicols}{2}
Чи стана Бижо Иванов,

чи слези долу в куп\'{я}та, (2)

отряза атлаз за хустан, (2)

моров гюмрусик за илек, (2)

Чи у Златкини отиде,

у Златкини Рангювчина, (2)

а чи на Златка думаши:

-- Златки ле, булка хубава,

уший ги, Златке, носи ги, (2)

за мойо здрави, за твойо! (2)

Отде го зачу Бижов'ца,

лух\'{у}са на двайси деня, (2)

понеси дилаф и ръжен,

чи у Златкини отиде,

а чи високо извика:

-- Бижо льо, дърто магари,

чи мяза ли ти, бре, Бижо,

по чузди пр\'{о}тки да хлопаш, (2)
%%вероятно е нужна бележка, която да изясни, че "протки" значи "портки", "портици"

при чузди булки да ходиш! (2)
\end{multicols}

\ed{Гюмрусик 'вид плат'.\\ 
Лух\'{у}са -- диал. форма от лехуса.\\
Хустан -- диал. форма от ``фустан''.}
