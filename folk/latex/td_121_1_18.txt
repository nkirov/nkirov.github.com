\begin{multicols}{2}
Забравил бях да са женя, (2)

\uu{Добриян, Добрийенино, иху,}

\uu{Ичиренино, иху,}

науми ми стара майка: (2)

-- Ожени са, клети сине, (2)

чи момите заскъпнаха -- (2)

идна мома за хиляда, (2)

със косъта за два града; (2)

идин ирген за кош плява, (2)

и тя да ми й барем плява, (2)

пък тя пуста ръженица! (2)

~~
\end{multicols}

\ed{ Ичиренин -- с.\,Ичера, общ.\,Сливен.}
