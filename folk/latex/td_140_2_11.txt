%Дельо на Дафа думаше
%На хоро (вечерно)
%Джиджев:хороводни
%%колебание "Делю" - "Дялю" - според мен трябва да се уеднакви

\begin{multicols}{2}
Дельо на Дафа думаше:

-- Любе Дафе ле, Дафе ле, (2)

стани ми, любе, отвори (2)
%%"отвори" вм. "отори" в ръкописа - според записа, чува се ясно, дори по-скоро "йотвори"; няма нужда да произвеждаме нечетивност

вашата мала вратица (2)

със твойта десна ръчица!
%%"твойта" вм. "тойта" - според записа
%%да се допише надолу - всички стихове /2 НК
\kray

Дафа на Дяла думаше: (2)

-- Ни мога, любе, да стана -- (2)

майка ми лежи до мене, (2)

руса ми коса под майка! (2)

Подай си, любе, ножчето, (2)

косата да си отрежа, (2)

да стана да ти отворя, (2)
%%"отворя" вм. "оторя" - според ръкописа

нашата мала вратица (2)

със мойта десна ръчица! (2)

Делю на Дафа думаше: (2)

-- Любе Дафо ле, Дафо ле, (2)

нимой си ряза косата, (2)

тебе е коса прилика -- (2)

кога на хоро излезем, (2)

нали ща речат хората: (2)

``Калодка Дельо и Дафа (2)

кък са двамата прилика!'' (2)
\end{multicols}

\dic{Kалодка -- избрани, хубави, един за друг.}

\editor{Колебание: Делю -- Дялю.}

\sing{Пяла Гина Ангелова Георгиева}
\village{ с. Лешниково}
\area{Хаманлийско}
\year{1936}
