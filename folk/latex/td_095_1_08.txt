%Насряд са село бирея
%Хороводна (лятно)
%Джиджев:хороводни

%%(лятно) е моя добавка от ръкописа и машинописа

\begin{multicols}{2}
-- Насряд са село бирея

илядо кол\`{я} биглишки, (2)

илядо, мамо, и двеста,

и двеста, мамо, и триста,

меника, мамо, кания,

кания, мамо, молия

млад керванджия да стана:
\kray

напреж керване да вода,

че ми й колата жилязна

и ми са руси биволи --

със копринени чулове,

със римизени юлари!

Мама Стояну думаше: 

-- Синко Стоене, Стоене,

ни ходи, синко, керванджи --

керванджи й е лоша работа,

лоша е, сино, проклета: 

де да е вода дълбока,

керванджи ще я нагази;

де да е върла ракия,

керванджи ще я напие!

Стоян мама си не слуша,

отиде млад керванджия.
\end{multicols}

\sing{Пяла Тота Илиева Кутанска}
\year{1909}
\village{Борец}
\bornArea{Пловдивско}
