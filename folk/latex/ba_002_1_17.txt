%Калуда по двор ходяши

\begin{multicols}{2}
Калуда по двор ходяши,

\uu{Калудо, клето сираче,}

и рони сълзи като дъжд,
%%наместо "и дребни сълзи ронеше" - според записа

Караджа дума Калуди:

-- Калудо, любе Калудо,
\kray

що по двор ходиш и плачеш?

Калуда дума Караджа:

-- Караджа, любе Караджа,

като ме питаш да кажа,

да кажа, да те не лъжа.

Като съм била мънинка,

мънинко дете в пелени,

при мене дошле, Караджа,

три жени, три орисници --

първата рекла, Караджа:

``Хайде да вземем туй дете!'';

втората рекла, Караджа:

``Хади да земем майка му!'';

третата рекла, Караджа:

``Нека да седи детето,

нека да седи й майка му,

да храни, да го отхрани,

да варди, да го отварди,

хубава мома да стани,

да расте, да си порасте,

хубава мома да стане,

да стане, да се ожени,

с Караджа да се събере,

тогаз ша земем детето,

да разплачеме майка му,

да разтъжиме Караджа!''
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
