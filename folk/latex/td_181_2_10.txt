%Я изгрей, слънчо, я изгрей
%Сватбарска (``когато прибулват булката в неделя сутринта'') 
%Джиджев:сватбарски

\begin{multicols}{2}
Я изгрей, слънчо, я изгрей

рано ми рано, в ниделя (2)

да видиш, слънчо, да видиш, 

чи как са чедо разделя (2)

от майка, чедо, йот баща, (2)
\kray

от мили братя и сестри! (2)

Ти как ще речеш, Марийка, (2)

на чузда майка ``мале мо'', (2)

на чузди баща ``техно бе'', (2)
%%???техно бе

на чузди братя ``драгинко'', (2)

на чузди зълви ``калинко''?! (2)
\end{multicols}

\village{с. Бояджик}
\area{Ямболско}
\year{1932}
\re
