\begin{multicols}{2}
-- Мар, хайде, дружко,

дружко, кардашко,

дружки та рукат

за студна вода!

-- Вървите, вървите,

дружки кардашки,

няма да дойда --

девер ми дойде, 

було ми метна (2)

на бяло лице!
\end{multicols}
