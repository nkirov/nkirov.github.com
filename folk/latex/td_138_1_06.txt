%Хранила Добра, гледала
%Хороводна (пролетно)
%Джиджев:хороводни

%%добавям (пролетно) - от ръкописа и машинописа

\begin{multicols}{2}
Хранила Добра, гледала

едно пиленце, славейче

в тяхната равна градинка,

на тая вишна-чурешка,
%%чува се: чурешка НК

храни го Добра, гледа го
%%липсва в машинописния текст НК

та че се Добра оглави,
%%"та че" вм. "че га" - по записа, да не се променя смисъла

оглави и ще се жени.

Славейче Добри думаше:

-- Вземи ме, Добро, скрий ме

във твоя сандък булкенски,
\kray

при твоя чеиз копринен,

че сутрин рано да пея,

та рано да те събудам!
%%"та" вм. "тя" - да не се променя смисъла

Твоята, Добро, свекърва

и голямата етърва,

тие щат, Добро, да речет:

``Та нашта булка, младата,

мар, била ранобудница,

та толко рано станува!''
\end{multicols}
