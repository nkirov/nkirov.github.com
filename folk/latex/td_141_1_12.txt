%Жътварска (по обед)
%Джиджев:жътварски

\begin{multicols}{2}
-- Слънце, ясно слънце,

кому мама рече
%%кому вм. мене - по муз. запис, в името на смисъла

силно да не печеш,
%%стихът не е изпят

ясно да не грейеш, (2)

че ми изгоряха

на нива жътвари,

на чеир косари!

~~
\end{multicols}

\ed{Косар -- в случая диалектна форма от ``косач". }
