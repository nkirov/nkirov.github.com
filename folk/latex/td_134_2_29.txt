\begin{multicols}{2}
Похвали са Димо,

Димо болярина, (2)

Димо златарина:

-- Дето яз го имам,
%%чува се: го НК

той го царя няма,

царя, господаря! (2)
%%чува се: господаря НК

Димови комшии
\kray

каил ни станали,

на царя казали,

царя хабер прати:

-- Да дой, Димо, да дой,

да дой да го питам, 

какво има Димо

йот меня повиче,

йот меня по-зиде!

Димо царя дума:

-- Имам, царю, имам

йот тебя повиче,

повиче, по-зиде -- 

яз си имам, царю,

дор девят ми сина

...
\end{multicols}
