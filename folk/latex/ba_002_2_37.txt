%Беляла йе Драганка
%Хороводна (пее се на засевки на двора)
%Джиджев:хороводни

%%(пее се на засевки на двора)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Беляла йе Драганка

тънки бели дарове (2)

с нейнити верни хайчици --

всичките белят и пеят,

пък тя бели и плаче. (2)

Хайчици я питаха:

-- Хайчице ле, Драганке,

мо, каква й твойта работа, 
%%стихът не се чува в записа

\rows
{всичките белим и пеем,}
{пък ти белиш и плачеш:}
\kray 

дали ти й майка мащеха (2)

или ти й татко пастрока?

-- Нито ми й майка мащеха,

нито ми й татко пастрока, (2)
%%"пастрока" вместо "пъстрока" - уеднаквявам с по-горе

най ми люби забягна --

стана дор шес месеца

нито си люби дохожда,

нито ми хабер провожда

за друг да се оженя!
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
