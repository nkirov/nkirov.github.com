%Кърпачо льо, люби ле
%Хороводна (пее се на засевки)
%Джиджев:хороводни

%%(пее се на засевки)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
\rows
{Кърпачо льо, люби ле,}
{закърпи ми чехлити,}

\rows
{че ма й мама сгодила}
{в долната махла за попа,}

\rows
{в долната махла за попа,}
{във горната за дяка,}

\rows
{във горната за дяка,}
{покрай нази за гърка.}

Не ща попа, чи тропа,

не ща дяка, чи кляка,

не ща гърка, чи мърка,

най ша взема гайдаря
%най ща, любе, гайдаря
%% в музикалния запис се чува: най ша взема гайдаря НК
\kray

да свири да са наиграя!
\end{multicols}

\ed{Дяк  'ученик'; 'грамотен човек, който пее в черквата'.}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
