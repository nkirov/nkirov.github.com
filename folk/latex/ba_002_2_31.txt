%Мари Радо, бяла Радо
%Хороводна (пее се на засевки на двора)
%Джиджев:хороводни

%%(пее се на засевки на двора)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
-- Мари Радо, бяла Радо,

ти ли любиш вайводата, (2)

ти ли него, той ли тебя?

Отговаря бяла Рада:

-- Опустял ми й тоз вайвода,

кой го люби, кой го види,

само ходи да се хвали:

``Мене Рада сама люби

и сама ми китки дава,

оскуба си ран босилек

да го бере, да го вие, (2)
%%стихът е пропуснат в машинописа, има го в записа

да го вие вити китки, (2)

вити китки, босилкови, 
%%стихът е пропуснат в машинописа, има го в записа
\kray

да го праща по другари,

по другари, по овчари!''
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
