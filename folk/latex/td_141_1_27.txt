\begin{multicols}{2}
-- Скрийте ма, мамо, скрийте ма

в нашата равна градинка,

под наший червен трендафил!

-- Не бива, сестро, скриване,

най бива, сестро, даване,
\kray

сватове на двор дойдаха!

~~
\end{multicols}
