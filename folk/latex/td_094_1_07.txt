%Край чукари зилени панари
%На жътва (на връщане от нивата) 
%Джиджев:жътварски

\begin{multicols}{2}
Край чукари зилени панари, (2)
%%в нотите "парнари" - чува се "панари", така е и в машинописа и ръкописа

на панари три сини чадъра, (2)

под чадъри млади момци лежат, (2)

млади момци, млади македонци, (2)
%%стихът не е изпят

помижду им млад войвода лежи, (2)
\kray

той на момци ти[х]ом проговара: (2)

-- Ой ва вази, момци македонци, (2)

тази вечер тука ще нощувам, (2)

утре вечер Вардара ще минем! (2)
\end{multicols}

\sing{Пяла: Стояна Петкова Докова}
\year{1902}
\village{с. Пъдарско}
\area{Пловдивско}
\rec

\ed{Панар -- нефиксирана диалектна дума; според контекста може би 'извор'.}
