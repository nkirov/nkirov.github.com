%Заспало ми й дитенци
%Лазарска (на малко дете) 
%Джиджев:лазарски

\begin{multicols}{2}
Заспало ми й дитенци, \uu{Лазаре,}

заспало ми й дитенци, 

под бял-червен триндафил, \uu{Лазаре,}

под бял-червен триндафил,

майчица го будеши: (2)

-- Стани ми, стани, дитенци, (2)

чи ми иди млад Лазар, (2)
\kray

с кaкво щи гo пoдарим? (2)

-- Него щи гo подарим (2)

шъпа бели грошови! (2)

~~
\end{multicols}
