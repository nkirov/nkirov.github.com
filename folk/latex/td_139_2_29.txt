%Вила се лоза гроздова
%Сватбарска (``на пряпола'')
%Джиджев:сватбарски

%%добавям (на пряпола) - от ръкописа

\begin{multicols}{2}
Вила се лоза гроздова,

около града Дервеня,
%%стихът отсъства в машинописа, добавям го от нотите и записа

не била лоза гроздова,

най била Енкя Дервенкя,

с побратима си хортува,

с побратим, мили Милушо:
%%стихът не е изпят

-- Кога за мене дойдете,

много сватове ялате --

малко съм мома ходила, (2)

много съм даре готвила:

на всяка сватя и дара,

на кумовете дор до два!
\end{multicols}

\sing{Пяла Доба Петкова Гегова}
\village{с. Брягово}
\area{Хасковско}
\year{1925}
\re
