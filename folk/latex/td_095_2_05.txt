%Добра на вода ще иде
%На хоро (пролетно, право хоро)
%Джиджев:хороводни

%%(пролетно, право хоро) е моя добавка от ръкописа и машинописа

\begin{multicols}{2}
\bbr
Добра на вода ще иде

\uu{Добро льо, мома хубава,}
%%пее се "мума", макар в нотите да е изписано "мари"
\ebr

из тези тесни сокаци,

колкото тесни, дваж кални, 

\five и на каваци -- бюлбюрци. 
%%вместо "и на Каваче бюлбюрче" - възстановявам смисъла, по записа

Бюлбюрчи пейе-говори: 
%%"пейе" вм. "пей" - по записа, възстановявам стиховата организация 5 + 3

-- Хай, да ми, Добро, пристаниш!

Добра на бюлбюр думаше: 

-- Как ще ти, бюлбюр, пристана,
\kray

\ten че ти си птиче хвъркато, 

къде ще да ме заведеш: 

че ти си нямаш ни майка,

че ти си нямаш ни баща,

че ти си нямаш ни къща,

\fifteen ни къща, бюлбюр, ни двори,

къде ща да ме заведеш?!

~~
\end{multicols}

\bel{Често си бемол е по-висок.}

\ed{\el{Бюлбюр} -- диалектна форма от ,,билбюл`` {\em тур.} 'птица, славей'.}
