%Провикна са зилен кръвен здравец
%Хороводна (пее се на засевки на двора)
%Джиджев:хороводни

%%(пее се на засевки на двора)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Провикна са зилен кръвен здравец,

провикна са от Стара планина:

-- Много здрави на малкити моми,

нека додат зилен да ме бьерат, (2)

\rows
{нека додат зилен да ме бьерат, }
{зелен кръвен здравко, млад пролетен!}
\kray

Мене ми са босилек предади --

босилек е цвете три месеца,

аз съм зелен и зимя, и летя!

Де го чу босилек в градинка,

той на здравец тихо проговаря:

-- Здравко, цвете, билка миризлива,

ти си зилен и зимя, и летя,

аз съм цвете летя три месеца:

без меня са дете не кръщава,

без меня са булка не венчава,

без меня са мрътво не заравя!
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
