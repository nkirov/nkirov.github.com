\begin{multicols}{2}
\bbr 
Забягнала Янка робинка,

\uu{хей, Енке, Енке ле, Енке робинке,} 
\ebr

та й бигала мало и много,

мало и много -- до девят годин,
%%чува се "до девят годин"

\five тя изляла при пашовити,

при пашовити вакли йовчари. 
\kray

Поискала йовчар да стане,

те я йотзели за йодаджия.

~~ 
\end{multicols}

\ed{\el{Йодаджия} -- вер. от ,,одая`` 'стая, гостна, килер; спалня'; в контекста ,,йодаджия`` 'одалиска'.}
