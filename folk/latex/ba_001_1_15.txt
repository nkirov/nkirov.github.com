%Помами си Гани Чобан
%Коледарска (на овчар)
%Джиджев:коледарски

%%добавям "(на овчар)" - по муз. запис

\begin{multicols}{2}
Помами си Гани чобан, \uu{колади ле,}

\uu{на чардаци позлатени, колади ле,} 
%%навсякъде се чува: чардаци! НК

помами си сиво стадо, 

заведи го под чардаци,

че повика чорбаджата
%%два пъти се чува: чорбаджата НК

да си доят сиво стадо. 

Събраха са чорбаджата,
\kray

хем дояха, хем думаха:

-- Тази сутрин тези овце,

тези овце мляко нямат!

Разсърди са Гани чобан,

той нарами дрянов кривак,

че помами сиво стадо,

заведи го по герени,

по герени, по зелени,

напаса го с буйна трева,

буйна трева, непасена;

напои го в бистра вода,

бистра вода, ненапита. 

Пак го заведе под чердаци,

че повика чорбаджата

да си доят сиво стадо. 

Събраха са чорбаджата,

хем дояха, хем гълчаха:

-- Тази сутрин тези овце,

тези овце мляко нямат! 

Разсърди са Гани чобан,

че си грабна дрянов кривак, 

че потири чорбаджата --

кому кривак, кому по два,

че е станало люта кавга, 

люта кавга и крамола.
\end{multicols}

\sing{Пял: Стоян Иванов Вандов}
\village{c. Кортен}
\area{Сливенско}
\year{1908}
\rec
