%Стоян надолу тръгнува
%На хоро (по Великден)
%Джиджев:хороводни

\begin{multicols}{2}
Стоян надолу тръгнува

и си на Добра думаше:

-- Добро льо, мари хубава,

я ще надолу да ида, 

мож ли ма, любе, ти чакаш
%%в записа май не се чува "ти"
%%аз го чувам НК

цели ми до три години,
%%стихът не е изпят

докът да ида и дойда?
\kray

Добра Стояна думаше:

-- Любе Стояне, Стояне,

я та по-много чакувам!

Мойта е майка мащеха,

до ще Великден, Гергювдень,

цъфнат ще китки сякакви --

и лале, любе, и зюмбюл,

и цървен, и бел тръндафел --

мама в градинка ще влезе,

набере китки сякакви,

на мене да ги донесе,

донесе, любе, за каже:

``Оти не береш да кичиш? --

Дали Стояна жалееш,

къкто ти беше любовник,

та ти не беше главеник?''
\end{multicols}


\sing{Пяла Пенка Петрова Славчева}
\village{с. Поляново}
\area{Хаманлийско}
\year{1901}
