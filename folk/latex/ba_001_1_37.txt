%Зазори са, забяли са
%Коледарска (из пътя)
%Джиджев:коледарски

%%добавям "(из пътя)" - по муз. запис

\begin{multicols}{2}
Зазори са, забяли са, \uu{колади ле,}

\uu{по хубава Влашка зимя, колади ле,}

ни е бяло ясна зора,
%%то не било ясна зора,
%%чува се "не е било"
%%чува се: бяло НК

то е бяло добър юнак,
%%най е било добър юнак,
%%чува се "то е било"
%%чува се: бяло НК

добър юнак с добра коня.
\kray

Той си искри хранен коня 
%%искри вм. ескри - в името на смисъла

да си пали Гюргювграда,

наизлезли гюргювградци,

гюргювградци, стари хора,

молят му са гюргювградци:

-- Ой та тебе, добър юнак,

не ни пали Гюргювграда,

ша та дарим добра дара,

добра дара, кило жито!

Пак си искри хранен коня,
%%искри вм. ескри - в името на смисъла

молят му се стари хора: 

-- Не ни пали Гюргювграда,

ша та дарим добра дара,

добра дара, мяра вино! 

Пак си искри хранен коня
%%искри вм. ескри - в името на смисъла

да запали Гюргювграда,

молят му са стари хора:

-- Не ни пали Гюргювграда,

ще та дарим малка мома!

Той си възпря хранен коня,

не запали Гюргювграда.

~~ 
\end{multicols}

\sing{Пял: Стоян Иванов Вандов}
\village{c. Кортен}
\area{Сливенско}
\year{1908}
\rec
