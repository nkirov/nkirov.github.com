%Доче, Доче, кадън Доче
%Сватбарска (срещу понеделника) 
%Джиджев:сватбарски

\begin{multicols}{2}
Доче, Доче, кадън Доче,

Доче пере на реката, (2)

турчин седи под върбата

и на Доче проговаря:

-- Подигни си дюлбенчето, (2)

дюлбенчето, чумберчето!
\kray

-- Бре, турчино, бре, гидийо,

да го видиш, файда нямаш --

мойто лице продадено,

продадено, заложено

в София града за хиляда!
\end{multicols}

\year{1925}
