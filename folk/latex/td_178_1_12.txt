\begin{multicols}{2}
-- Камиларю, говедарю,

изведи ма от таз гора,

от таз гора есенлива,

заведи ма на наш баир

да погледам отвисоко,

отвисоко, нашироко --

наше село огън гори,

огън гори, метла мете!

Нека гори, да изгори --

мойта майка й моя баща,

що хранили, що гледали

девет чеда момичета:

всяко село по йдно дали,

за мен село не йостанало,

мене дали камиларя

да му паса камилите, (2)

камилите, пущините!

\rows
{Денем лежат, вечер пасат, }
{среди нощем вода пият!}
\end{multicols}

\sing{Пяла Битова група от читалище ``Развитие'', с. Генерал Тошево, Ямболско.}
\village{с. Генерал Тошево}
\area{Ямболско}
