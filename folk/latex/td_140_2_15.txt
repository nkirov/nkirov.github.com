%Жаляла Бойка буля си
%На хоро (по Великден)
%Джиджев:хороводни

\begin{multicols}{2}
Жаляла Бойка буля си,

\uu{Бойке ле, мари хубава,}

цели ми девет години --

нито си глава омила,

нито си дрипи променила.
%%по-добре "нито" вместо "нето" НК
\kray

Га стана девет години,

че си главата омила

и си дрипи променила,

че отзе темян и восък,

че си на гробe отиде
%%"гробе" вм. "гроба" - според ръкописа

и си на буля порука:

-- Бульо льо, мила и драга,

стани ми, бульо, стани ми!

-- Ни мога, лельо, да стана --

змийка ми очи изпила,

снага ми му[х]ъл фанала!

-- Лельо льо, мила и драга,
%%добавям "льо" - според ръкописа

дума ще да та попита,

правичко да ми обадиш! --

Ожени ли са батьо ти,

води ли булка хубава,

ходи ли мойто ходене,

носи ли мойто носене,

гледа ли мойте дечица,

гледа ли, хатър на мама,

на мама още на тейна?

-- Бульо льо, мила и драга,

ние си батя оженихме

и си буля доведохме:

ни ходи т'ойто ходене,

ни носи т'ойто носене,

ни гледа хатър на мама,

на мама, още на тейно, 

ни гледа т'ойте дечица!
\end{multicols}

\sing{Пяла Гина Ангелова Георгиева}
\village{ с. Лешниково}
\area{Хаманлийско}
\year{1936}
