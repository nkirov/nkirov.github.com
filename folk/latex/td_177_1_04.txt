\begin{multicols}{2}
\rows
{Съби мами си думаше: }
{-- Карат ма, мамо, карат ма, }

\rows
{карат ма, мамо, карат ма }
{нашите селски чорбаджа,}

\rows
{нашите селски чорбаджа, }
{чорбаджа, селски думбаджа}

\rows
{във Цариграда да ида, }
{във Цариграда при царя.}

\rows
{Зарана, мамо, зарана, }
{зарана рано да раниш,}

\rows
{зарана рано да раниш, }
{да ми кончето назобиш, }

\rows
{да ми кончето назобиш, }
{мене от сън да събудиш!}

\rows
{Майка му Съби думаше: }
{-- Събе ле, синко Събе ле,}

\rows
{бе, тук стой, синко, не ходи }
{те ша та, синко, убият!}
\kray

\rows
{Съби майка си думаше:}
{-- Ша ида, мамо, ша ида,}

\rows
{ша ида да се оплача}
{от тия турци проклети,}

\rows
{дето ми всичко отнимат!}
{Съби майка си не слуша,}

\rows
{че си кончето възседна,}
{във Цариграда отишъл, }

\rows
{във Цариграда, при царя,}
{отишъл и са оплакал.}

\rows
{В селото Съби чакали}
{тези ми турски заптии --}

\rows
{и те са Съби хванали,}
{хванали й са го убили.}
\end{multicols}

\ed{Думбаджа -- диалектна форма от ``думбазин'' тур. 'надут, горделив, ботаташ'.}
