\begin{multicols}{2}
Чули сме ва, чули сме ва,

чули, та сме дошли,

\rows
{стани, господине, }
{стани, нине, йотори ни!}

\five 'Ку си сънян, будян бъди,

стани, господине,

стани, йотори ни!

'Ку си пиян, трезвян бъди,

стани, нине, стани,

\ten стани, господине!

Овце ти са изягниха,

стани, господине,

стани, йотори ни!

Кози ти са изяриха, 

\fifteen стани, нине, стани,

стани, господине!

\end{multicols}

\begin{multicols}{2}
\kray

Котки ти са изкотиха,

стани, господине,

стани, йотори ни!
\end{multicols}
