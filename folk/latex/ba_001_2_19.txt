\begin{multicols}{2}
\rows
{Тръгнали ми са, тръгнали }
{две Ради, дор две итърви, }

\rows
{ходели сено да косят. }
{Косили накосили го, }
\kray

\rows
{тепили, натепили го,}
{тръгнали да си отиват.}

\rows
{Тихън е вятър повяло,}
{силен е дъжд заваляло,}

\rows
{реката мътна затече,}
{дърви й камъни повлече.}

\rows
{Те са седели, чакали}
{да се реката оттече,}

\rows
{не се реката оттича,}
{по-кална, мътна притича.}

\rows
{По-малка Рада нагази,}
{нагази и я премина,}

\rows
{голяма Рада нагази,}
{не можа река да мине,}

\rows
{реката Рада повлече,}
{от бряг я във бряг блъскаше,}

\rows
{косата \'{и} се закачи}
{за ракидово коренче.}

\rows
{Рада се ясно провикна:}
{-- Бре Боже, бре мили Боже,}

\rows
{няма ли нейде някого}
{да дойде да ме извади?}
\end{multicols}
