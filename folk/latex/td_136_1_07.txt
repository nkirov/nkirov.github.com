%Коледарска (на овчар)

%%в базата отсъства нотописа!!

\begin{multicols}{2}
Помамял си,  \uu{коладе ле}, дили Нано, \uu{коладе ле}, 

помамял си сиво стадо,

помамял ги въз планина,
%%стихът не е изпят

въз планина, въз тривнина. 

Отговаря вайводата:

-- Ой вай вази, трима юнака,

я идети при дили Нано
%%чува се: я идете при дил Нано НК

да ви даде три овена,
\kray

три овена за юнаци

и шаро ягне за вайвода!

Отишле са трима юнака:

-- Ой та тебе, дили Нано,

нас на прати вайводата

да ни дадеш три овена,

три овена за юнаци

и шаро ягне за вайвода!

Отговаря дили Нано:

-- 'Ку би давал дили Нано,

три овена на юнаци

и шаро ягне за вайвода,

не би мамил сиво стадо!

Върнали са трима юнака:

-- Ой та тебе, вайводо льо,

не ни даде дили Нано

три овена за юнаци

и шаро ягне за вайвода!

Догневя го вайводата,

-- Ой въй вази, девет юнака,

хай да идем при дили Нано

да си вземем три овена,

три овена за юнаци

и шаро ягне за вайвода! 

Отишле са при дили Нано,

отговаря вайводата:

-- Ой тай тебе, дили Нано,

що не даде три овена,

три овена за юнаци

и шаро ягне за вайвода?

Догневя го дили Нано,

издигна си дрянов кривак,

че замахна той надясно,

дор обърне той наляво,

останал е сам вайводата...
\end{multicols}
