%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Мари свате мокрополо,

\uu{нидей до, мари, нидей до,}
 
\uu{ни бой са, мари, ни ща ти до,}

каква си ми снаха дала,
\kray

тя не може в стан да влезе!

Тя в стан влазя и ма пита:

-- Каква й тази меча дупка,

едно скача, друго сяда,

сичко вика ``скръцта-мръцта''?
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
