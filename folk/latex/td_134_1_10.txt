%Мама слънце дума
%Жътварска ("га напиче слънцето")
%Джиджев:жътварски

\begin{multicols}{2}
Мама слънце дума:

-- Слънце, ясно слънце,

кому мама ръча,

ръча и поръча

ясно да ни греиш,

чи изгори, слънце,

жътвари по ниви,
%%този и долния стих - сричка повече

овчари по чардаци,

косачи по ливади?! 
%%косачи са!   НК
\kray

Слънце мале дума:

-- Мале, стара мале,

кому да угодя? --

Малки моми, мамо, 

бяли платна бялят,

на мене са молят:

``Греи, ясно слънце,

греи да избялим

наште тънки дари --

да си дарим, слънце,

нашта млого рода!''

~~
\end{multicols}
