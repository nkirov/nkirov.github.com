\begin{multicols}{2}
О[в]довяла й лисицата,
%%чува се "лисицата" НК 

\uu{дирлядо-дирли-лилядо-дирлядо,}

със дванайсет лисиченца: 
%%чува се "лисиченца" НК

-- Де щи мама да ва види?
%%в машинописа стихът е тук в записа с два стиха надолу, оставям го тук по смисъл

Най-малкото, най-хитрото,

то на мама си думаше: 
\kray

-- Ще ни видиш, мила мамо,

на боляра на шията,

на чорбаджията в кисията!
\end{multicols}
