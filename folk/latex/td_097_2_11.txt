\begin{multicols}{2}
-- Глава ли та боли, Донке мама,
%%колебание "боли" - "були", избирам "боли"

глава [и] половина, Донке мама, ма, 

глава [и] половина? 

-- Ни ма глава боли, мамо ма,

\five сърцето ма боли, мила мамо ма,

за първото люби,
\kray

за първото люби, 

люби нивинчано,

с мени нисъбрано. 

\ten 'Ко бях го видяла,

щях да му поръчам: 
%%щях вм. щах - за яснота

кога Тунджа мине,

вода да ни пии --

Тунджа носи, мамо,

\fifteen билки ми омразни,

билки ми раздялни. 
%%коментар от набора: 
%%(Г.Г. - тая песен трябва да я чуя; стиховата организация е невероятно объркана????)

~~
\end{multicols}
