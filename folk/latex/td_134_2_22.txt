\begin{multicols}{2}
-- Роданчице, малка моме, 

ти като си най-манинка,
%%чува се "манинка" (с дълго "а"!) НК

най-манинка, най-глупава, 

колко либовника имаш? 

-- Имам, имам, старо мале, (2)

сякой село по двамина, 
%%стихът не е изпят

в Дряном село -- деветима.

~~ 
\end{multicols}
