%Станинине, господине
%Коледарска (в двора на къщата)
%Джиджев:коледарски

\begin{multicols}{2}
Станинине, господине,

\uu{иди му ляпам чела,}

\uu{мушни му ляпом пяим,}

'ку си заспал, буден бъди,
%%вм. "'ку си заспал, трезвен бъди," - по муз. запис
%%размествам местата на този и следващия стих - по муз. запис

'ку си пиян, трезвен бъди,

че ти идат добри гости,

добри гости коледари,
%%стихът не е изпят
%%пее се !!! НК

по-добър ти хабер носят:
\kray

кравите са изтелиха --

ваклушати, каврушати;

пчелите са изровиха --

с\'{е} мушички златни крилца;

конете са изждребиха --

с\'{е} илченца, чекалченца!

~~ 
\end{multicols}

\bel{От дълга употреба на този призив припевните думи са се 
изменили дотолкова, че са загубили своя смисъл и се повтарят механично.}

\area{Харманлийско}
\bornYear{1911}
