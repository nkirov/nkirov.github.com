%Рада на пъкя, кажи-речи, седеше
%Хороводна
%Джиджев:хороводни

\begin{multicols}{2}
Рада на пъкя, кажи-речи, седеше,

на пъкя, на хоратата, иха-ха,

на пъкя, на хоратата,
%%"на хоратата" - какво ли е това? "Хората"?

отдолу, кажи-речи, иде Андрея, 

Андрея Минин Стамовчин, иха-ха,

Андрея Минин Стамовчин.

Андрея, кажи-речи, Радки думаше:

-- Помози ти бог, Радки ле, иха-ха,

помози ти бог, Радки ле!

Рада Андрея, кажи-речи, думаше:

-- Я си във пъкя минавай, иха-ха,

и са със мен ни закачай,
\kray

аз съм болярска, кажи-речи, дъщеря

и съм хаджийска унука, иха-ха,

и съм хаджийска унука,

пък ти си, кажи-речи, Минин Стамовчин,

майка ти ходи за дърва, иха-ха,

ти ходиш да \'{и} поемаш!

Андрея, кажи-речи, Радки думаше:

-- То ни е, Радки, срамота, иха-ха,

чи съм син Минин Стамовчин,

чи мама, кажи-речи, ходи за дърва,

ами е, Радке, срамота, иха-ха,

ами е, Радке, срамота:

помниш ли, кажи-речи, Радке, знаеш ли,

когито сливи крадохме, иха-ха,
%%Слава вм. сливи - по ръкописа

със Дона Попадиина?!
\end{multicols}

\sing{Пял свещ. Цоню Цонев Антонов}
\year{1917}
\bornVillage{гр. Котел}
