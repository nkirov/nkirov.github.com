%Йостана Ванкя сираче
%На трапеза (също: на попрелки -- на разнисане)
%Джиджев:трапезни

%%също от тефтера (на попрелки, на разнисане)
%%последователно встъпване чрез начално "и"

\begin{multicols}{2}
Йостана Ванкя сираче,
%%как да пишем заглавието в този случай: "Йостана" или "И, остана" или "Остана"???

без майка, Ванкя, без тейно,

чи са майка хи ожени (2)
%%стихът отсъства в ръкописа и машинописа, тук - от нотите
%%съкращавам стиха "че за майка хи дойдаха" - няма го в записа

във Светлен село голямо. 
%%"във Светлен" вм. "от Виден" - така чувам аз музиката

Ванкя си плаче, не рачи.
%%тук нещо по смисъл е изпуснато

Станали и са тръгнали,
%%вм. "че стана Ванкя, че тръгна" - по музиката

вървели, що са вървели,

га до гробища отишли,
\kray

Ванкя са на гроб отбила

и на татка си нарежда:

-- Я стани, тейне, погледни

бащови двори големи --

кък на е майка пръснала

по тея ми чузди къщи: 

дворето са тревясали,

портите са гъбясали!

~~
\end{multicols}

\ed{Светлен -- може би с.\,Светлен, общ.\,Кирково, обл.\,Кърджали.}
