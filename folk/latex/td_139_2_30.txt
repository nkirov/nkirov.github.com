%Стоян са юнак зажени
%Сватбарска (``като тръгнат за булката'')
%Джиджев:сватбарски

%%добавям "като тръгнат за булката" - от ръкописа

\begin{multicols}{2}
Стоян са юнак зажени,

той тежка сватба подигна,

много сватове калеса:

педесе коля сватове, (2)

шейсета -- лични кумове,

далек за булка да идат,

през море булка да вземят.
\kray

Вървели, що са вървели,

до сухо дърво стигнали,

сухо им дърво продума:

-- Кумове, още сватове,

вие се назад върнете,
%%добавям "се" - заради смисъла; нагласят се и сричките

че ви булчето умряло!
\end{multicols}

\sing{Пяла Доба Петкова Гегова}
\village{с. Брягово}
\area{Хасковско}
\year{1925}
\re
