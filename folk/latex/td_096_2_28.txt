\begin{multicols}{2}
Пиле ми пеи, \uu{мамо,}

пиле ми пеи,

гора люлеи, \uu{мамо,}

гора люлеи -- 

гора и вода. (2)
%%пропуснат стих в ръкописа - тук по записа

Не било пиле, (2)

най било Нанка, (2)

Нанка в градинки, (2)
\kray

Нанка в башчийки, (2)
%%"Нанка" вм. "Янка" - предполагам, че е грешка (така е и в ръкописа); има само един персонаж в песента - тук второ лично име

чи ръкав шие, (2)

ръкав на гергьов, (2)

шии с коприна, (2)

шии и кълне: (2)

-- Да са проваляш, (2)

дето ма дади (2)

на чер циганин, (2)

на гол дервишин: (2)

пет\'{а}ли сич\'{е}, (2)

волови кове, (2)

искри изскачат, (2)

та Нанка парят (2)

по бяло лице! (2) 
\end{multicols}

\bel{Отначало ла бемол е неустановен, но в края се затвърдява.}

\ed{Башчийки -- вер. от ``бахча'' тур. 'зеленчукова градина'. }
