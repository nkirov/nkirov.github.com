%Скрийте ма, брайне, скрийте ма
%Сватбарска (когато влязат в момините двори) 
%Джиджев:сватбарски

\begin{multicols}{2}
-- Скрийте ма, брайне, скрийте ма,

във вашта равна градинка, (2)
%%"вашта" вм. "ваща" - същото, но по-четивно

под бял, под червен трендафил!

-- Не бива, сестро, не бива, 

сватове на д'ор дойдаха, (2)

дойдаха и наредиха,
%%стихът не е изпят

кумове на стол седнаха!

~~
\end{multicols}

\sing{Пяла Доба Петкова Гегова}
\village{с. Брягово}
\area{Хасковско}
\year{1925}
\re
