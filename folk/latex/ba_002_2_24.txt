%Дядо Пано, дермянджийо
%На сватба (хоро на сладката ракия)
%Джиджев:хороводни

%%(пее се на сладката ракия)
%%За мен и цикълът „на сладката ракия” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
-- Дядо Пано, дермянджийо, (2)

въри ли ти водиничката? (2)

-- Въри, въри, търкаля са, (2)

мели кадар за децата! (2)

\five -- Дядо Пано, дермянджийо, (2)

колко ти са дичицата? (2)

-- Пет на оран, пет на копан (2)

и петнайси сусам берат, (2)
\kray

двечки скоро проходяха (2)

\ten и две сега пропълзяха,

и две сега на път имам!

-- Дядо Пано, дермянджийо, (2)

те ти стигат, да са живи! (2)
\end{multicols}

\ed{\el{Кадар} {\em тур.} 'колкото, тъкмо'; в контекста 'колкото е нужно за децата'.}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}



\placedate{с.\,Кортен, общ.\,Нова Загора, 17 февруари 1966\,г.}
