%Величка на балкон седи
%На трапеза (на гости) 
%Джиджев:трапезни

\begin{multicols}{2}
Величка на балкон сиде, (2)

бяла тантела плетеше, (3)

любовна песен пейеше, (3)

ем си нагоре гледаше.

Отгоре идат войници, (2)
%%чува се: отгоре идат войнице НК

тия Виличка думаха: 

-- Не пей, Велико, таз песен, (2)
%%чува се: Не пей Велико таз песен НК

таз песен, любовна песен,
\kray

чи ти я либе юмряло -- (2)

сръбски го куршум прониза

от лява стърна в сърцето, (2)

от дясна стърна в ногите, (3)

та си на място останя!

Ка чу Величка тез думи, (2)

на[х]лупи кърпа до хочи,

порони сълзи до зимя, (2)

та па ми викна, заплака: (3)

-- Божни ле, милничък боже, (3)

не стигна ли ми, божни ле, (3)

девет године сирашки, (3)
%%сирашки вм. сирашко - за смисъл

на девет майки да ода, (2)

дисетя майка циганка, (3)

ми сига, боже, без любе?! 
\end{multicols}

\editor{Колебание Величка -- Виличка.}
