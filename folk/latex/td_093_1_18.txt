\begin{multicols}{2}
-- Рада, бяла Радо, (2)

що по поле йоди, (2)

глава ли та боле, (2)

глава половина (2)
\kray

или клето сърце? (2)

Я Рада им каже: (2)

-- Ни ма глава боле, (2)

на[й] съм чула, байне, (2)

мама ма годила (2)

приз Дунав за влахът, (2) 
%%"влахът" вм. "вланът" - в името на смисъла, вер. правописна грешка

за Банкова сина! (2)

Аз Банко не зная, (2)

току съм го чула, (2)
%%"току" вм. "токо(у)" - изяснявам колебанието на Джиджев в посока на яснотата

чула и разбрала -- (2)

родовит е Банко, (2)
%%вм. "родовите" - възстановявам смисъла

млого рода има, (2)

даре не ме стигат: (2)

двеста шите кърпе (2)

и двеста нешите, (2)

петстотин ръкава, (2)

шестстотин месаля! (2)
\end{multicols}
