\begin{multicols}{2}
\uu{Йо, Яно йегне}, 

пладнина прейожда,

мома не дойожда

обяд да донисе,

Ей че брайно иде,
%%добавям "ей" - по ръкописа и записа

отдалеч извика:
%%чувам по-скоро "отдалече 'звика"

-- Жъни, жъни, Яно,

мама та й сгодила
%%предпочитам "сгодила"

за сина Костадина!
%%гласна в повече, но всички се чуват - изпяването е сгъстено

~~
\end{multicols}

\ed{Прейожда -- диалектна форма от 'прехождам', 'преминавам'.}
