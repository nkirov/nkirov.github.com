%Рано е Радка ранила
%На попрелки
%Джиджев:седенкарски

%%в базата няма ноти

\begin{multicols}{2}
Рано е Радка ранила,

ранинко, Радка, посред нощ,

петли не бяха попели, (2)
%%петли вм. петлите - по муз. запис и ръкописа, за сричките

попели и повторили.

Рано е Радка ранила

със бели менци за вода

от тихи, от бял Дунава --

слънцето да я не види,

слънцето, още месеца

и на небото звездите. 
%%на вм. от - така е по записа и в ръкописа
\kray

На пътя Радка посрещна

два змея, Радка, два брата

като два черни облака,

Змейко на Радка думаше:

-- Радке ле, моме хубава,

много си, Радке, хубава, (2)

майка ти е магьосница: 

снощи край вази минахме,

край ваште двори широки,
%%ваште вм. ваще - същото, по-четивно

край ваште порти високи, 
%%ваште вм. ваще - същото, по-четивно

майка ти по двор ходеше

и в ръки змийче държеше,

с остра го игла бодеше

и си на змийче думаше:

``Както се вие туй змийче,

тъй да се вият момците

край наша Радка хубава,

хубава Радка, гиздава!''
\end{multicols}

\rec
