%Куме ле, кум Костадине
%Сватбарска (``когато отиват за кумовете, из пътя'') 
%Джиджев:сватбарски

\begin{multicols}{2}
Куме ле, кум Костадине,

еж грозде, не наявай се, (2)

пий вино, не напивай се, 

че щим далеко да идем,

далеко, булка да земем,
%%стихът не е изпят

през девет села в десето! 
\end{multicols}

\village{с. Брягово}
\area{Хасковско}
\year{1925}
\re
