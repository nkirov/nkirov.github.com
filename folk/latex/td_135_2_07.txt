%Запил ми са й храбром юнак
%Коледарска (на стопанин, който много пие)
%Джиджев:коледарски

\begin{multicols}{2}
Запил ми са й храбром юнак

\uu{среди ми село, на мегданя,}

пил ми два дня, пил ми три дня --

слугаря му млада буля. 
%%следващите два стиха не са изпяти

Него вино не йопива,

не ракия прекалява,

йотговаря млада буля:
%%предпочитам "йотговаря"

-- Ой та тебе, храбром юнак,

сега става два дня й три дня,

не ти вино не йопива,
\kray

не ракия прекалява?!

Йотговаря храбром юнак:
%%предпочитам "йотговаря"

-- Мари бульо, млада бульо,

като гледам тебе в очи,

мене вино не йопива,

не ракия прекалява!

Наздраве ти храбром юнак,

от бога ти тва наздраве,

догодина пак със здраве!
\end{multicols}
