%Войници, Стойне, избират
%Хороводна
%Джиджев:хороводни
%%този и следващите записи са от окръжен събор - надпяване в гр. Елхово. 2 естрада: певчески групи и танцови състави!!! Тук има и думи на водещите.

\begin{multicols}{2}
-- Войници, Стойне, избират,

\uu{Стойне ле, юбавице ле,}

кой за две годин, кой за три,

и мене, Стойне, избрая

за три години млад войник,

млад войник, млад кавалерист,

жъ ма чакаш ли, любе ле?

-- Чакам те, любе Никола,
%%"любе" вм. "чакам те" - според записа

\uu{Никола, младо войниче,}

дорде от село излянеш,

жъ вляна в горна горничка,

жъ се премена нареда,

жъ вляна в долна градинка,

жъ скъсам цвете всякакво,

жъ взема бяли бакъри,

та че за вода жъ ида,

друго любе жъ залюбя,

друго любе, по-юбаво!
\end{multicols}
