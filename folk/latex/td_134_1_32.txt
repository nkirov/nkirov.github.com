%Джанъм, Тонки ле, Тодорке

\begin{multicols}{2}
-- Джанъм, Тонки ле, Тодорке,
%%чува се "Тодорке" НК

мар, милната батива, (2)

та къде йе кака ти, (2)

та са повън не види, (2)

\five повън, Дойне ле, по двори: (2)

не на дръвник за дърва, (2)
\kray

не на сокак за телци? (2)

-- Байни ле, бате Иване, (2)

ний си кака сгодихме (2)
%%предпочитам "сгодихме"

\ten за ино долно долненче, (2)

за ино бяло бакалче. (2)

-- Джанъм, Тонки ле, Тодорке,

га си кака сгодихте (2)
%%предпочитам "сгодихме"

ти си на какино място, (2)

\fifteen хем по-бяла й чървена, (2)

хем по-тънка-висока!

-- Байне ле, бате Иване,

нали сме пуста роднина, (2)

роднина по сватовщина,

\twenty мама й стрина -- две етърви,

чичо й тате -- два брата?

-- Джанъм, Тонке ле, Тодорке,

дълбока вода бой няма, (2)

честа чилия сяк няма, (2)

\twentyfive дребни камъни брой нямат, (2)

хубава мома род няма!
\end{multicols}
