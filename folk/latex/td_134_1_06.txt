%Иван стреля стрелите
%Сватбарска (``на пряпола'')
%Джиджев:сватбарски

\begin{multicols}{2}
Иван стреля стрелите,

та стрелите му падаха

на Донкините дворови.

Донка си стрели сбираше, 
%%предпочитам "сбираше"

сбираше, в къщи внисаше,
%%предпочитам "сбираше"

на майка си ги даваше:
\kray

-- Мар, н\'{а} ти, мамо, тез стрели,
%%така ли се слага ударение?

сватови, мамо, издойдоха,

дворови ти изпълниха:

коне ти извързаха

под червения трендафил,

йозди ти намятаха

по желтите дюни! 
\end{multicols}

\ed{Дюна нар. 'дюля'.}

\sing{Пяла Жела Господинова Филипова}
\village{с. Навъсен}
\area{Харманлийско}
\year{1907}
\re
