%Брайне, руду славей 
%Жътварска (по пладне)
%Джиджев:жътварски

\begin{multicols}{2}
-- Брайне, рудо славей, 

всичкити жътвари

пладня пладнувая,

а ний с теби, братко, (2)

обяд ни сме яли,
\kray

камо ли пладнина!

Брайно Янка дума:

-- Карай, Янке, карай,

накрай да излезем,

ето мама иди

и обяд ни носи --

отдалеч са смеи,

отблизо казува:

``Карай, Янке, карай,

мама та й сгодила (2)

през Дунав за вла[х]я --

той си влашки гълчи:

ти му ни разбираш,

той си вода иска,

ти му ляба даваш!'' 
\end{multicols}

\sing{Пяла: Куна Вълева Радкова}
\village{с. Бабек}
\area{Пловдивско}
\year{1904}
\rec
