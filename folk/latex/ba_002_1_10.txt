%Вълко от горя слазя
%Хороводна (на засевки, вкъщи)
%Джиджев:хороводни 

%(пее се на засевки вкъщи)

%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
\rows
{Вълко отгоря слазя,}
{горя, от гората,}

\rows
{срешнала го Ганка,}
{Ганка Вълко дума:}

\rows
{-- Вълко, любе Вълко,}
{защо не ми набра}
\kray

\rows
{от гора горов цвят,}
{синя метличена,}

\rows
{синя метличена,}
{бяла кокичена,}

\rows
{от наром нарчиче,}
{от каром картопу?}

\rows
{Вълко Ганки дума:}
{-- Гане, любе Гане,}

\rows
{набрал бях ти, Гане,}
{набрал бях ти, джанъм,}

\rows
{срещнаха ма, Ганке,}
{твойте трите хайки,}

\rows
{твойте трите хайки,}
{трите ти душманки.}

\rows
{Първата ми рече:}
{-- Дай ги мене, Вълко,}

\rows
{дай ги мене, Вълко,}
{Ганка болна лежи!}

\rows
{Втората ми рече:}
{-- Дай ги мене, Вълко,}

\rows
{дай ги мене, Вълко,}
{Ганка душа бере!}

\rows
{Третата ми рече:}
{-- Дай ги мене, Вълко,}

\rows
{дай ги мене, Вълко,}
{Ганка вчера умря,}

\rows
{Ганка вчера умря,}
{днес я заровихме!}

\rows
{И аз им повярвах,}
{че им го отдадах.}
\end{multicols}

\ed{Каром -- вер. диалектна форма от ``кара'' 'черен, тъмен на цвят'.\\
Картоп 'декоративен градински храст с бели топчести съцветия, калина, 
Viburnum opulus sterile', 'калина, цвете хортензия, Hydrangea hortensis', 'цвете снежна китка'.\\
Нарчиче -- диалектна форма от ``нарцис''.
}
