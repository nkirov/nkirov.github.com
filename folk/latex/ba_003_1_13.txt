\begin{multicols}{2}
\rows
{Тодоро, любе Тодоро,}
{що ти пирото трипери:}

\rows
{дали го вятър люлеи}
{или самичко трипери?}
\kray

\rows
{Тодора дума Ивану:}
{-- Иване, любе Иване,}

\rows
{нито го вятър люлее,}
{нито самичко трипери,}

\rows
{трепери, любе, трипери,}
{трепери да ти пристани,}

\rows
{трепери да ти пристани,}
{на тебе булка да стана!}
