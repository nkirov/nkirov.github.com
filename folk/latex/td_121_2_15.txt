\begin{multicols}{2}
Залибих си ново люби, \uu{мамо},

залибих си ново люби,

ново люби, дърводелчи, \uu{мамо},

ново люби, дърводелчи,

дето прави нови хурки, (2)

нови хурки и кавали. (2)

Направил ми нова хурка, (2)

сам на нея са изписал -- (2)

като прида, да го гледам, (2)

да го гледам, да се радвам; (2)

направил си меден кавал, (2)

на кавала ме изписал -- (2)

като свири, да ме гледа, (2)

да ме гледа, да ми смига! (2)
\end{multicols}

\sing{Пяла Бонка Христова Бърдарова}
\year{1903}
\bornVillage{гр. Котел}
