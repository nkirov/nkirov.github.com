%Ой, Гьорге, Гьорге джанъм, куюмджи Гьорге
%Хороводна (на Гергьовден)
%Джиджев:хороводни

%%добавям (на Гергьовден) - по записа, ръкописа и машинописа

\begin{multicols}{2}
-- Ой, Гьорге, Гьорге джанъм, куюмджи Гьорге,

ти си се прочул, \uu{Гьорге ле}, и си са разчул,

че ляпо леиш, ляпо и спорно,

ще ти донеса две стари пари

да ми излеиш две вити гримни,

две вити гримни, пръстен нишански!
%%две вити гримни вм. "и да излееш" - по записа; "нишански" вм. "мишански" - 
%%по записа, "пръстен нишанлия" е формула; кърпата долу е мишанска
\kray

-- Ой, Маро, Маро джанъм, хубава Маро,

ти с' са прочула, \uu{Маро льо}, и с' са разчула,

че тънко предеш, тънко и спорно,

ще ти донеса еллия памук

да ми изпредеш, да ми изтъчеш

две мъжки ризи, кърпа мишанска!

~~
\end{multicols}

%%от тефтера: 
\dic{Куюмджи -- леяр.\\
Eллия -- стара турска мярка; малка мярка за тегло -- ок. 1/4 кг.} 

\ed{Еллия -- диалектна форма на ``иллия'' 'стара малка мярка за тегло'.\\
Мишански -- вер. производно от ``нишан, мишан'' 'знак, белег'; ``мишанска'' 'белязана, изрисувана'.}

\sing{Пяла Здравка Димитрова Делчева (Попконстантинова)}
\year{1901}
\village{Любеново}
\area{Хасковско}
