%Засвиряла й тънка свирка
%Хороводна (на засевки)
%Джиджев:хороводни

%%(пее се на засевки на двора)

%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Засвиряла й, \uu{мари}, тънка свирка,

тънка свирка, \uu{мари}, кокаляна,

\uu{ха, хо, хо, хо}, тънка свирка, \uu{мари,}

\uu{бре, бре, бре, бре}, кокаляна.

Свирка свири, та говори:

-- Коя мома, люби няма,
\kray

нея царя, ша дадеми

царя да \`{и} й, любовника!

Обади са, Станената майка:

-- Наша Стана, люби няма,

нея царя да дадеми,

царя да \`{и} й, любовника!

Обади са бяла Стана:

-- Мълчи, мамо, онемяла,

й с двете очи, ослепяла,

ваша Стана, люби има -- 

в всяко село, двама-трима,

в нашто село, девемина!

~~
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
