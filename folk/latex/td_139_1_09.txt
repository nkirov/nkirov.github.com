%Излези, излези
%Сватбарска (``когато влизат (въвеждат) булката в момковите двори'')
%Джиджев:сватбарски

\begin{multicols}{2}
-- Излези, излези, 

ти, момкова майко,

излези, излези 

вън на равни двори,

та да ми погледнеш 

какво булче водим --

тънко, та високо, 

бяло чернооко!
\end{multicols}
