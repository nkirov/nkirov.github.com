%Лельо льо, мила и драга
%На хоро (по Великден)
%Джиджев:хороводни

\begin{multicols}{2}
-- Лельо льо, мила и драга, (2)

га беше, лельо, лефтера, (2)

бяла бя като кадъна, (2)

цървена като ябълка, (2)
\kray

яга съ, леле, ожени, (2)
%%"леле" вм. "леля" - разчитането е спорно, така е по-логично; според мен най-добре е "лельо"

поцърня, лельо, погрозня -- (2)

църна ми като циганка,

жельта ми като миранза! (2)
%%миранза вм. меранзя - според ръкописа

-- Бульо льо, мила и драга, (2)

кък ни ща, бульо, поцърня, (2)

поцърня, бульо, погрозня -- (2)

църна ми като циганка, (2)

жълта ми като миранза! (2)

Ние сме девет етърви, (2)

кога от баир дойдеме, (2)
%%предпочитам "от"

сяка си земи дятенце, (2)

сяка си люшне люлчица, (2)

я зема, бульо, ведрото, (2)

до девет крави издоя; (2)

доде телците излъча, (2)

тея са, бульо наспават, (2)

че па ще рано да стана, (2)

до девет хляба изпеча, (2)

девет бурала избърквам; (2)

кога на нива тръгниме, (2)

сяка си земе дятенце, (2)

сяка си земе люлчица, (2)

я зема, бульо, торбата, (2)

я зема девет кавръми; (2)

кога на нива идеме, (2)

тия си деца прихраньват, (2)

я зема, бульо, кавръма, (2)

я си чакъма зафатам! (2)
\end{multicols}
