%Засвивал са й сив-бял сокол
%Коледарска (на мома)
%Джиджев:коледарски

%%добавям "(на мома)" - по муз. запис

\begin{multicols}{2}
Засвивал са й, \uu{Дойни ле, колади ле,} 

сив-бял сокол, \uu{момни ле, колади ле,}

над момено равно двори

да си гони пъстра птица. 

Де го видя малка мома:
\kray

-- Хъшом-къшом, сив-бял сокол,

не ми гони пъстра птица! --

Тя ще дойде на къщата,

на къщата, под стряхата,

ще си свие вито гнездо,

ще измъти пъстри пилци,

ще ми пеят сутрин рано,

сутрин рано, на ставани,

вечер късно на лягани;

кога дойдат добри гости,

добри гости, сватовници,

че ще рекът добри гости:

``Барим мома, барим чудо,

как научи този дивеч 

да си пее тъй хубаво?!''

~~
\end{multicols}

\sing{Пял: Стоян Иванов Вандов}
\village{c. Кортен}
\area{Сливенско}
\year{1908}
\rec
