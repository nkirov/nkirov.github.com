%Райна в лозята отива
%Хороводна (пее се на засевки)

%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Райна в лозята отива, \uu{мари,}

Райна в лозята отива,

в лозята за бяло грозде, \uu{мари,}

в лозята за бяло грозде.

Иван подиря й вървеше, (2)

Иван на Райна думаше: (2)

-- Чакай ма, Райне, чакай ма, (2)

двама в лозята да идем, (2)
\kray

да вървим, да хортуваме, (2)

верничко да са питаме, (2)

ше ли ма земеш, Райни мо! (2)
\end{multicols}

\bel{Тактове, отбелязани с 2 и 4 са с тенденция към съкращаване.}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
