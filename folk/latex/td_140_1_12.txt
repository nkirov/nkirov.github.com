%%добавям (на седянка, на жътва) - от ръкописа

\begin{multicols}{2}
Славчо на Яна думаше:

-- Яно льо, бяла кадъно,

ти чула ли си, Яно льо,

чи ще развълно да стане,

развълно, разбег година?
%%чува се: развълно, Яно, правдено НК 

Кой има стара майчица,

да гледа, да се нагледа;

кой има мъжко дитенце,

да радва, да се нарадва;
\kray

кой има любе севгалъ,

да хортува, нахортува,

да гледа, да се нагледа,

чи ще развъл да стане,

развълно и разбегно!

Яна на Славчо думаше:

-- 'Ку стане, Славчо, да стане,

много са моми за тебе,

по-много момци за мене!

Славчо на Яна думаше:

-- Яно ле, бяла кадъно,

многу са, ам' за какво са:

момците мене ни мясат,

ти мясаш, бела кадъно!
\end{multicols}


\ed{Разбегно -- нефиксирана диалектна дума, вер. от ``разбягвам се''; в случая 'размирна'.\\
Развъл -- диалектна форма от ``развала''.\\
Севгалъ -- диалектна форма от 'севда'; в контекста със значение 'любима'. 
}
