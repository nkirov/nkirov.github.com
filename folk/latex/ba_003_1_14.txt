\begin{multicols}{2}
\rows
{Марийке Куцаричена, \uu{мари,}}
{Куцаричено момиче,}

\rows
{изтъка ли си платното, \uu{мари,}}
{дето го нави, основа}
%%"основа" вместо "нареди" - според записа; "основа" - вер. от "наснова"???
\kray

\rows
{на първи велик четвъртък,}
{срещу разпетия петък?}

\rows
{Марийка дума Ивано:}
{-- Иване, любе Иване,}

\rows
{че кога да го изтъка,}
{като съм болна лежала,}

\rows
{като съм болна лежала,}
{болна, Иване, неволна.}

\rows
{-- Марийке Куцаричена,}
{Куцаричено момиче,}

\rows
{лъжи, Марийке, кого щеш,}
{меня не можеш излъга --}

\rows
{нова пътека направи}
{от къщи дор до вратника,}

\rows
{от къщи дор до вратника,}
{за мене, любе, да гледаш!}
\end{multicols}
