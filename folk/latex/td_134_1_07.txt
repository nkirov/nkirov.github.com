%Заспала бяла Тодорка
%Жътварска (по обед)
%Джиджев:жътварски

\begin{multicols}{2}
\uu{И}, Заспала бяла Тодорка

накрай морето, крайчето: 
%%идеи какво значи "крайчето"

-- Таман бях сладко заспала,

ду'нали буйни вятрове, (2)
%%ду'нали вм. дуунали - по муз. запис, в името на сричките

литнали силни дъждове,
\kray

та разбудили Тодорка! 

Тодорка ветрови клела:
%%сричка повече

-- Вятре ле, ненадуйнико, 

дъждо льо, неналятнико,

тамам бях сладко заспала! 

Я имам братче умряло, 

с\`{я} става девят години,

насън го ни бях сънила, 

сега го насън сънувах, 

чи дойди, како, та речи:

``Како ле, милинка како,

ти речи, како, на мама, 

кърпата да си опере --

ни мога вода да пия:

водата ми е, како льо,

се кална, како, се мътна!''
\end{multicols}
