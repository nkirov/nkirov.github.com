%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Снощи минах покрай сел'то,

\uu{Тодоро льо, Дюлгерюва,} (2)

покрай сел'то Тодорен'то,
%% на музикалния запис е друго! НК

хорото им играяше.
%% на музикалния запис е друго! НК

Повъзпрях се да погледам,
%%стихът се чува на записа - добавям го
\kray

те са бяха наредили --

до сяка мома и момък,

до Тодора Дюлгерюва

сал при нея ерген няма.

Сидях, сидях, та ги гледах,

побих кола, вързах коня

и аз се хванах на хорото.

Улових се до Тодора,

до Тодора Дюлгерюва,

чи на лудо аз затропах,

пък Тодора ми продума:

-- Ой та тебе, лудо-младо,

аз съм, лудо, годеница!

~~
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
