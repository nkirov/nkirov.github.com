%Мама Марийки думаши
%Хороводна (пее се на мегдана)
%Джиджев:хороводни

%%(пее се на мегдана)

\begin{multicols}{2}
Мама Марийки думаши:

-- Марийке, казъм Марийке,

защо, Марийке, закъсни

на чушма, казъм, за вода --

дали си стомни счупила,

или са в люби загледа, (2)
%%горните два стиха са разместени в машинописа - давам ги според записа

или на морни волове път чака?
\kray

Марийка дума мама си:

-- Мари мамо, мила мамо,

аз ни си стомни учупих,

нито са в люби загледах,

ни на морни волове път чаках,

най седях, мамо, най гледах,

най гледах, най си ярувах,

как горят, мамо, как тлеят

Караджовите дворове.

Не горят, мамо, от огън,

най горят, мамо, от бога,

от бога, от черна чума!
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
