%Куме лье, кум Костадине
%Сватбарска (``когато отиват за булката, из пътя'')
%Джиджев:сватбарски

\begin{multicols}{2}
-- Куме лье, кум Костадине,

пий вино, не опивай ся,
%%упивай вм. "запивай ся" - по записа и ръкописа

чи щем далеко да идем,

далеко, срещу бял Дунав,

млада кумица да вземем!
\end{multicols}
