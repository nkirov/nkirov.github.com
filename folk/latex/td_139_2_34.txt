\begin{multicols}{2}
Ти, момкова мале,

излези, погледни,

какво чудо водим:

тънко, та високо, (2)
\kray

бяло и цървено, 

на Стоян прилика,

на тебе отмяна! 
%%предпочитам "отмяна"
\end{multicols}

\area{Хасковско}
\year{1925}
