\begin{multicols}{2}
\rows
{Кани са Кольо на Рада, \uu{мари,} }
{кани са да я завлече.}

\rows
{Де зачу Рада тез думи, \uu{мари,} }
{сама си було изтъка, }

\rows
{сама си було изтъка, }
{сама са Рада пребули, }

\rows
{че скочи с бели бакъри, }
{че отиди за в'да студена. }

\rows
{Там си завари, завари }
{Кольо ми, нейното любе. }

\rows
{Кольо на Рада думаше: }
{-- Радо мо, любе Радо мо, }

\rows
{кога са, Радо, ожени, }
{кога ти беше сватбата? -- }

\rows
{Не чухме гайди да свирят, }
{не чухме зълви да пеят. }
\kray

\rows
{Рада на Кольо думаше: }
{-- Кольо льо, либе Кольо льо, }

\rows
{мама са болна разболя: }
{не даде гайди да свирят, }

не даде зълви да пеят! (2)

~~
\end{multicols}
