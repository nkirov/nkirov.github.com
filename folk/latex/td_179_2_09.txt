%Я изгрей, слънце, я изгрей
%Сватбарска (``когато прибулват булката'')
%Джиджев:сватбарски

\begin{multicols}{2}
Я изгрей, слънце, я изгрей

рано ми рано в ниделя,

да видиш, слънце, да чуеш,

че как са чедо разделя

от майка, чедо от баща,

от мили братя и сестри!
\kray

Радка мами си думаше:

-- Прощавай, мамо, прощавай,

дето съм яла и пила,

дето съм дрехи правила!

~~
\end{multicols}

\sing{Пяла Минка Господинова Вълева}
\village{с. Скалица}
\area{Ямболско}
\year{1928}
\re
