%Два царя, мамо, два царя
%На хоро (по Великден)

\begin{multicols}{2}
Два царя, мамо, два царя,

два царя, църни коларе, (2)

два царя кавга правили, (2)
%%"кавга" вм. "кавка" - според записа; възстановявам смисъла, в ръкописа "кавка"

правили и прилагали (2)
%%"прилагали" вм. "привадали" в ръкописа - според записа, възстановявам смисъла; стихът отсъства в машинописа

на сяко село кинига, (2)
%%бележка, че става дума за "книга"

кинига и по жандаре: (2)
\kray
%%до тук всеки стих се повтаря НК

кой има син, да писува, (2)

и който няма, да пише. (2)

Баню чорбаджи син нямал, (2)

той имал девет дъщерки -- (2)
%%"дъщерки" вм. "дъщери" - по ръкописа

най-мънинката Тодорка, (2)

тя била хитра й разумна, (2)

тя на баща си думаше: (2)

-- Ти иди, баще, на пазар, (2)

купи Тодорке хранен кон (2)

и още пушка бойлийка, (2)

и още сабя френгия (2)

да са Тодорка оружи! (2)

...
\end{multicols}

\dic{Оружи -- въоръжи.}
