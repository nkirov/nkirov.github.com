%Вързала йе божа майка
%Коледарска (в къща с малко дете)
%Джиджев:коледарски

%%добавям "(в къща с малко дете)" - по муз. запис

\begin{multicols}{2}
Вързала йе божа майка,
%%чува се: божа НК

\uu{под люляка разцъфтяла,}

вързала йе златна люлка,

положела малко дете,
%%"малко дете" вм. "млада бога" в машинописа - варианти!!!

залюля го и запя му. 

Де я зачула гората,

запряла се да послуша, 
%%чува се "възпряла се"
\kray

послуша и пак тръгнала;

де я зачула водата,

запряла се да послуша, 

послуша и пак тръгнала;

едно дърво не запряло, 

проклела го й божа майка:

-- Това дърво мир да няма, 

да трепери диня нощя!

Едно птиче не запряло,

проклела го й божа майка:

-- Това птиче кукувиче,

то да пее до средлято,

до средлято, до Еньовден!

~~
\end{multicols}

\sing{Пял: Стоян Иванов Вандов}
\village{c. Кортен}
\area{Сливенско}
\year{1908}
\rec
