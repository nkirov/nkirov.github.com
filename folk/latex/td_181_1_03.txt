%Я изгрей, слънце, я изгрей
%Сватбарска (``когато прибулват булката в неделя сутринта'') 
%Джиджев:сватбарски

\begin{multicols}{2}
Я изгрей, слънце, я изгрей

рано ми рано в неделя

да видиш, слънце, да видиш,

че как се чедо разделя

от майка, чедо, от баща,

от мили братя и сестри!
\kray

-- Че как ще речеш, Ганки мо, 

на чужда майка ``мале мо'',

на чужди баща ``тайно ле'',

на чужди братя ``драгинко'',

на чужди сестри ``калино''?
\end{multicols}

\year{1932}
