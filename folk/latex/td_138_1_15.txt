%Разболял се е млад Милен
%Хороводна (зимно, по сватби)
%Джиджев:хороводни

%%добавям (зимно, по сватби) от ръкописа и машинописа

\begin{multicols}{2}
Разболял се е млад Милен

под бейличките хамбаре, (2)

до широките мигдане, 
%%добавям стиха тук, според записа, долу се повтаря - пак според записа

край бейличките хармане,

на широките мигдане, (2)

дето хорото играи.

Майка му седи над глава,

тейко му седи над ноги,
%%чува се "тейко" - не пипам

мама Милену думаше:

-- Сино Милене, Милене,
\kray

дигни си, сино, главата,

главата от възглавката,

снагата от постелката,

да видиш, сино, да чуиш

какво се хоро играи

на широките мигдане --

до всяка мома и момък,

до твойто любе, до Руска,

до нехи, сино, играят

двамина вакли овчаре,

двамина с църни сукаре!

Милен на мама думаше:

-- Нека си, мамо, играят,

нека си къцат цървуле,

Руска е моя, мамо ле: 

пръстеня ми е у нехи,

пръстеня бурмалията; 

герданя хи е у мене,

герданя мержанлията!

И да играят, мамо ма,

Руска е моя, пак моя!
\end{multicols}

\ed{Бейлишки тур. 'който принадлежи на бег, бей'.}
