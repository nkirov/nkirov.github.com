%Слънцито трепна да зайде
%Жътварска (по мръкнало, на отиване)
%Джиджев:жътварски

%%на отиване - добавка от тефтера; според мен значи "на тръгване"
%%а не на отиване на нивата

%%разликите между звук и текстове са толкова големи, че ги давам отделно
%%тук пак подозирам здрави манипулации

\begin{multicols}{2}

\singvar
Слънцито трепна да зайде,

Марийка чикъм зафаща,

дите на Марийка думаше:

-- Хай да си, майко бе, вървим!

Марийка дите думаше:

-- Нивата зъ си пожена,

чикъма зъ си ожена,

снопите зъ си събера,

\textvar
%%(има малко общо с муз. запис)

Слънцито трепна да зайде,

Марийка чикъм зафаща,

малко \'{и} дите придума:

-- Мар, мамо, мар, мила мамо,

хайди да си, мамо, ходим,

чи времи вечи за ходине!

Марийка дите думаше:

-- Мойо мило дите, ти, малко,

мама зъ си чикъм ож\'{е}ни,
%%бележка, че "ще чекъм ожъне", а не "ожени"

пък тогава ще си, чедо, отидем!

Бързала Марийка, бързала,

чикъма Марийка свършила,

снопите Марийка събрала,

дитето за ръка фанала

и си във къщи отишла. 
\end{multicols}

\sing{Пяла: Гела Желева Димитрова}
\village{с. Мустрак}
\area{Свиленградско}
\bornYear{1926}
