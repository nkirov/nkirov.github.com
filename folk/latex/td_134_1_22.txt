%Задали ми са, задали
%Хороводна
%Джиджев:хороводни

\begin{multicols}{2}
Задали ми са, задали,

петстотян черни татари,

шестотян руси маджари,

гледайте да ги гледаме!

Назде щът, холам, да сочат? --

Те са сокали, сокали
%%трябва бележка - "сокали = сочили"

към Петка, към болярина,

към Петка, към джелепина.

Право в Петкови отидаха

и от конето слязаха,

сами си кони разхождат,

една сай дума сдумуват: 
%%предпочитам "сдумуват"
\kray

-- Хадти да ричем, да ричем:

``Петко льо, болярино льо,

Петко льо, джелепино льо,

нам са загуби, загуби

пет коня със пет товара

и с пет сребърни синджира, --

я ни конето платайте,

я жъ ви вземим момата,

момата хубавицата, (2)

булката гиздавицата!''

~~
\end{multicols}

\dic{Назде -- накъде.\\
Хадти -- хайдети, хайде.}

\sing{Пяла Димка Запрянова Матева}
\year{1933}
\village{Навъсен}
\area{Харманлийско}
