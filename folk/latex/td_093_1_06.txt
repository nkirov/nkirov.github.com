%Сама са пушка отпъна
%Хороводна (повръщаница)
%Джиджев:хороводни

%%добавям "(повръщаница)" - по машинописа

\begin{multicols}{2}
Сама са пушка отпъна,

та ми удари, удари (2)

два братя, до два близнака --

Стоян ми падна, па стана,
%%"па" вм. "ни" - по записа, значима промяна на смисъла

Иван ми падна, ни стана.

Над Ивану са виея

два орла, черни гарвани, 

Иван на орли думаше: 

-- Вийте са, вийте, два орла, (2)

два орла, до два гарвана,
\kray

с уста ми вода носете,

с крила ми сянка чинете,

доде ми душа излезе! 

Като ми душа излезе,

вие си долу кацнете,

бяло са мисо наяжте,

черни са кърви напийте;

дясна ми ръка ни яжте,

и на ръката -- ръкавът!

Вия ми ръка зимете

и си високо фръкнете,

и си широко гледайте

де има дворни каминни,

де има порти жилезни,

тамка ми ръка пуснете --

мама навънка ж' излези,

тя ж' ми ръка намери

и ж' ми ръка познае

по малко пръстче пръстенче!
%%"по" вм. "на" - за смисъла, в ръкописа е имало "по", сетне задраскано
\end{multicols}
