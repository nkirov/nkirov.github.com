\begin{multicols}{2}
Иван на Рада думаше:

-- Любе Раде ле, Раде ле, (2)

я щъ надолу да ида, (2)

надолу към Карадениз,

да си ма, любе, жалела (2)

година и половина --
\kray

на хоро да ни излизаш,

ни китки да се ни кичиш!

-- Любе Иване, Иване,

жалям та, любе, жалям та,

по сърце зъ та пожаля,

яла по лице ни мога! --

Сига са, любе, задало

гиздаво лято, пролято,

зъ цъфнат китки сякакви,

майка в градинка зъ флезе,

зъ набере китки сякакви

на мене зъ го подаде:

``Земи ги, Раде, кичи са!''

-- Любе Раде ле, Раде ле,

земи ги, Раде, кичи са,

по сърце да ма жалеиш!

~~
\end{multicols}

\ed{Карадениз -- неустановено селище или местност.}

\sing{Пяла Марина Стоянова Русева}
\village{ с. Лешниково}
\area{Хаманлийско}
\year{1902}
