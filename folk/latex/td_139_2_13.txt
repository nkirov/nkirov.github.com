%Тейнова булка, Гроздано
%Жътварска (сутрин)
%Джиджев:жътварски

\begin{multicols}{2}
-- Тейнова булка, Гроздано,

можеш ли да ми ожениш

нивата девет декара?

Гроздана нищо не каза,

станала рано сутринта 

и ми на нива застана,
%%в муз. запис - "и ми"

женала, що е женала,
%%що вм. ко - по записа, по-ясно

ден ми до кръста пладнина.

Когато кръста пладнина
%%чувам следващ стих "Когато кръста планина"
%%не разбирам смисъла: пладнина?? и пак не разбирам 
\kray

седнала, та обядвала,

легнала, малко поспала

и си на нива застана.

Женала, що е женала,

слънцето трепти да зайде,

Грозданка нива дожена

и е снопите събрала

и си към село тръгнала. 
\end{multicols}
