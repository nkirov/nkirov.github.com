\begin{multicols}{2}
\rows
{-- Бре, овчарко, оловарко, }
{изведи ма от таз гора,}

\rows
{изведи ма от таз гора, }
{от таз гора ясенова,}

\rows
{ясенова, росенова, }
{заведи ма на наш баир, }

\rows
{заведи ма на наш баир, }
{на наш баир, нашо село --}

\rows
{нашо село огин гори, }
{огин гори, митла мите!}

\rows
{-- Нека гори, да изгори }
{мойта майка, моят баща,}

\rows
{мойта майка, моят баща: }
{те храняли, те гледали,}

\rows
{те храняли, те гледали, }
{девет чеда изгледали,}

\rows
{девет чеда изгледали, }
{девет чеда момичета!}
\end{multicols}
