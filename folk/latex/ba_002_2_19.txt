%Мама йе Каля сгодила
%Хороводна (пее се на мегдана)

%%(пее се на мегдана)

\begin{multicols}{2}
Мама йе Каля сгодила,
%%предлагам "сгодила"

\uu{Кальо мо, годьеници ле,}

откак я й мама сгодила,
%%предлагам "сгодила"

тя легна Каля зля болна.

Ден лежи, Каля, ден ходи,
\kray

кат \'{и} сватбата наближи,

тя легна, Каля, секи ден.

Мама на Каля думаше:

-- Кальо мо, казъм Кальо мо,

хайди мяната да върним,

мяната, казъм, пръстена!

Каля мама си думаше:

-- Не връщай, мамо, мяната,

не ми разваляй сявдата,

сявдата, мамо, мерака --

от сявда чиляк умира,

умира, я полудява!
\end{multicols}

\bel{Си -- нестабилен тон.\\
Тактът, отбелязан с 2 е скъсен.}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
