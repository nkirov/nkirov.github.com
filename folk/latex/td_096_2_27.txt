%Гана турчин залюби
%Хороводна (право, винаги)

%%добавката (право, винаги) е моя - от ръкописа и машинописа

\begin{multicols}{2}
Гана турчин залюби,

\uu{Гане луда полуда,}

на шъга го й любяла,

на шъга китки давала,
%%стихът не е изпят
%%изпят е! НК

турчин шъга ни знае.

Турска съй сватба задало,

излязло мало й голямо
\kray

да видят сватбата де ж' иди,

сватбата у Ганкини отиде.

Ганка си плачи, ни рачи,

кадъни Гана думая:

-- Да ни си го лъгала,

да ни си го любяла --

турчин шъга ни знае!

~~
\end{multicols}

\sing{Пяла Куна Вълева Радкова}
