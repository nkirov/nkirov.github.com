%Погърмя, потрищя
%Хороводна
%Джиджев:хороводни
%%в записа няма изрична уговорка, че песента е хороводна - дали не в графа "без уточнена функция"

\begin{multicols}{2}
Погърмя, потрищя

снощи на вичеря,

дребян дъжд удари,

сиво стадо свари: 
%%"свари" наместо "овчари" -- според записа.

стадо недоено,

чердак непреметен,

ведра неумити,

цидилки непрани. 

-- Овчари, овчари,

де ви й кихаята?

-- Нам ни й кихаята,

в село, на мех'ната,

руйно вино пие,

млади булки гледа --

наред наредени,

се с бели дюлбени,
\kray

с алени престилки,

с белополи ризи.

~~
\end{multicols}
