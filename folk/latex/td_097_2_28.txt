%Съди-гъди Ганка, Ганка да ни ходи
%На моабет
%Джиджев:трапезни

\begin{multicols}{2}
Съди-гъди Ганка, \uu{мамо}, Ганка да ни ходи,

\uu{ах, мила мамо}, в Долната махала,

дето Ганка ходи, там никой ни ходи,

там никой ни ходи,

само идно ходи, турско арнавутче,

селско коруджийче, 
\kray

то си Ганка либи, либи щи я вземи,

либи щи я вземи!

Ганка мама дума, Ганка мама дума:  
%%изобретявам повторението "Ганка проговаря", за да напасна стиховата организация; 
%%правя леки промени и в долния стих

-- Мамо, мила мамо, 

ти, га ни си щяла, турчин зет да имаш,

дъщеря кадъна,

ти да си ма малка, малка закопала,

в турските гробища,

там да ма приливат, белите кадъни,

белите кадъни!
\end{multicols}

\bel{Същата песен от кавала, изпята от жена му.}

\sing{Пяла Недялка Донева Йонкова}
\year{1915}
\village{с. Розовец}
\area{Пловдивско}
