%Войник замина
%На хоро (вечерно)
%Джиджев:хороводни
%%като че ли има начално "и"

\begin{multicols}{2}
Войник замина (2)

приз момини д'орье, (2)

мома го видье, (2)

тя го приследи. (2)

Войник хи дума: (2)
%%"войник" вместо "Мър, той" - според записа

-- Мар, върни ми са, 

малка дивойко,
\kray

нашта е село
%%нашту вм. "нащу" - същото, но по-четивно

твърде далеко! --

Еднъж го кажат
%%"еднъж" вм. "еднаж" - според ръкописа

гора зелена,

другож го кажат

море дълбоко.

-- Върви ми, върви,

младо войниче,

я ще да стана

пиле фъркато,

та ще прифръкна

гора зилена;

я ще да стана

риба мирянка,

та ще прибродя

море дълбоко!
\end{multicols}

\sing{Пяла Гина Ангелова Георгиева}
\village{ с. Лешниково}
\area{Хаманлийско}
\year{1936}
