%Иван на Яна думаши

%%(пее се на засевки на двора)

\begin{multicols}{2}
Иван на Яна думаши:

-- Яно мо, Ангилино мо,

играиш, Яно, играиш,

играиш, нищо не знаеш:

майка ти, Яно, умряла,

пък баща ти го запряха (2)

за майкенити борчове,
\kray

за вино и за ракия!

Яна Ивану думаше:

-- Иване, любе Иване,

досега си бил любовник,

отсега натъй побратим --

оттука да ме поведеш,

във дома да ме заведеш!

Иван си Яна поведе,

че я във тях си заведе,

Яна Ивану думаше:

-- Иване, любе Иване,

бре, как ма хитро излъга

самичка у вас да дойда!

~~
\end{multicols}
