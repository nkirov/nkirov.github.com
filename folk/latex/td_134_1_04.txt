%Превяле са, пресяле са
%Сватбарска (``на замески'')
%Джиджев:сватбарски

\begin{multicols}{2}
Превяле са, пресяле са

девят сита копринени,

под ситата -- темна мъгла,

под мъглата -- младо булче, 
%%"под" вм. във" - по записа

над глава му -- треперушка,
%%"над" вм. "на" - по записа

треперушка треп трепаря, (2)
%%чува се: трипаря НК

треп трепаря, лед лъщяи!

~~ 
\end{multicols}
