{Разболял ми съ й млад Милен,}
{млад Милен, млада войвода,}

\rows
{все лежи девет години,}
{девет постели е съдрал.}

\rows
{Миленовата майчица,}
{тя на Милена думаше,}

\rows
{тя на Милена думаше:}
{-- Синко Милене, Милене,}

\rows
{какво се хоро играе}
{пред нашти порти високи --}

\rows
{до всяка мома и ерген,}
{до твойта Радка двамина!}

\rows
{-- Нека си, мамо, играят,}
{Радка е моя, пак моя:}

\rows
{пръстенът ми е у нея,}
{чумберът \`{и} е у мене!}
\end{multicols}
