%Чичо ле, чичо Караджо
%Седенкарска (на летни попрелки)
%Джиджев:седенкарски

\begin{multicols}{2}
-- Чичо ле, чичо Караджо,

в село [д]желепе дойдаха, 

[д]желепе и кърджелепе,

тежки салгъне мятаха,

мятаха и размятаха:

кому по двеста, по триста,

мене са, чичо, метнали

хиляда и петстотини --

тва приплата ли са, чичо? 
\kray

-- Стойене, чичо Стойене,

теб ли кахъра забера? --
%%теб вм. тебе - за изрядност в сричките

Хамбар ти пълен с пшеница,

ти имаш серес биволи

и имаш кола железна: 

тежка кирия нат'вари, 

във Стамбол да я закараш,

закараш, да я продадеш,

салгъне да си изплатиш!
\end{multicols}

\dic{Салгъне -- данъци.} 

\ed{Кърджелепе -- производна дума от ``джелеп, джелепин'' тур. 'търговец на добитък'.\\
Салгън 'епидемия', 'настинка'. В контекста -- със значение 'данък, налог'.}
