%Драгано, Драганке
%На хоро (на мегдана, на Лазарица и при други случаи)
%Джиджев:хороводни

\begin{multicols}{2}
Драгано, Драганке,

слънцето трипере, (2)

трипере да зайде

зад гора зилена,

\five зад поле широко.

Иван кон сидлаве,

сидлаве, бидлаве,

буля му го пита:

-- Бре драгинко й [И]ване,

\ten мил булян деверо,

къдя щеш да идеш?
\kray

-- Ж' ида, бульо, ж' ида

на сбор, на панаир,
%%предпочитам "сбор" 

на света Троица,

\fifteen гайда жъ си купа,

гайда -- за дванайсе, (2)

кавал -- за тринайсе, (2)

свирка -- за петнайсе!

~~
\end{multicols}
