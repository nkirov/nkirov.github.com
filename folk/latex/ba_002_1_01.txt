%Тръгнала йе Яна
%Хороводна (на засевки)
%Джиджев:хороводни 

%(пее се на засевки)

%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските 
\begin{multicols}{2}
Тръгнала йе Яна

приз гора зилена

билки да набере,

билки ляковилки.

Не намери, Яна

билки ляковилки,

най намери Яна
%%вместо "най е намеряла" - според записа 

брата си Ивана,
\kray

той под пелин лежи,

пелин върши кърши, (2)

здрава глава вържи.

Яна Ивану дума:

-- Байни ле, Иване,

що под пелин лежиш,

пелин върши кършиш, (2)

здрава глава вържиш?

Иван Яна дума:

-- Яно, сестро Яно,

на тебе да кажа,

ти си от мен малка,

пък си бързорека,

не можеш да траеш, (2)

на мама ша кажеш;

пък мама ша каже

на татя й на батя,

пък бати ша каже 

на буля й на кака,

пък буля ша каже

на нейните сестри,

а те щат да кажат

на техните хайки!

Яно, сестро Яно,

змейница ме люби, (2)

змейница вдовица, (2)

с четири дечица!
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
