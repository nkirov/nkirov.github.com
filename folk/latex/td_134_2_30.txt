\begin{multicols}{2}
Заспало ми йе дитенце,
%%чува се: дитенце НК

мамино, мило момченце,
%%чува се: мумченце НК

майка му го разбуждаше:

-- Стани ми, стани, дитенце,

\five мамино, мило момченце,

да погледаш лазарките

как хубавичко играят

със желти чехли потропват!
\end{multicols}
