%Карай, Яно, карай, накрай да излезем 
%На жътва (след обед)
%Джиджев:жътварски

\begin{multicols}{2}
-- Карай, Яно, карай, накрай да излезем,

че накрая има сянка оряхова,

и под сянка, Яно, три моми загорки,

три моми загорки, за Загоре питат,
\kray

за Загоре питат, ром\'{а}ния кълнат:

``Пуста да остане, тази ром\'{а}ния --

девет братя имах, сал един остана,

и той глава вързал, глава й половина!'' 
%%й вм. и - за сричките

~~
\end{multicols}
