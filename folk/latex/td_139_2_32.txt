\begin{multicols}{2}
-- Тва, що са белейе,

белейе, лъщейе,

срещу тих бял Дунав,

дали са сватове

\five или са снягове?

-- 'Ку бяха снягове,

щяха да са стопят,

'ку бяха сватове,

щяха да притръгнат!

\ten Не били снягове,

най били сватове.

~~ 
\end{multicols}
