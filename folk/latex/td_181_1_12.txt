\begin{multicols}{2}
Иван на Рада думаше:

-- Радо мо, любе Радо мо, (2)

аз ще на жътва да ида

отвъд морето, в полето,

\five дето са ражда, Радо мо, (2)

белия ориз, Радо мо,
\kray

черния пипер, любе ле,

тебе те тука оставям 

ни болна, Раде, ниволна --

\ten кога си рожба добиеш,

хаберя да ми проводиш

по ствичка, по лястовичка!

Ако е, Радо, момченце,

аз ще му, Радо, изпратя

\fifteen девет катъра алтъни;

ако е, Радо, момиче,

аз ще му, Радо, изпратя

мотика, Радо, лопата,

бял тамян, Радо, жълт восък!
\end{multicols}

\ed{\el{Ствичка} -- диалектна форма от ,,птичка``.}
