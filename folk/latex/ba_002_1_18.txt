%Мари, накрай село
%Хороводна (на засевки)
%Джиджев:хороводни 

%%(пее се на засевки)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските
%%Име на песента според Джиджев: Мари, накрай село една къща имаше НК

\begin{multicols}{2}
Мари, накрай село

дор три коня пасяха,
%%следващите два стиха възстановявам от нотите и записа, по-следващите 2 - от записа. 

мари, бири кътър, 

бири бейгир бир даа,

\rows
{мари катър бейгир, }
{хер сезин олсун абеним. }

Мари, накрай село

една градинка имало,
%% разлика с музикалния запис НК

\rows
{мари, в градинката }
{дор три цветя цъфтяха. }

Мари, бери лале,

бери зюмбюл бир да гюл,
\kray

мари, лале зюмбюл,

хер сезин олсун гюл беним.

Мари, накрай село 

дор три коня пасяха,

мари, бери катър,

бери бейгир бир да ат,

мари, катър бейгир,

хер сезин олсун ат беним.

Мари, насред село

една къща имаше,

мари, във къщата 

дор три жени живеят,

мари, бери баба,

бери гелим бир да къс,

мари, баба гелим,

хер сезин олсун къс беним.
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
