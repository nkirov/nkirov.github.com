\begin{multicols}{2}
Митьо на Бойка думаше: (2)

-- Бойки мо, любе Бойки мо, (2)

доходят ли ти, Бойки мо, (2)

сутрана меки милинки, (2)
\kray

на обяд топли симиди, (2)

за вечер геврек, гевреци? (2)

Бойка на Митьо думаше: (2)

-- Дохождат, Митьо, дохождат, (2)

провождай, Митьо, провождай, (2)

че аз ща тебе да зема! (2)

~~
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
