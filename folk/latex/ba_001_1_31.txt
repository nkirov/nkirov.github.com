%Похваля са Видяна буля
%Коледарска (пее се на мома)
%Джиджев:коледарски

%%добавям "(пее се на мома)" - от муз. запис

\begin{multicols}{2}
Похваля са Видяна буля,

\uu{Видо льо, момни ле, колади ле,}

че си има хубава Вида:

-- Нийде ми няма хубава Вида:
%%хубава вм. "калина" - по муз. запис: може като варианти

ни в Цариграда в бели гъркини,
%%тук и надолу чувам по-скоро "у" вм. "в"

ни в Дряно поле в бели кокони,

най ми има хубава Вида

в Котел града, във Котелско! 
%%в муз. запис "Котелско"
%%чува се: във Котелско НК

Де я чуха два овчара,

два овчара, два другаря,

те са чудят и са маят

как да видят хубава Вида. 

Че събраха двете стада,
%%стихът е изпуснат в машинописа, тук по муз. запис

начесаха перчените,

накривиха калпаците,

подсукаха мустаците,
%%в муз. запис "подсукаха"

нарамиха криваците,

че сгласиха два кавала,
%%в муз. запис "кавалите"

два кавала, две уйдурми,
%%стихът не е изпят

помамиха сиви стада,

че минават край Видяни,
%%чува се: Видяни НК

край Видяно равни двори --
%%чува се: Видяно НК

дано излези хубава Вида. 
%%в муз. запис "видат" вм. "излези" - може да отбележим вариантността

Не излязла хубава Вида,
%%в муз. запис "не видели" вм. "не излязла" - може да отбележим вариантността

най излязла Видяна буля, 
%%чува се: Видяна НК

че повика Видяна буля:
%%в муз. запис - "извика"
%%чува се: Видяна НК

-- Излез, излез, калино Видо,
%%чува се: излез НК

излез, излез, да погледаш
%%чува се: излез НК

какво чудо в пътя върви!

Кат излезе хубава Вида,
%%в муз. запис - "че излязла"

лице \'{и} са бялеяше
%%в муз. запис този и следващите два стиха са изпяти в края на песента
%%чува се: билеяше НК

като пряспа дваж навята,

дваж навята, триж присята; 
%%това е края на песента НК

очи \'{и} са чернееха

като угар дваж орана,

дваж орана, триж бъркана;

вежди \'{и} са тънчеяха
%%чува се: тънчеяха НК

като гайтан на нов дюкян. 
%%дякюн вм. дюген - по муз. запис
%%чува се: гайтан НК
\end{multicols}

\sing{Пял: Стоян Иванов Вандов}
\village{c. Кортен}
\area{Сливенско}
\year{1908}
\rec
