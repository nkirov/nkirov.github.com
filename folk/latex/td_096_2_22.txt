%Снощи си доде млад Ваньо
%Лазарска (когато се съберат у кумата на Великден) 
%Джиджев:лазарски

\begin{multicols}{2}
Снощи си доде млад Ваньо,

\uu{Ваньо льо, вакъл карагьоз,}

ощи от коня ни е слял,

той на мама си думаше: 

-- Вземи ми, мамо, Иванка,
%%добавям "ми" - по записа, наместват се сричките

Иванка -- снага напета,

Иванка -- лице пребяло!
\kray

Ако я вземиш, ако не,

аз щъ да ида във Влашко,

във Влашко, мами, при чичо --

чичо от сивда забягна:

сивда от болес по-лошо,

сивда от огин пo-п\'{а}ри,

сивда от болес по-боли!

~~
\end{multicols}

\sing{Пяла Куна Вълева Радкова}
\village{с. Бабяк}
\area{Пловдивско}
\year{1904}
\rec
