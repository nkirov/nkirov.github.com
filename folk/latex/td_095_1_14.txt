%Разболя се Яна
%На жътва (сутрин) 
%Джиджев:жътварски

\begin{multicols}{2}
Разболя се Яна, 
%%чува се: разболя се НК

\uu{Яно, моме Яно,}

на арабска нива,

арабче я варди,

и са богу моли: 
%%чува се: и са богу моли НК
\kray

-- Боже, вдигни Яна,

курбан ша ти кола

на ден, на Гергьовден; 

свещи да ти пала

на ден, на Спасовден!

~~
\end{multicols}
