\begin{multicols}{2}
Гена аби тъче, (2)

\rows
{аби-ямурлуци}
{и черни шеяци.}

До стана \`{и} има (2)

\five шише със ракия, (2)

пък на натрата \`{и} --

тава със смокини.

Виднъж Гена тупни, (2)

\rows
{две ракии цукни,}
{\ten три смокини смукни.}

Генините братя, (2)

\rows
{те на Гена думат,}
{думат и я мъмрат:}

--  Гено, сестро, Гено, 
%%тук се нарушава структурата на повторенията на стиховете - дали нарочно? НК
 
\fifteen Гено, откажи са
 
от туй пусто пийки! (2)

Нали си сгодена -- (2)

\rows
{ша чуй годеника,}
{годежът ша върни,}

\twenty годежът ша върни: (2)

\rows
{пръстен бурмалия,}
{рубе чаръклия!}
\end{multicols}
