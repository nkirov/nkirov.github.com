%Находи ли са, Гино мо
%Сватбарска (га прибулват булката)
%Джиджев:сватбарски

\begin{multicols}{2}
-- Находи ли са, Гино мо,
%%"Гену" вм. "Гину" - надолу навсякъде е Гена

бащини равни дворови, (2)

майчина герджик примяна?

Как ще да трайе, да трайе

\five Гинин'та майка за Гина, (2)

за Гинин'та младос и хубос?
%%разменям местата на "младост" и "хубост"
\kray

-- Както й траяло, траяло

сухото доли за вода,

широко поле за трива

\ten и сухото дърво за листа,

тъй ще да трайе, да трайе

Гинин'та майка за Гина,

за Гинин'та хубос и младос!
\end{multicols}
