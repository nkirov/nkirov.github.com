\begin{multicols}{2}
Ни сме знаяле, 
%%чува се: Ний сме знаяле

\uu{драги куме ле,}
%%чува се: драги куме ле

ни сме знаяле,

че щеш кум да се, (2)
%%нататък чувам "кум да помолъ /да ти излеем" - песента се насочва в друга посока. Какво да правим?

кум да кумуваш, (2)
%%чува се, няма го в машинописа НК

да те излеем, (2)
%%чува се, няма го в машинописа НК
\kray

да те увием (2)

вити краваи, (2)

и на краваи --  (2)

златни башчии, (2)

и във башчии -- (2)

златна ябълка! (2)

~~ 
\end{multicols}

\ed{Башчии -- вер. от ``бахча'' тур. 'зеленчукова градина'.}
