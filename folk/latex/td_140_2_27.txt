%Субашче Стана думаше
%На хоро (лятно)
%Джиджев:хороводни

\begin{multicols}{2}
Субашче Стана думаше:

-- Стано мари, бяла Стано, (2)

земи мене, бяла Стано!

-- Земам та, земам, субашче,

'ку сбереш лято до лято
%%предпочитам "сбереш"

и още зима до зима,
\kray

прикарай море приз д'оре!

Замъчило се субашче

от Гьоргьовден до Гьоргьовден,

сберало лято до лято
%%предпочитам "сберало"

и още зима до зима,

прикара море приз д'оре,

чи у Станини отиде

и си на Стана порука:

-- Излези, бяла Стано льо,
%%"излези" вм. "Излети" - напук на ръкописа, в името на смисъла; може да е било "излез ти"

да видиш лято до лято

и още зима до зима,

и още море приз д'оре!

~~
\end{multicols}

\sing{Пяла Гина Ангелова Георгиева}
\village{ с. Лешниково}
\area{Хаманлийско}
\year{1936}
