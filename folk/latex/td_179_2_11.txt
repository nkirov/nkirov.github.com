%Женала й Милка, Милке мо
%На жътва
%Джиджев:жътварски

\begin{multicols}{2}
Женала й Милка, Милке мо,

дорде йе пладня станало,

седнала да си почине,

като й седнала, легнала,

като легнала, заспала,

кат насън майка си видяла,

видяла и хортувала. 

Мама на Милка думаше: 
\kray

-- Милки мо, мила мамина,

много та малка оставих,

ни можах да те науча

на моя, Милке, табихет --

кога доди сетна събота, 

да си главата не миеш,

да си дюлбена не переш,

че вече ми се й додяло,

додяло, Милке, дощяло

хумена вода да пия,

хумена и сапунена!

Силни вятрове духнали,

Милка от сън събудили,

Милка ги люто кълнеше:

-- Вятро льо, ненавейнико,

защо ма от сън събуди,

таман със мама хортувах?!
\end{multicols}

\sing{Пяла: Минка Господинова Вълева}
\village{c. Скалица}
\area{Ямболско}
\bornYear{1928}
