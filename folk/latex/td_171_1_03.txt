%Побляла йе вакла йовца
%Коледарска 
%Джиджев:коледарски

%%бележка от Лозанка - под нотите: "Има недописани няколко такта"
%%"- сравнение със звукозаписа"

\begin{multicols}{2}
Побляла йе вакла йовца,

\uu{млада ми Калино льо,}  
%%дали не са роднински названия?? НК

\uu{бяла Малино льо,}    
%%дали не са роднински названия?? НК

снощи вечер в полугаре:

хем поблява и кръстосва,

и кръстосва триста йовце,

триста йовце, тежък сайван.
%%чува се: тежка НК

Де я зачу вакъл йовчар,

вакъл йовчар от планина,
%%стихът не е изпят

от планина, от равнина,
%%стихът не е изпят

отговаря вакъл йовчар:

-- Ой та тебя, вакла йовца,

що побляваш толкоз жално:

гладна ли си или жадна?

Отговаря вакла йовца:

-- Ой та тебя, вакъл йовчар,

не съм гладна, не съм жадна,

най ми няма шаро ягне,

шаро ягне, най-милото!

Чувай, чобан, тебя пеем!
\end{multicols}

\sing{Пели: Група за коледарски песни при читалище ``Н. Й. Вапцаров'',  с. Каравелово}

\area{Ямболско}
