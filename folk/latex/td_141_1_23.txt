%Заспала мома Дафина
%На хоро
%Джиджев:хороводни

\begin{multicols}{2}
Заспала мома Дафина,

\uu{Дафино, моме хубава,}

под дърво, под дафиново,

под клонче, под маслиново.
\kray

Ситен ми дъждец повал\`{е},
%%добре е да има ударение "повалЕ"

сепна се мома, разбуди

и почна люто да кълне:

-- Дъждо льо, ненавалел се,

вятро ле, ненадухал се?! --

Стана ми девет години,

откък е мама умряла,

насън я не бях видяла,

сигя ми се мама присъни
%%"присъни" вм. "пресъни", "сигя" вм. "сиге" - по ръкописа; второто разчитане е спорно, но уяснява смисъла

и тихичко ма запита:

-- Дафино, мила мамина,

стана ли добра къщница?

Дафина мама говори:

-- Аз си са, мамо, ожених

и си са, мамо, задомих!
\end{multicols}
