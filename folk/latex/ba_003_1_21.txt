%Сънлива, дремлива дивойко
%Хороводна (пее се на засевки)
%Джиджев:хороводни

%%(пее се на засевки)
%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
-- Сънлива, дремлива дивойко, (2)

сън та на хорото ни пуска! (2)

-- Сън ме пуска, мама ме не пуска, (2)
%%"пуска" вм. "пуща" - според записа, както е и в горните стихове

защото е мойта майка мащеха -- (2)
\kray

кат поискам на хоро да ида, (2)

тя ми дава две пещи да паля, (2)

паля, паля, дорде ги опаля. (2)

Пак поискам на хоро да ида, (2)

тя ми дава две хурки да преда, (2)

преда, преда, дорде ги опреда. (2)

Пак поискам на хоро да ида, (2)

тя ми дава два стана да тъка, (2)

тъка, тъка, дорде ги изтъка, (2)

и хорото се е вече пръснало. (2)

~~
\end{multicols}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
