%У момяни двори
%На сватба (``когато извеждат момата'')
%Джиджев:сватбарски

\begin{multicols}{2}
У момяни двори

тъмна мъгла пада,

у момкови двори

ясно слънци греи. 

-- Я ни плачи, моме,

сига са ни плачи! (2)
%%чува се, няма го в машинописа НК

я ни тъжи, моме,
%%не се пее НК

сига са ни жали,
%%"сига" вм. "тука" - по записа

сига са ни плачи!
%%"сига" вм. "тука" - по записа
%%не се пее НК
\kray

Тука имаш, моме,

само идна майка,

кога идиш, моме,

във момкови двори,

тамка имаш, моме,

свекър и свекърва, (2)

зълви и девери!

~~
\end{multicols}

\sing{Пяла Дафина Николова Кучева}
\village{с. Свежен}
\area{Пловдивско}
\year{1910}
\rec
