%Скрийте ма, брайно, скрийте ма
%Сватбарска (``когато влезат в двора на момата'')
%Джиджев:сватбарски

\begin{multicols}{2}
-- Скрийте ма, брайно, скрийте ма

във вашта равна градинка,
%%вашта вм. ваща - същото, но по-ясно. 

под червения трендафил!
%%стихът е изпят, но не е записан в ръкописа и машинописа

Братя на сестра думаха:

-- Не бива, сестро, скриване,

най бива, сестро, даване!
\end{multicols}
