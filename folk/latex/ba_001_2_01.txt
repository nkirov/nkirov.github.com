%Пасла й мома братом коня
%Коледарска (из пътя)
%Джиджев:коледарски

%%добавям "из пътя" - по муз. запис

\begin{multicols}{2}
Пасла й мома, \uu{колади ле,} 

братом коня, \uu{колади ле,} 

как го й пасла, сън заюсна,

сън заюсна, кон изпусна,

кон преплава през бял Дунав.
%%чува се "през бял" вм. "тих бял"

Покрай върви вакъл овчар,
%%"Покрай въри" вм. "оттъй Дунав" - по муз. запис
%%чува се: покрай въри, предлагам "върви" НК

мома му са жално моли:
\kray

-- Ой та тебе, вакъл овчар,

улови ми братом коня,

ша та даря добра дара,

добра дара -- вита китка!

Отговаря вакъл овчар:

-- Ако ща китка, изви ща си!

Мома му са пак замоли:

-- Улови ми братом коня,

ша ти даря тънка риза!

-- Ако ща риза, купи ща си!

Мома му са пак моляше:

-- Улови ми братом коня, 

ша те даря тънка снага!

Той улови хранен коня. 
\end{multicols}

\sing{Пял: Стоян Иванов Вандов}
\village{c. Кортен}
\area{Сливенско}
\year{1908}
\rec
