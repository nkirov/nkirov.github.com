%Стоян на Ганка думаше
%Хоро (на Лазарица) 
%Джиджев:лазарски

\begin{multicols}{2}
Стоян на Ганка думаше:

-- Люби Гане ле, Гане ле,

чакай ма, любе, почакай

службата да си изкарам,

службата, трите године!
%%чува се: трите године НК

Ганка на Стоян думаше:
%%"Стуян" вм. "Герги" - по записа, така е и в началото

-- Люби, Стояне, Стояне,
%%"Стуяне" вм. "Герги ле" - по записа, така е и в началото

-- Която й мома сбързала,

%%съкращавам стихът "за Лазарица примяна" - не е изпят, не се връзва в логиката на песента

тя тебе три дни не чака,

белким ти до три години!
\end{multicols}

\sing{Пяла Гела Желева Димитрова}
\village{с. Мустрак}
\area{Свиленградско}
\year{1926}
\re
