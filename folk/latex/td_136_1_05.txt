%Сокол гони
%Коледарска (на мома)
%Джиджев:коледарски

\begin{multicols}{2}
Сокол гони, \uu{Дойне ле, коладе ле},  

пъстра птичка, \uu{момне ле, коладе ле}, 

кък я гони, увива я, 

увива я у момян двори.

Птичка пищи: ``Изляз, изляз,
%%ха така - пряка реч насред стиха!!!

да ма 'тървеш от тоз сокол:
%%'тървеш вм. утървеш - по муз. запис
\kray

ше ти пея сутрин рано,

сутрин рано, вечер късно,

вечер късно на лягане;

ше ти извия вито гнездо,

ше ти снеса сини яйца,

ше ти измътя пъстри пилци;

ше ги изуча, кога тръгниш,

кога тръгниш за в'да да идиш,

да ти кацат по кобилица!

Който чуйе, да са чуди,

който види, да завиди: 

``Блазя, боже, на таз мома

как изучи тези пилци!''
\end{multicols}

\sing{Пял Стоян Гьоргев Гайдаров}
\village{c. Победа}
\area{Ямболско}
\year{1895}
\rec
