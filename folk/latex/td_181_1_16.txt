%Слънцето трепти, зайдева
%На жетва (по икиндия)
%Джиджев:жътварски

\begin{multicols}{2}
Слънцето трепти, зайдева,

Димитър чикъм зафаща

между две сине баиря,

между две гори зилени, (2)

между две води студени, 

сестра му Радка до него. 

Димитър Радке думаше:
\kray

-- Я викни, сестро, та запей --

от едно гърло два гласа,

от един език две думи!

Радка байно си думаше:

-- Ни мога, байно, да пея,

ни мога, байно, ни смея,

мойто е любе хайдутин:

по гласа ще ма познае,

при нази, байно, ша доди,

тебе ще, байно, убие,
%%съкращавам "да" от "да убие" - за сричките

мене ще, байно, да вземе!
\end{multicols}
