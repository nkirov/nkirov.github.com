\begin{multicols}{2}
\rows
{Седянки се кладяха, }
{мене ме не пущаха, }

\rows
{мама й тате викаха: }
{``Слагай да вечеряме,}

\rows
{постилай да лягаме!'' }
{Сложих, вечеряхме, }

\rows
{постилах и легнахме, }
{мама, тате заспаха,}

\rows
{меня сън ме не хваща, }
{че излязох на двори}

\rows
{и погледнах нагоре: }
{сяко любе с любето,}

\rows
{мойто любе е само, }
{другари му викаха:}

\rows
{``Ела седни тук при нас, }
{друго любе залюби!''}
\end{multicols}
