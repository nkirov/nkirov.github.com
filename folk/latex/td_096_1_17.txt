%Тиче Дунав мътян-калян
%Из къра (когато ходи с овцете)
%Джиджев:жътварски

\begin{multicols}{2}
Тиче Дунав мътян-калян, (2)

та си влачи малка мома, (2)

покар Дунав йовчар ходи,
%%бележка, че "покар" значи "покрай"

йовчар ходи, стадо пасе,

малка мома му са моли:

-- Ха бре, овчарю, млад чобан, (2)

влез, чобаньо, извади ма, (2)

ще ти даря добра дара, (2)

добра дара, тънка риза!

Млад чобан си отговаря:
%%в муз. записа "Отговаря вакъл овчар"

-- Ази не съм млад бакалин
%%в записа "аз не съм си млад бакалин" - сричките са еднакви
\kray

да си нося тънка риза,

най си искам, малка моме,

твойта, моме, тънка снага,

тънка снага, бяло лице!
\end{multicols}

\editor{Покар 'покрай'.}
