%Как бе рикла Янка
%Хороводна (пее се на засевки)

%%За мен цикълът „на засевки” трябва да се отнесе като подцикъл към сватбарските

\begin{multicols}{2}
Как бе рикла Янка: (2)

\rows
{-- Дай ма, мамо, дай ма, }
{дай ма, ожени ма,}

далеч друго село,

на Руженско село, (2)
\kray

на руженски ерген!

Нека да съм далеч,

лесно да се хваля, (2)

хвалба да ми мине, (2)

хвалба и хората --

как ходела Янка,

как носела Янка

алени престилки,

белополи ризи,

ширити, пирити,

пера паунови!

Послушали Янка, 

че я далеч дали,

далеч, друго село,

на Руженско село.

Хваляла се Янка

до година, до две,

хвалба й не минава,

хвалба, не хората.

Хабер проводила:

-- Нека додат мама,

мама й голям бати,

тети да не доди,

той ма далеч дади,

далеч, друго село!

~~
\end{multicols}

\ed{Руженско село -- може да е с.\,Ружица, общ.\,Болярово, обл.\,Ямбол.}

\sing{Пял Стоян Иванов Вандов}
\year{1908}
\village{Кортен}
\area{Сливенско}
