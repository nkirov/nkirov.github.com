\begin{multicols}{2}
\uu{Мaри Дрaгано,} тейно Дрaгана кълнеше: 

-- Няма ли боляс за тебе,

я болес, я черна чума,
\kray

да ти лицето посърне,

да ти снагата oвехне,

да ти кoсата изпада,

чи ми са й вече додяло:

сяка ми вечер -- момари,

тая ми вечер -- сватбари?!
\end{multicols}
