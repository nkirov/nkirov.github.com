%%добавям "(на мома с много поклонници)" - по муз. запис

\begin{multicols}{2}
Изтекло йе идно дърво,

\uu{Добри ле, да мой Добре,}

\uu{Добри ле, мой коладе,}
%%(стих 2-3 са подчертани - рефрени ли са???)

стволовато, клоновато.

Под дървото триста млади,

триста млади, голобради

наред седят, ядат-пият,

на б\'{а}ш седи войник Петър. 
%%(у Джиджев "на баща" - сменям "на б\'{а}ш")
%%чува се "на баш" - така трябва да е заради сричките
\kray

Край тях мина малка мома,

на рамо \'{и} кобиличка,

кобиличка с бели менци;

на глава \'{и} кичом китка,

кичом китка, вихро венче, 

край тях мина, селям н' даде. 

Обади се войник Петър:

-- Ой ва вази, триста момци,

що седите, що гледате? --

Я станете двама-трима,

уловете малка мома,

напийте \'{и} студна вода,

отземете й кичом китка,

кичом китка, вихро венче!

Че станали двама-трима,

проговаря малка мома:

-- Ой ва вази двама-трима,

мир минете й заминете,

че аз имам трима братя,

ще река на мойте братя:
%%олком обадя на мойте братя: 
%%????(ако)

първия е при кадия,

при кадия кафеджия;

втория е при пашата,

при пашата чибукчия;

третия е при султана,

при султана велик везир!
\end{multicols}

\ed{Селям тур. 'поздрав,  поздравяване'. Селям алеким тур. 'добър ден'.}

\rec
