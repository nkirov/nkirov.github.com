\begin{multicols}{2}
Кавали свирят, мамо, в усои, 

Грозданка мете, мамо, дворове. (2)

-- Я вземи, мила мамо, метлата,

па дай на татка, мамо, греблото, (2)

пък аз ще ида, мамо, в усои

да видя, мила мамо, кой свири! (2)
\kray

Ако е, мила майко, мил братец,

ще се върна, майко, веднага; (2)

ако е, мила майко, любето,

ще се завърна, майко, след година (2)

със мъжко дете, мамо, на ръце

и с друго, мила майко, на сърце! (2)
%%"майка" вм. "майко" - за смисъла
\end{multicols}

\sing{Пяла Тотка Георгиева Кацарска}
\year{1936}
\village{с. Розовец}
\area{Пловдивско}
