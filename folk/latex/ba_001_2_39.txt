%(на хоро на мегдана)

\begin{multicols}{2}
\rows
{Очи, черни очи }
{шалави, палави,}

\rows
{шалави, палави, }
{що плахо гледати -- }

\rows
{заложени ли сте }
{или продадени?}

\rows
{-- Не сме заложени,  }
{нито продадени --}

\rows
{мене мама й била, }
{била и сгодила}
\kray

за Банова сина,

за Мерджан войвода.

Аз си Бана не знам,

не знам, не познавам,

сал като го чувам

Бано болярина, (2)

Бано джелепина, 

много рода имал,

рода и роднини;

мойто й дари малко:

двестя шити кърпи, (2)

триста сюлючени,

петстотин ръкави,

шестотин мисалки!
\end{multicols}

\ed{Сюлючени -- нефиксирана диалектна дума.}

\sing{Пял Стоян Иванов Вандов}

\area{Сливенско}
