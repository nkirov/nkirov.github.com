%%на първа коректура стои сигнатура 134-2-34, поправена на 134-1-34????
%%td_134_1_20 НК

\begin{multicols}{2}
Русе, Русе, тънко Русе, \uu{джанъм},

разшета са тънко Русе, 

разшета са тънко Русе, \uu{джанъм},

ту навънка, ту навкъщи, 
%%чува се "ту" НК

загуби си ресен гердан,
%%чува се "ресен" НК

ресен гердан, с\'{ъ}рмала колан.

Върнала са да го търси,

на път срещна лудо-младо,

набедила лудо-младо:

-- Ти ми найди ресен гердан, (2)

ресен гердан, сърмала колан,

ти го найди, отдай ми го!

Лудо-младо люто са кълне:
\kray

-- 'Ку съм нашел ресен гердан,

да са вия като гердан

окол твойта бяла шийка;

'ку съм нашел сърмала колан

да са вия като колан

окол твойто тънко кръсче!
\end{multicols}

\sing{Пял Ради Иванов Славов}
\year{1911}
\village{Навъсен}
