%Заспало ми й дитенце
%Лазарска (из пътя не пеят, на малко дете) 
%Джиджев:лазарски

\begin{multicols}{2}
Заспало ми й дитенце (2)

под бял-чървен триндафил, (2)

майчица го будиши:  (2)

-- Стани ми, стани, дитенце, (2)

\five да си видиш Лазара, (2)

чи ни Лазар ката ден, (2)

ами й виднъж в година: (2)

като ружа в градина! (2)
\end{multicols}

\area{Пловдивско}
\year{1904}
