01: #include <iostream>
02: #include <vector> 
03: using namespace std; 
04: /**
05:    Returns all values within a range
06:    @param v a vector of floating-point numbers
07:    @param low the low end of the range
08:    @param high the high end of the range
09:    @return a vector of values from v in the given range
10: */
11: vector<double> between(vector<double> v,
12:                  double low, double high)
13: {  vector<double> result;
14:    for (int i = 0; i < v.size(); i++)
15:       if (low <= v[i] && v[i] <= high)
16:          result.push_back(v[i]);
17:    return result;
18: }
