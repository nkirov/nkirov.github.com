 Име

 Популярност (нула по подразбиране)

 Get-функция за име
 Функция, която увеличава популярността с едно.
 Set-функция за най-добрия приятел (приема указател). Тук реализирайте логиката по пресмятане на популярността, като използвате функцията по-горе.
 Get-функция за популярност

#include <iostream>
#include <string>
#include <vector>

using namespace std;

class Person {
private:
    string name;
    Person* bestFriend;
    int popularity;
   
    void increasePopularity() {
        //code here
        popularity++;
    }
   
public:
    Person(string n) {
        //asign member here
        name = n;
        bestFriend = NULL;
        popularity = 0;
    }
   
    void setBestFriend(Person* p)
    {
        //set best friend
        bestFriend = p;
        //use increasePopularity() to calculate popularity
        p->increasePopularity();
    }
   
    Person * getBestFriend()
    {
        return bestFriend;
    }
   
    string getName()
    {
        return name;
    }
    // getters here
   
    int pop()
    {
        return popularity;
    }
};

Person* findPerson(const vector<Person*>& persons, string name)
{
    //code here
    for (int i=0; i<persons.size(); i++)
    {
        if (persons[i]->getName() == name) return persons[i];
    }
    return NULL;
}

void print(const vector<Person*> persons)
{
    for (int i=0; i<persons.size(); i++)
    {
        cout << persons[i]->getName() << " ";
        if (persons[i]->getBestFriend() != NULL)
            cout << persons[i]->getBestFriend()->getName() << " ";
        cout << persons[i]->pop() << endl;
    }
}

{
