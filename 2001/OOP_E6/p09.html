using namespace std;
int main()
{
 double a, b, c;
 cout << "Enter the coeficients a, b and c:";
 cin >> a >> b >> c;
 double det = b*b - 4*a*c;
 if (det >= 0)
 {
  double x1 = (-b+sqrt(det))/(2*a);
  double x2 = (-b-sqrt(det))/(2*a);
  cout << "The roots are: " << x1 << " and " << x2;
 }
 else cout << "The equation has no real roots."

 return 0;
}
