Условие на задача (дава се на отборите):
Даден е масив A с N елемента - естествени числа (0<N<16)и цяло число S.
Да се намерят елементи на масива, чиято сума е равна на S.

Вход:
Първият ред на файла съдържа числата N и S - броя на елементите ан масива
и търсената сума.
Следващият ред съдържа елементите на масива.
Файлът може да съдържа няколко тестови примера със същата структура.
На последния ред на файла е числото 0.

Име на входния файл: input6.txt

Изход:
На всеки пример съответства ред от изходния файл, съдържащ елементите на
масива, чиято сума е S и знак + след всеки елемент.
Ако задачата няма решение, на съответния ред се поставя знака *.

Име на изходния файл: output6.txt

// Примерна програма на С++
// файл sum.cpp
//
#include <iostream>
#include <fstream>
#include <cmath>

int two_p(int n)
{ return static_cast<int>(pow(2,n)); }

unsigned int N;
int A[10]={0};
int SUM;

void count(ofstream &fout)
{
 int NN = two_p(N);
 unsigned int i, j;
 for(i=1; i<NN; i++)
 {
  int sum=0;
  for(j=0; j<N; j++) if (two_p(j) & i) sum+=A[j];
  if (sum==SUM)
  {
   for(j=0; j<N; j++) if (two_p(j) & i) fout << A[j] << "+";
   fout << endl;
   return;
  }
 }
 fout << "*\n";
}
int main()
{
 ifstream fin("input6.txt");
 ofstream fout("output6.txt");
 do
 {
  fin >> N;
  if (N > 0)
  {
   fin >> SUM;
   for (int i=0; i<N; i++) fin >> A[i];
   count(fout);
  }
 }
 while (N > 0);
 return 0;
}

// Примерна програма на С++
// файл equ.cpp
//
#include <iostream>
#include <fstream>

using namespace std;

int seq[10000];
int N;

int find()
{
 int rep=seq[0], num, numm;
 num=numm=1;
 for (int i=1; i<N; i++)
 {
  if (seq[i] == rep) num++;
  else
  {
   if (num>numm) numm=num;
   rep=seq[i]; num=1;
  }
 }
 return numm;
}

int main()
{
 ifstream fin("input7.txt");
 ofstream fout("output7.txt");
 do
 {
  fin >> N;
  if (N>0)
  {
   for (int i=0; i<N; i++) fin >> seq[i];
   fout << find() << "\n";
  }
 }
 while (N>0);
 return 0;
}
