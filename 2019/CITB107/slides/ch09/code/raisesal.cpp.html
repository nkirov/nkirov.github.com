04: /**
05:    Raise all values in a vector by a given percentage.
06:    @param v vector of values
07:    @param p percentage to raise values by
08: */
09: void raise_by_percent(vector<double>& v, double p)
10: {  for (int i = 0; i < v.size(); i++)
11:       v[i] = v[i] * (1 + p / 100);
12: }
