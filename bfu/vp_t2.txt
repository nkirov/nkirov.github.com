int infun(int i1, int &i2)
    { return (++i1-i2++); }

{
  int j, j1, j2;
  cin >> j1 >> j2 ;
  j=infun(j1,j2);
  cout << j << j1 << j2 << "\n";
}

Дадена е следната програма:
#include <iostream>
using namespace std;
int f(int k1, int &k2)
{ k2++; return (k2-k1++); }
void main()
{
  int i1, i2;
  cin >> i1 >> i2 ;
  cout << f(i1,i2) << i1 << i2 << "\n";
}

+ При въвеждане на 0 0 ще се отпечата 101
