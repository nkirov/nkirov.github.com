  struct LocItem {					// a locatable item
    Item<Key,Element>	item;				// the item
    Position		pos;				// its position
    //
    LocItem(const Key& k = Key(), const Element& e = Element())
      : item(k, e), pos() { }		        	// constructor
    const Key& key() const 				// get key
      { return item.key(); }
    Element& element()					// get element
      { return item.element(); }
  };
