  class Position {	                      		// a hash table position
  private:
    EntryPtr node;					// pointer to entry
  public:
    Position(EntryPtr n = NULL)				// constructor
      { node = n; }
    Element& element()  				// get element

    const Key& key() const 				// get key
      { return node->key(); }

  };
